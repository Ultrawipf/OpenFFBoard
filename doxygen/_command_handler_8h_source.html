188 template<typename TVal>
189 static CommandStatus handleGetSet(const ParsedCommand& cmd,std::vector<CommandReply>& replies,TVal& value){
190 if(cmd.type == CMDtype::set){
191 value = static_cast<TVal>(cmd.val);
192 }else if(cmd.type == CMDtype::get){
193 replies.emplace_back(value);
194 }else{
195 return CommandStatus::ERR;
196 }
197 return CommandStatus::OK;
198 }
