86void HidFFB::hidOut(uint8_t report_id, hid_report_type_t report_type, uint8_t const* buffer, uint16_t bufsize){
87 hid_out_period = (micros() - lastOut); // For measuring update rate
88 lastOut = micros();
89 // FFB Output Message
90 const uint8_t* report = buffer;
91 uint8_t event_idx = report_id - FFB_ID_OFFSET;
92
93
94 // -------- Out Reports --------
95 switch(event_idx)
96 {
97
98 case HID_ID_NEWEFREP: //add Effect Report. Feature
99 new_effect((FFB_CreateNewEffect_Feature_Data_t*)(report));
100 break;
101 case HID_ID_EFFREP: // Set Effect
102 set_effect((FFB_SetEffect_t*)(report));
103 break;
104 case HID_ID_CTRLREP: // Control report. 1=Enable Actuators, 2=Disable Actuators, 4=Stop All Effects, 8=Reset, 16=Pause, 32=Continue
105 ffb_control(report[1]);
106 //sendStatusReport(0);
107 break;
108 case HID_ID_GAINREP: // Set global gain
109 set_gain(report[1]);
110 break;
111 case HID_ID_ENVREP: // Envelope
112 set_envelope((FFB_SetEnvelope_Data_t *)report);
113 break;
114 case HID_ID_CONDREP: // Spring, Damper, Friction, Inertia
115 set_condition((FFB_SetCondition_Data_t*)report);
116 break;
117 case HID_ID_PRIDREP: // Periodic
118 set_periodic((FFB_SetPeriodic_Data_t*)report);
119 break;
120 case HID_ID_CONSTREP: // Constant
121 set_constant_effect((FFB_SetConstantForce_Data_t*)report);
122 break;
123 case HID_ID_RAMPREP: // Ramp
124 set_ramp((FFB_SetRamp_Data_t *)report);
125 break;
126 case HID_ID_CSTMREP: // Custom. pretty much never used
127 //printf("Customrep");
128 break;
129 case HID_ID_SMPLREP: // Download sample
130 //printf("Sampledl");
131 break;
132 case HID_ID_EFOPREP: //Effect operation
133 {
134 set_effect_operation((FFB_EffOp_Data_t*)report);
135 break;
136 }
137 case HID_ID_BLKFRREP: // Free a block
138 {
139 free_effect(report[1]-1);
140 break;
141 }
142
143 default:
144 {
145 break;
146 }
147 }
148
149}

169uint16_t HidFFB::hidGet(uint8_t report_id, hid_report_type_t report_type,uint8_t* buffer, uint16_t reqlen){
170 // Feature gets go here
171
172 uint8_t id = report_id - FFB_ID_OFFSET;
173
174 switch(id){
175 case HID_ID_BLKLDREP:
176 //printf("Get Block Report\n");
177 // Notice: first byte ID is not present in the reply buffer because it is handled by tinyusb internally!
178 memcpy(buffer,&this->blockLoad_report,sizeof(FFB_BlockLoad_Feature_Data_t));
179 return sizeof(FFB_BlockLoad_Feature_Data_t);
180 break;
181 case HID_ID_POOLREP:
182 //printf("Get Pool Report\n");
183 memcpy(buffer,&this->pool_report,sizeof(FFB_PIDPool_Feature_Data_t));
184 return sizeof(FFB_PIDPool_Feature_Data_t);
185 break;
186 default:
187 break;
188 }
189 return 0;
190}

217void HidFFB::ffb_control(uint8_t cmd){
218 //printf("Got Control signal: %d\n",cmd);
219 if(cmd & 0x01){ //enable
220 start_FFB();
221 }if(cmd & 0x02){ //disable
222 stop_FFB();
223 }if(cmd & 0x04){ //stop TODO Some games send wrong commands?
224 stop_FFB();
225 //start_FFB();
226 }if(cmd & 0x08){ //reset
227 //ffb_active = true;
228 stop_FFB();
229 reset_ffb();
230 // reset effects
231 }if(cmd & 0x10){ //pause
232 stop_FFB();
233 }if(cmd & 0x20){ //continue
234 start_FFB();
235 }

251void HidFFB::new_effect(FFB_CreateNewEffect_Feature_Data_t* effect){
252 // Allocates a new effect
253
254 uint8_t index = find_free_effect(effect->effectType); // next effect
255 if(index == 0){
256 blockLoad_report.loadStatus = 2;
257 //CommandHandler::logSerial("Can't allocate a new effect");
258 return;
259 }
260 FFB_Effect new_effect;
261 new_effect.type = effect->effectType;
262 this->effects_calc->logEffectType(effect->effectType);
263
264 set_filters(&new_effect);
265
266 effects[index-1] = std::move(new_effect);
267 // Set block load report
268 //reportFFBStatus.effectBlockIndex = index;
269 blockLoad_report.effectBlockIndex = index;
270 used_effects++;
271 blockLoad_report.ramPoolAvailable = (MAX_EFFECTS-used_effects)*sizeof(FFB_Effect);
272 blockLoad_report.loadStatus = 1;
273 sendStatusReport(index);
274

277void HidFFB::set_effect(FFB_SetEffect_t* effect){
278 uint8_t index = effect->effectBlockIndex;
279 if(index > MAX_EFFECTS || index == 0)
280 return;
281
282 FFB_Effect* effect_p = &effects[index-1];
283
284 if (effect_p->type != effect->effectType){
285 effect_p->startTime = 0;
286 set_filters(effect_p);
287 }
288
289 effect_p->gain = effect->gain;
290 effect_p->type = effect->effectType;
291 effect_p->samplePeriod = effect->samplePeriod;
292
293 effect_p->enableAxis = effect->enableAxis;
294 effect_p->directionX = effect->directionX;
295 effect_p->directionY = effect->directionY;
296#if MAX_AXIS == 3
297 effect_p->directionZ = effect->directionZ;
298#endif
299 if(effect_p->duration == 0){ // Fix for games assuming 0 is infinite
300 effect_p->duration = FFB_EFFECT_DURATION_INFINITE;
301 }
302 effect_p->duration = effect->duration;
303 effect_p->startDelay = effect->startDelay;
304 if(!ffb_active)
305 start_FFB();
306 sendStatusReport(effect->effectBlockIndex); // TODO required?
307 //CommandHandler::logSerial("Setting Effect: " + std::to_string(effect->effectType) + " at " + std::to_string(index) + "\n");

320 effect->conditions[axis].negativeCoefficient = cond->negativeCoefficient;
321 effect->conditions[axis].positiveCoefficient = cond->positiveCoefficient;
322 effect->conditions[axis].negativeSaturation = cond->negativeSaturation;
323 effect->conditions[axis].positiveSaturation = cond->positiveSaturation;
324 effect->conditions[axis].deadBand = cond->deadBand;
325 //effect->conditionsCount++;

327 effect->conditions[axis].positiveSaturation = 0x7FFF;
328 }

334void HidFFB::set_effect_operation(FFB_EffOp_Data_t* report){
335 if(report->effectBlockIndex == 0 || report->effectBlockIndex > MAX_EFFECTS){
336 return; // Invalid ID
337 }
338 // Start or stop effect
339 uint8_t id = report->effectBlockIndex-1;
340 if(report->state == 3){
341 effects[id].state = 0; //Stop
342 //CommandHandler::logSerial("Stop" + std::to_string(id));
343 }else{
344
345 // 1 = start, 2 = start solo
346 if(report->state == 2){
347 //CommandHandler::logSerial("Start solo" + std::to_string(id));
348 for(FFB_Effect& effect : effects){
349 effect.state = 0; // Stop all other effects
350 }
351 }
352 if(effects[id].state != 1){
353 set_filters(&effects[id]);
354 }
355
356 //CommandHandler::logSerial("Start" + std::to_string(id));
357 effects[id].startTime = HAL_GetTick() + effects[id].startDelay; // + effects[id].startDelay;
358 effects[id].state = 1; //Start
359
360
361 }
362 //sendStatusReport(report[1]);
363}
