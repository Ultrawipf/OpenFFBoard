71CommandStatus SystemCommands::internalCommand(const ParsedCommand& cmd,std::vector<CommandReply>& replies,CommandInterface* interface){
72 CommandStatus flag = CommandStatus::OK;
73
74 switch((FFBoardMain_commands)cmd.cmdId)
75 {
76 case FFBoardMain_commands::help:
77 {// help
78 if(cmd.type == CMDtype::info){
79 replies.push_back(CommandReply(this->getCsvHelpstring()));
80 }else{
81 std::string reply = interface->getHelpstring() + "\nAvailable classes (use cls.0.help for more info):\n";
82 //std::string reply = "";
83 for(CommandHandler* handler : CommandHandler::cmdHandlers){
84 CmdHandlerInfo* info = handler->getCommandHandlerInfo();
85 reply += std::string(info->clsname) + "." + std::to_string(info->instance)+"\n";
86 }
87 reply += "\n"+this->getCommandsHelpstring();
88 replies.push_back(CommandReply(reply));
89 }
90
91 break;
92 }
93
94 case FFBoardMain_commands::save:
95 for(PersistentStorage* handler : PersistentStorage::flashHandlers){
96 handler->saveFlash();
97 }
98 break;
99
100 case FFBoardMain_commands::reboot:
101 NVIC_SystemReset();
102 break;
103
104 case FFBoardMain_commands::dfu:
105 RebootDFU();
106 break;
107 case FFBoardMain_commands::lsmain:
108 mainchooser.replyAvailableClasses(replies);
109 break;
110
111 case FFBoardMain_commands::vint:
112 {
113 replies.push_back(CommandReply(getIntV()));
114 break;
115 }
116
117 case FFBoardMain_commands::debug:
118 return handleGetSet(cmd, replies, SystemCommands::allowDebugCommands);
119
120 case FFBoardMain_commands::vext:
121 {
122 replies.push_back(CommandReply(getExtV()));
123 break;
124 }
125
126 case FFBoardMain_commands::main:
127 {
128 if(cmd.type == CMDtype::get){
129 uint16_t buf=mainclass->getInfo().id;
130 Flash_Read(ADR_CURRENT_CONFIG, &buf);
131 replies.push_back(CommandReply(buf));
132 }else if(cmd.type == CMDtype::set){
133 if(mainchooser.isValidClassId(cmd.val)){
134 Flash_Write(ADR_CURRENT_CONFIG, (uint16_t)cmd.val);
135 if(cmd.val != mainclass->getInfo().id){
136 NVIC_SystemReset(); // Reboot
137 }
138 }
139 }
140 break;
141 }
142 case FFBoardMain_commands::errors:
143 {
144 replyErrors(replies);
145 break;
146 }
147 case FFBoardMain_commands::flashdump:
148 replyFlashDump(replies);
149 break;
150
151 case FFBoardMain_commands::flashraw:
152 {
153 if(cmd.type == CMDtype::setat){
154 Flash_Write(cmd.adr, cmd.val);
155
156 }else if(cmd.type == CMDtype::getat){
157 CommandReply reply;
158 reply.type = CommandReplyType::INT;
159 uint16_t val;
160 if(Flash_Read(cmd.adr,&val)){
161 reply.val=val;
162 }
163 replies.push_back(reply);
164 }
165 break;
166 }
167
168 case FFBoardMain_commands::errorsclr:
169 ErrorHandler::clearAll();
170 break;
171
172 case FFBoardMain_commands::swver:
173 {
174 CommandReply reply;
175 reply.type = CommandReplyType::STRING_OR_INT;
176
177 extern const uint8_t SW_VERSION_INT[3];
178 reply.reply += std::to_string(SW_VERSION_INT[0]) + "." + std::to_string(SW_VERSION_INT[1]) + "." + std::to_string(SW_VERSION_INT[2]);
179
180 reply.val = (SW_VERSION_INT[0]<<16) | (SW_VERSION_INT[1] << 8) | (SW_VERSION_INT[2]);
181 replies.push_back(reply);
182 break;
183 }
184
185 case FFBoardMain_commands::hwtype:
186 {
187 replies.push_back(CommandReply(HW_TYPE,HW_TYPE_INT));
188 //replies.push_back(CommandReply(HAL_GetDEVID()));
189 break;
190 }
191 case FFBoardMain_commands::devid:
192 {
193 replies.push_back(CommandReply(HAL_GetDEVID(),HAL_GetREVID()));
194 break;
195 }
196
197
198 case FFBoardMain_commands::mallinfo: // UNUSED since freertos
199 {
200 CommandReply reply;
201 struct mallinfo info = mallinfo();
202 reply.adr = info.uordblks;
203 reply.val = info.uordblks;
204 reply.reply +="Usage: ";
205 reply.reply += std::to_string(info.uordblks);
206 reply.reply +=" Size: ";
207 reply.reply +=std::to_string(info.arena);
208 reply.type = CommandReplyType::STRING_OR_DOUBLEINT;
209 replies.push_back(reply);
210 break;
211 }
212
213 case FFBoardMain_commands::heapfree:
214 {
215 replies.push_back(CommandReply(xPortGetFreeHeapSize(),xPortGetMinimumEverFreeHeapSize()));
216 break;
217 }
218#if configUSE_STATS_FORMATTING_FUNCTIONS>0
219 case FFBoardMain_commands::taskstats:
220 {
221 char repl[800];
222 vTaskGetRunTimeStats(repl);
223 replies.push_back(CommandReply("\n"+std::string(repl)));
224 break;
225 }
226#endif
227 case FFBoardMain_commands::lsactive:
228 {
229 for(CommandHandler* handler : CommandHandler::cmdHandlers){
230 if(handler->hasCommands()){
231 ClassIdentifier i = handler->getInfo();
232 CmdHandlerInfo* hi = handler->getCommandHandlerInfo();
233 CommandReply reply;
234 reply.type = CommandReplyType::STRING_OR_DOUBLEINT;
235 reply.adr = hi->instance;
236 reply.val = hi->clsTypeid;
237 reply.reply += std::string(i.name)+ ":" + hi->clsname + ":" + std::to_string(hi->instance) + ":" + std::to_string(i.id) + ":" + std::to_string(hi->commandHandlerID);
238 replies.push_back(reply);
239 }
240 }
241 break;
242 }
243 case FFBoardMain_commands::format:
244 if(cmd.type == CMDtype::set && cmd.val==1){
245 HAL_FLASH_Unlock();
246 EE_Format();
247 HAL_FLASH_Lock();
248 }
249 break;
250
251 default:
252 flag = CommandStatus::NOT_FOUND;
253 break;
254 }
255 return flag;
256}

261void SystemCommands::replyFlashDump(std::vector<CommandReply>& replies){
262 std::vector<std::tuple<uint16_t,uint16_t>> result;
263 Flash_Dump(&result);
264
265 for(auto entry : result){
266 CommandReply reply;
267 uint16_t adr;
268 uint16_t val;
269 std::tie(adr,val) = entry;
270 //reply.reply += std::to_string(adr) + ":" + std::to_string(val) + "\n";
271 reply.adr = adr;
272 reply.val = val;
273 reply.type = CommandReplyType::DOUBLEINTS;
274 replies.push_back(reply);
275 }
276}
