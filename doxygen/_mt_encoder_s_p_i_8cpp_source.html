87void MtEncoderSPI::updateAngleStatus(){
88 uint8_t txbuf[4] = {0x03 | MAGNTEK_READ,0,0,0};
89 uint8_t rxbuf[4] = {0,0,0,0};
90 spiPort.transmitReceive(txbuf, rxbuf, 4, this, 1000);
91
92 uint32_t angle17_10 = rxbuf[1];
93 uint32_t angle9_4 = rxbuf[2];
94 uint32_t angle3_0 = rxbuf[3];
95
96 nomag = (angle9_4 & 0x02) >> 1;
97 overspeed = (angle3_0 & 0x04) >> 2;
98 angle9_4 = (angle9_4 & 0xFC) >> 2;
99 angle3_0 = (angle3_0 & 0xE0) >> 5;
100
101 curAngleInt = (angle3_0) | (angle9_4 << 4) | (angle17_10 << 10);
102
103 if(lastAngleInt < 0x10000 && curAngleInt > 0x30000){ // Underflowed
104 rotations--;
105 }
106 else if(curAngleInt < 0x10000 && lastAngleInt > 0x30000){ // Overflowed
107 rotations++;
108 }
109
110 curPos = rotations * getCpr() + curAngleInt;
111 lastAngleInt = curAngleInt;
112}
