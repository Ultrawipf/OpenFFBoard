101void MtEncoderSPI::updateAngleStatus(){

103 return; // ignore this update

118void MtEncoderSPI::updateAngleStatusCb(){
119 uint8_t angle17_10 = rxbuf[1];
120 uint8_t angle9_4 = rxbuf[2];
121 uint8_t angle3_0 = rxbuf[3];
122
123 nomag = (angle9_4 & 0x02) >> 1;
124 overspeed = (angle3_0 & 0x04) >> 2;
125 angle9_4 = (angle9_4 & 0xFC) >> 2;
126 angle3_0 = (angle3_0 & 0xE0) >> 5;
127
128 curAngleInt = (angle17_10 << 10) | (angle9_4 << 4) | (angle3_0);
129
130 if(lastAngleInt < 0x10000 && curAngleInt > 0x30000){ // Underflowed
131 rotations--;
132 }
133 else if(curAngleInt < 0x10000 && lastAngleInt > 0x30000){ // Overflowed
134 rotations++;
135 }
136
137 curPos = rotations * getCpr() + curAngleInt;
138 lastAngleInt = curAngleInt;
139 this->updateInProgress = false;
140}

160CommandStatus MtEncoderSPI::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
161 switch(static_cast<MtEncoderSPI_commands>(cmd.cmdId)){
162 case MtEncoderSPI_commands::cspin:
163 if(cmd.type==CMDtype::get){

177 }else{
178 return CommandStatus::ERR;
179 }
180 break;
181 default:
182 return CommandStatus::NOT_FOUND;
183 }
