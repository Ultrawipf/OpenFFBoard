79// Handles starting next transfer for >1 byte transfers
80void PCF8574Buttons::Run(){
81 while(true){
82 WaitForNotification();
83 if(lastByte+1 < this->numBytes){
84 // Next transfer
85 lastData = 0;
86 readByteIT(0x20+ ++lastByte, &lastData);
87 }else{
88 lastByte = 0;

122uint8_t PCF8574Buttons::readButtons(uint64_t* buf){
123
124 if(invert){
125 *buf = ~lastButtons;
126 }else{
127 *buf = lastButtons;
128 }
129 *buf &= mask;
130
131
132 // Update
133 if(HAL_GetTick()-lastSuccess > timeout && readingData){
134 readingData = false;
135 port.resetPort();
136 lastSuccess = HAL_GetTick();
137 }
138 if(!port.isTaken() && !readingData){
139 readingData = true;
140 //lastButtons = 0;
141 currentButtons = 0;
142 lastByte = 0;
143 readByteIT(0x20, &lastData); // Read first address
144 }
145 return btnnum;
146}

159void PCF8574Buttons::setBtnNum(uint8_t num){
160 num = clip<uint8_t,uint8_t>(num, 1, 64); // up to 8 PCF8574 can be chained resulting in 64 buttons
161 this->btnnum = num;
162 this->numBytes = 1+((num-1)/8);
163
164 mask = (uint64_t)pow<uint64_t>(2,num)-(uint64_t)1; // Must be done completely in 64 bit!
165 port.resetPort();
