96// Check if endpoint is ready (not busy and not stalled)
97TU_ATTR_ALWAYS_INLINE static inline
98bool usbd_edpt_ready(uint8_t rhport, uint8_t ep_addr)
99{
100 return !usbd_edpt_busy(rhport, ep_addr) && !usbd_edpt_stalled(rhport, ep_addr);
101}
