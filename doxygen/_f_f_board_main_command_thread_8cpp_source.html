43void FFBoardMainCommandThread::Run(){
44 while(true){
45 // Ask command interfaces if new commands are available if woken up
46 for(CommandInterface* itf : CommandInterface::cmdInterfaces){
47 if(itf->hasNewCommands()){
48 itf->getNewCommands(commands);
49 this->executeCommands(commands, itf);
50 commands.clear();
51 if(commands.capacity() > 20)
52 commands.shrink_to_fit();
53 }
54 }
55
56 FFBoardMainCommandThread::threadSem.Take(); // Stop thread again. will be resumed when parser ready
57
58 Delay(1); // Give the scheduler time
59 }
60}

86void FFBoardMainCommandThread::executeCommands(std::vector<ParsedCommand> commands,CommandInterface* commandInterface){
87
88 //cpp_freertos::CriticalSection::SuspendScheduler();
89 for(ParsedCommand& cmd : commands){
90 cmd.originalInterface = commandInterface;
91 this->results.clear();
92
93 CommandResult resultObj;
94 CommandStatus status = CommandStatus::NOT_FOUND;
95 CommandHandler* handler = cmd.target;
96
97 CmdHandlerCommanddef* cmdDef = handler->getCommandFromId(cmd.cmdId);
98 // check flags
99 bool validFlags = cmdDef != nullptr;
100 if(cmdDef->flags & ( CMDFLAG_GET | CMDFLAG_SET | CMDFLAG_INFOSTRING | CMDFLAG_GETADR | CMDFLAG_SETADR) ){
101 // A type flag is preset. check it
102 validFlags = static_cast<uint32_t>(cmd.type) & cmdDef->flags; // type uses the same flag values
103 }
104 if(CommandHandler::isInHandlerList(handler) && validFlags){ // check if pointer is still present in handler list
105 // Call internal commands first
106 status = handler->internalCommand(cmd,resultObj.reply);
107
108 // internal commands did not return anything call regular custom commands
109 if(status == CommandStatus::NOT_FOUND){
110 status = handler->command(cmd,resultObj.reply);
111 }
112
113 }
114 // If status is not no reply append a reply object. If command was not found the reply vector should be empty but the not found flag set
115 if(status != CommandStatus::NO_REPLY){
116 resultObj.handlerId = handler->getCommandHandlerID();
117 resultObj.originalCommand = cmd;
118 resultObj.type = status;
119 resultObj.commandHandler = handler;
120 //this->results.push_back(resultObj);
121 // Not found if result is empty
122 this->results.push_back(resultObj);
123 }
124 if(!this->results.empty()){
125 for(CommandInterface* itf : CommandInterface::cmdInterfaces){
126 // Block until replies are sent
127 uint32_t remainingTime = 100;
128 while(!itf->readyToSend() && --remainingTime){
129 Delay(1);
130 }
131 if(remainingTime)
132 itf->sendReplies(results, commandInterface);
133 }
134 }
135 }
136 //cpp_freertos::CriticalSection::ResumeScheduler();
137}
