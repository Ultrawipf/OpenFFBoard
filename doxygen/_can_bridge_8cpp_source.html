27CanBridge::CanBridge() {
28 this->port = &canport;
29
30 // Set up a filter to receive everything
31 CAN_FilterTypeDef sFilterConfig;
32 sFilterConfig.FilterBank = 0;
33 sFilterConfig.FilterMode = CAN_FILTERMODE_IDMASK;
34 sFilterConfig.FilterScale = CAN_FILTERSCALE_32BIT;
35 sFilterConfig.FilterIdHigh = 0x0000;
36 sFilterConfig.FilterIdLow = 0x0000;
37 sFilterConfig.FilterMaskIdHigh = 0x0000;
38 sFilterConfig.FilterMaskIdLow = 0x0000;
39 sFilterConfig.FilterFIFOAssignment = rxfifo;
40 sFilterConfig.FilterActivation = ENABLE;
41 sFilterConfig.SlaveStartFilterBank = 14;
42
43 txHeader.StdId = 12;
44 txHeader.ExtId = 0;
45 txHeader.RTR = CAN_RTR_DATA;
46 txHeader.IDE = CAN_ID_STD; //Use std id
47 txHeader.DLC = 8; // 8 bytes
48 txHeader.TransmitGlobalTime = DISABLE;
49
50 this->filterId = this->port->addCanFilter(sFilterConfig);
51 conf1.enabled = true;
52
53 //CommandHandler::registerCommands();
54 registerCommand("can", CanBridge_commands::can, "Send a frame or get last received frame");
55 registerCommand("rtr", CanBridge_commands::canrtr, "Send a RTR frame");
56 registerCommand("spd", CanBridge_commands::canspd, "Change or get CAN baud");
57
58 this->port->setSilentMode(false);
59 this->port->takePort();
60}

147void CanBridge::cdcRcv(char* Buf, uint32_t *Len){
148
149 // Not a gvret binary command
150
151 if(*Buf == 0xE7){
152 gvretMode = true;
153 }
154
155 if(gvretMode == false || (*Buf != 0xE7 && *Buf != 0xF1)){ // Not gvret command
156 FFBoardMain::cdcRcv(Buf, Len);
157 return;
158 }
159
160
161 /*
162 * Data format in buf:
163 * 0xE1,0xE1,0xF1,cmd,data...
164 * data = buf+2
165 */
166
167 // Find start marker
168 uint8_t pos = 0;
169 for(uint8_t i = 0; i < *Len-pos; i++){
170 pos = i+1;
171 if((Buf[i]) == 0xF1){
172 break;
173 }
174 }
175
176 std::vector<char> reply;
177 while(pos < *Len){
178
179 uint8_t datalength = *Len;
180 uint8_t* data = (uint8_t*)(Buf+pos+1);
181 uint8_t cmd = *(Buf+pos);
182
183 // find next F1 or end
184 for(uint8_t i = 0; i < *Len-pos; i++){
185 datalength = i;
186 if((data[i]) == 0xF1){
187 break;
188 }
189 }
190 pos = pos+datalength+2; // Move offset for next loop
191
192
193 switch(cmd){
194
195 case(0):
196 { // send can
197 if(*Len < 8)
198 break;
199 uint32_t id = *data | (*(data+1) << 8) | (*(data+2) << 16) | (*(data+3) << 24); // = dat[2] | dat[3] >> 8 | dat[3] >> 16 | dat[4] >> 21
200 uint64_t msg = 0;
201 uint8_t bus = data[4];
202 uint8_t msgLen = data[5];
203 if(bus == 0){
204 memcpy(&msg,data+6,std::min<int>(msgLen, 8)); // Copy variable length buffer
205 sendMessage(id, msg,msgLen);
206 }
207
208 break;
209 }
210 case(1):
211 { // sync. Microseconds since start up LSB to MSB
212 uint32_t time = HAL_GetTick()*1000;//HAL_CAN_GetTxTimestamp(CanHandle, txMailbox); // or use systick and scale.
213 std::vector<char> t = {0xF1,cmd,(char)(time & 0xff), (char)((time >> 8) & 0xff), (char)((time >> 16) & 0xff), (char)((time >> 24) & 0xff)};
214 reply.insert(reply.end(),t.begin(),t.end());
215
216 break;
217 }
218 case(2): // get digital in
219
220 break;
221
222 case(3): // get analog in
223
224 break;
225
226 case(4): // set digital out
227
228 break;
229
230 case(5): // set can config
231 {
232 uint32_t speed = *data | (*(data+1) << 8) | (*(data+2) << 16) | (*(data+3) << 24);
233 speed = speed & 0x1fffffff;
234 uint8_t b3 = *(data+3); // config byte
235 if(b3 & 0x80){
236 conf1.enabled = (b3 & 0x40) != 0;
237 conf1.listenOnly = (b3 & 0x20) != 0;
238 }
239 this->port->setSpeed(speed);
240 break;
241 }
242 case(6): // get can config
243 {
244 std::vector<char> t =
245 { 0xF1,cmd,
246 (char)(conf1.enabled | conf1.listenOnly << 4),(char)(conf1.speed & 0xff), (char)((conf1.speed >> 8) & 0xff),(char)((conf1.speed >> 16) & 0xff),(char)((conf1.speed >> 24) & 0xff),
247 (char)(conf2.enabled | conf2.listenOnly << 4),(char)(conf2.speed & 0xff), (char)((conf2.speed >> 8) & 0xff),(char)((conf2.speed >> 16) & 0xff),(char)((conf2.speed >> 24) & 0xff),
248 };
249 reply.insert(reply.end(),t.begin(),t.end());
250 break;
251 }
252
253
254 case(7): // get device info
255 {
256 std::vector<char> t = {0xF1,cmd,1,1,1,0,0,0};
257 reply.insert(reply.end(),t.begin(),t.end());
258 break;
259 }
260
261
262 case(8): // set single wire mode
263
264 break;
265
266 case(9):{ // keep alive
267 reply.insert(reply.end(),keepalivemsg.begin(),keepalivemsg.end());
268 break;
269 }
270 case(10): // set system
271
272 break;
273
274 case(11): // echo can frame
275
276 break;
277
278 case(12): // Num buses
279 {
280 std::vector<char> t = {0xF1,cmd,numBuses};
281 reply.insert(reply.end(),t.begin(),t.end());
282 }
283 break;
284 case(13): // ext buses
285 break;
286 case(14): // set ext buses
287 break;
288 default:
289 // unknown command
290 break;
291 }
292
293 }
294 tud_cdc_n_write(0,reply.data(), reply.size());
295 tud_cdc_write_flush();

300CommandStatus CanBridge::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
301
302 switch(static_cast<CanBridge_commands>(cmd.cmdId)){
303 case CanBridge_commands::can:
304 if(cmd.type == CMDtype::get){
305 replies.emplace_back(*(int64_t*)lastmsg.data,lastmsg.header.StdId);
306 }else if(cmd.type == CMDtype::setat){
307 sendMessage(cmd.adr,cmd.val);
308 }else{
309 return CommandStatus::ERR;
310 }
311 break;
312
313
314 case CanBridge_commands::canrtr:
315 if(cmd.type == CMDtype::get){
316 replies.emplace_back(*(int64_t*)lastmsg.data,lastmsg.header.StdId);
317 }else if(cmd.type == CMDtype::setat){
318 sendMessage(cmd.adr,cmd.val,8,true); // msg with rtr bit
319 }else{
320 return CommandStatus::ERR;
321 }
322 break;
323
324 case CanBridge_commands::canspd:
325 if(cmd.type == CMDtype::get){
326 replies.emplace_back(this->port->getSpeed());
327 }else if(cmd.type == CMDtype::set){
328 this->port->setSpeed(cmd.val);
329 }
330 break;
331
332 default:
333 return CommandStatus::NOT_FOUND;
334 }
335
336 return CommandStatus::OK; // Valid command found
337}
