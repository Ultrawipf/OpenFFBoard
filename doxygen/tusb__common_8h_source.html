85static inline uint32_t tu_u32(uint8_t b1, uint8_t b2, uint8_t b3, uint8_t b4)
86{
87 return ( ((uint32_t) b1) << 24) + ( ((uint32_t) b2) << 16) + ( ((uint32_t) b3) << 8) + b4;
88}

90static inline uint16_t tu_u16(uint8_t high, uint8_t low)
91{
92 return (uint16_t)((((uint16_t) high) << 8) + low);
93}

149 * - _GET_NTH_ARG() takes args >= N (64) but only expand to Nth one (64th)
150 * - _RSEQ_N() is reverse sequential to N to add padding to have
151 * Nth position is the same as the number of arguments
152 * - ##__VA_ARGS__ is used to deal with 0 paramerter (swallows comma)
153 *------------------------------------------------------------------*/

164 _41,_42,_43,_44,_45,_46,_47,_48,_49,_50, \
165 _51,_52,_53,_54,_55,_56,_57,_58,_59,_60, \

257typedef struct

263static inline const char* tu_lookup_find(tu_lookup_table_t const* p_table, uint32_t key)
264{
265 for(uint16_t i=0; i<p_table->count; i++)
266 {
267 if (p_table->items[i].key == key) return p_table->items[i].data;
268 }
269
270 return NULL;
271}
