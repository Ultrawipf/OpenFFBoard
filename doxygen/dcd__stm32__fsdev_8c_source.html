31/**********************************************
32 * This driver has been tested with the following MCUs:
33 * - F070, F072, L053, F042F6
34 *
35 * It also should work with minimal changes for any ST MCU with an "USB A"/"PCD"/"HCD" peripheral. This
36 * covers:
37 *
38 * F04x, F072, F078, 070x6/B 1024 byte buffer
39 * F102, F103 512 byte buffer; no internal D+ pull-up (maybe many more changes?)
40 * F302xB/C, F303xB/C, F373 512 byte buffer; no internal D+ pull-up
41 * F302x6/8, F302xD/E2, F303xD/E 1024 byte buffer; no internal D+ pull-up
42 * L0x2, L0x3 1024 byte buffer
43 * L1 512 byte buffer
44 * L4x2, L4x3 1024 byte buffer
45 *
46 * To use this driver, you must:
47 * - If you are using a device with crystal-less USB, set up the clock recovery system (CRS)
48 * - Remap pins to be D+/D- on devices that they are shared (for example: F042Fx)
49 * - This is different to the normal "alternate function" GPIO interface, needs to go through SYSCFG->CFGRx register
50 * - Enable USB clock; Perhaps use __HAL_RCC_USB_CLK_ENABLE();
51 * - (Optionally configure GPIO HAL to tell it the USB driver is using the USB pins)
52 * - call tusb_init();
53 * - periodically call tusb_task();
54 *
55 * Assumptions of the driver:
56 * - You are not using CAN (it must share the packet buffer)
57 * - APB clock is >= 10 MHz
58 * - On some boards, series resistors are required, but not on others.
59 * - On some boards, D+ pull up resistor (1.5kohm) is required, but not on others.
60 * - You don't have long-running interrupts; some USB packets must be quickly responded to.
61 * - You have the ST CMSIS library linked into the project. HAL is not used.
62 *
63 * Current driver limitations (i.e., a list of features for you to add):
64 * - STALL handled, but not tested.
65 * - Does it work? No clue.
66 * - All EP BTABLE buffers are created based on max packet size of first EP opened with that address.
67 * - No isochronous endpoints
68 * - Endpoint index is the ID of the endpoint
69 * - This means that priority is given to endpoints with lower ID numbers
70 * - Code is mixing up EP IX with EP ID. Everywhere.
71 * - Packet buffer memory is copied in the interrupt.
72 * - This is better for performance, but means interrupts are disabled for longer
73 * - DMA may be the best choice, but it could also be pushed to the USBD task.
74 * - No double-buffering
75 * - No DMA
76 * - Minimal error handling
77 * - Perhaps error interrupts should be reported to the stack, or cause a device reset?
78 * - Assumes a single USB peripheral; I think that no hardware has multiple so this is fine.
79 * - Add a callback for enabling/disabling the D+ PU on devices without an internal PU.
80 * - F3 models use three separate interrupts. I think we could only use the LP interrupt for
81 * everything? However, the interrupts are configurable so the DisableInt and EnableInt
82 * below functions could be adjusting the wrong interrupts (if they had been reconfigured)
83 * - LPM is not used correctly, or at all?
84 *
85 * USB documentation and Reference implementations
86 * - STM32 Reference manuals
87 * - STM32 USB Hardware Guidelines AN4879
88 *
89 * - STM32 HAL (much of this driver is based on this)
90 * - libopencm3/lib/stm32/common/st_usbfs_core.c
91 * - Keil USB Device http://www.keil.com/pack/doc/mw/USB/html/group__usbd.html
92 *
93 * - YouTube OpenTechLab 011; https://www.youtube.com/watch?v=4FOkJLp_PUw
94 *
95 * Advantages over HAL driver:
96 * - Tiny (saves RAM, assumes a single USB peripheral)
97 *
98 * Notes:
99 * - The buffer table is allocated as endpoints are opened. The allocation is only
100 * cleared when the device is reset. This may be bad if the USB device needs
101 * to be reconfigured.
102 */

213void dcd_init (uint8_t rhport)
214{
215 /* Clocks should already be enabled */
216 /* Use __HAL_RCC_USB_CLK_ENABLE(); to enable the clocks before calling this function */
217
218 /* The RM mentions to use a special ordering of PDWN and FRES, but this isn't done in HAL.
219 * Here, the RM is followed. */
220
221 for(uint32_t i = 0; i<200; i++) // should be a few us
222 {
223 asm("NOP");
224 }
225 // Perform USB peripheral reset
226 USB->CNTR = USB_CNTR_FRES | USB_CNTR_PDWN;
227 for(uint32_t i = 0; i<200; i++) // should be a few us
228 {
229 asm("NOP");
230 }
231 reg16_clear_bits(&USB->CNTR, USB_CNTR_PDWN);// Remove powerdown
232 // Wait startup time, for F042 and F070, this is <= 1 us.
233 for(uint32_t i = 0; i<200; i++) // should be a few us
234 {
235 asm("NOP");
236 }
237 USB->CNTR = 0; // Enable USB
238
239 USB->BTABLE = DCD_STM32_BTABLE_BASE;
240
241 USB->ISTR = 0; // Clear pending interrupts
242
243 // Reset endpoints to disabled
244 for(uint32_t i=0; i<STFSDEV_EP_COUNT; i++)
245 {
246 // This doesn't clear all bits since some bits are "toggle", but does set the type to DISABLED.
247 pcd_set_endpoint(USB,i,0u);
248 }
249
250 USB->CNTR |= USB_CNTR_RESETM | (USE_SOF ? USB_CNTR_SOFM : 0) | USB_CNTR_ESOFM | USB_CNTR_CTRM | USB_CNTR_SUSPM | USB_CNTR_WKUPM;
251 dcd_handle_bus_reset();
252
253 // Enable pull-up if supported
254 if ( dcd_connect ) dcd_connect(rhport);
255}

290// Enable device interrupt
291void dcd_int_enable (uint8_t rhport)
292{
293 (void)rhport;
294 // Member here forces write to RAM before allowing ISR to execute
295 __DSB();
296 __ISB();
297#if CFG_TUSB_MCU == OPT_MCU_STM32F0 || CFG_TUSB_MCU == OPT_MCU_STM32L0
298 NVIC_EnableIRQ(USB_IRQn);
299
300#elif CFG_TUSB_MCU == OPT_MCU_STM32L1
301 NVIC_EnableIRQ(USB_LP_IRQn);
302
303#elif CFG_TUSB_MCU == OPT_MCU_STM32F3
304 // Some STM32F302/F303 devices allow to remap the USB interrupt vectors from
305 // shared USB/CAN IRQs to separate CAN and USB IRQs.
306 // This dynamically checks if this remap is active to enable the right IRQs.
307 #ifdef SYSCFG_CFGR1_USB_IT_RMP

309 {
310 NVIC_EnableIRQ(USB_HP_IRQn);
311 NVIC_EnableIRQ(USB_LP_IRQn);
312 NVIC_EnableIRQ(USBWakeUp_RMP_IRQn);
313 }
314 else
315 #endif
316 {
317 NVIC_EnableIRQ(USB_HP_CAN_TX_IRQn);
318 NVIC_EnableIRQ(USB_LP_CAN_RX0_IRQn);
319 NVIC_EnableIRQ(USBWakeUp_IRQn);
320 }
321#elif CFG_TUSB_MCU == OPT_MCU_STM32F1
322 NVIC_EnableIRQ(USB_HP_CAN1_TX_IRQn);
323 NVIC_EnableIRQ(USB_LP_CAN1_RX0_IRQn);
324 NVIC_EnableIRQ(USBWakeUp_IRQn);
325
326#elif CFG_TUSB_MCU == OPT_MCU_STM32G4
327 NVIC_EnableIRQ(USB_HP_IRQn);
328 NVIC_EnableIRQ(USB_LP_IRQn);
329 NVIC_EnableIRQ(USBWakeUp_IRQn);

331#else

423static void dcd_handle_bus_reset(void)
424{
425 //__IO uint16_t * const epreg = &(EPREG(0));
426 USB->DADDR = 0u; // disable USB peripheral by clearing the EF flag
427
428 // Clear all EPREG (or maybe this is automatic? I'm not sure)
429 for(uint32_t i=0; i<STFSDEV_EP_COUNT; i++)
430 {
431 pcd_set_endpoint(USB,i,0u);
432 }
433
434 dcd_pma_alloc_reset();
435 dcd_edpt_open (0, &ep0OUT_desc);
436 dcd_edpt_open (0, &ep0IN_desc);
437
438 USB->DADDR = USB_DADDR_EF; // Set enable flag, and leaving the device address as zero.
439}

441// Handle CTR interrupt for the TX/IN direction
442//
443// Upon call, (wIstr & USB_ISTR_DIR) == 0U
444static void dcd_ep_ctr_tx_handler(uint32_t wIstr)
445{
446 uint32_t EPindex = wIstr & USB_ISTR_EP_ID;
447 uint32_t wEPRegVal = pcd_get_endpoint(USB, EPindex);
448
449 // Verify the CTR_TX bit is set. This was in the ST Micro code,
450 // but I'm not sure it's actually necessary?
451 if((wEPRegVal & USB_EP_CTR_TX) == 0U)
452 {
453 return;
454 }
455
456 /* clear int flag */
457 pcd_clear_tx_ep_ctr(USB, EPindex);
458
459 xfer_ctl_t * xfer = xfer_ctl_ptr(EPindex,TUSB_DIR_IN);
460 if((xfer->total_len != xfer->queued_len)) /* TX not complete */
461 {
462 dcd_transmit_packet(xfer, EPindex);
463 }
464 else /* TX Complete */
465 {
466 dcd_event_xfer_complete(0, (uint8_t)(0x80 + EPindex), xfer->total_len, XFER_RESULT_SUCCESS, true);
467 }
468}

470// Handle CTR interrupt for the RX/OUT direction
471//
472// Upon call, (wIstr & USB_ISTR_DIR) == 0U
473static void dcd_ep_ctr_rx_handler(uint32_t wIstr)
474{
475 uint32_t EPindex = wIstr & USB_ISTR_EP_ID;
476 uint32_t wEPRegVal = pcd_get_endpoint(USB, EPindex);
477 uint32_t count = pcd_get_ep_rx_cnt(USB,EPindex);
478
479 xfer_ctl_t *xfer = xfer_ctl_ptr(EPindex,TUSB_DIR_OUT);
480
481 // Verify the CTR_RX bit is set. This was in the ST Micro code,
482 // but I'm not sure it's actually necessary?
483 if((wEPRegVal & USB_EP_CTR_RX) == 0U)
484 {
485 return;
486 }
487
488 if((EPindex == 0U) && ((wEPRegVal & USB_EP_SETUP) != 0U)) /* Setup packet */
489 {
490 // The setup_received function uses memcpy, so this must first copy the setup data into
491 // user memory, to allow for the 32-bit access that memcpy performs.
492 uint8_t userMemBuf[8];
493 /* Get SETUP Packet*/
494 if(count == 8) // Setup packet should always be 8 bytes. If not, ignore it, and try again.
495 {
496 // Must reset EP to NAK (in case it had been stalling) (though, maybe too late here)
497 pcd_set_ep_rx_status(USB,0u,USB_EP_RX_NAK);
498 pcd_set_ep_tx_status(USB,0u,USB_EP_TX_NAK);
499 dcd_read_packet_memory(userMemBuf, *pcd_ep_rx_address_ptr(USB,EPindex), 8);
500 dcd_event_setup_received(0, (uint8_t*)userMemBuf, true);
501 }
502 }
503 else
504 {
505 // Clear RX CTR interrupt flag
506 if(EPindex != 0u)
507 {
508 pcd_clear_rx_ep_ctr(USB, EPindex);
509 }
510
511 if (count != 0U)
512 {
513#if 0 // TODO support dcd_edpt_xfer_fifo API
514 if (xfer->ff)
515 {
516 dcd_read_packet_memory_ff(xfer->ff, *pcd_ep_rx_address_ptr(USB,EPindex), count);
517 }
518 else
519#endif
520 {
521 dcd_read_packet_memory(&(xfer->buffer[xfer->queued_len]), *pcd_ep_rx_address_ptr(USB,EPindex), count);
522 }
523
524 xfer->queued_len = (uint16_t)(xfer->queued_len + count);
525 }
526
527 if ((count < xfer->max_packet_size) || (xfer->queued_len == xfer->total_len))
528 {
529 /* RX COMPLETE */
530 dcd_event_xfer_complete(0, EPindex, xfer->queued_len, XFER_RESULT_SUCCESS, true);
531 // Though the host could still send, we don't know.
532 // Does the bulk pipe need to be reset to valid to allow for a ZLP?
533 }
534 else
535 {
536 uint32_t remaining = (uint32_t)xfer->total_len - (uint32_t)xfer->queued_len;
537 if(remaining >= xfer->max_packet_size) {
538 pcd_set_ep_rx_cnt(USB, EPindex,xfer->max_packet_size);
539 } else {
540 pcd_set_ep_rx_cnt(USB, EPindex,remaining);
541 }
542 pcd_set_ep_rx_status(USB, EPindex, USB_EP_RX_VALID);
543 }
544 }
545
546 // For EP0, prepare to receive another SETUP packet.
547 // Clear CTR last so that a new packet does not overwrite the packing being read.
548 // (Based on the docs, it seems SETUP will always be accepted after CTR is cleared)
549 if(EPindex == 0u)
550 {
551 // Always be prepared for a status packet...
552 pcd_set_ep_rx_cnt(USB, EPindex, CFG_TUD_ENDPOINT0_SIZE);
553 pcd_clear_rx_ep_ctr(USB, EPindex);
554 }
555}

557static void dcd_ep_ctr_handler(void)
558{
559 uint32_t wIstr;
560
561 /* stay in loop while pending interrupts */
562 while (((wIstr = USB->ISTR) & USB_ISTR_CTR) != 0U)
563 {
564
565 if ((wIstr & USB_ISTR_DIR) == 0U) /* TX/IN */
566 {
567 dcd_ep_ctr_tx_handler(wIstr);
568 }
569 else /* RX/OUT*/
570 {
571 dcd_ep_ctr_rx_handler(wIstr);
572 }
573 }
574}

576void dcd_int_handler(uint8_t rhport) {
577
578 (void) rhport;
579
580 uint32_t int_status = USB->ISTR;
581 //const uint32_t handled_ints = USB_ISTR_CTR | USB_ISTR_RESET | USB_ISTR_WKUP
582 // | USB_ISTR_SUSP | USB_ISTR_SOF | USB_ISTR_ESOF;
583 // unused IRQs: (USB_ISTR_PMAOVR | USB_ISTR_ERR | USB_ISTR_L1REQ )
584
585 // The ST driver loops here on the CTR bit, but that loop has been moved into the
586 // dcd_ep_ctr_handler(), so less need to loop here. The other interrupts shouldn't
587 // be triggered repeatedly.
588
589 if(int_status & USB_ISTR_RESET) {
590 // USBRST is start of reset.
591 clear_istr_bits(USB_ISTR_RESET);
592 dcd_handle_bus_reset();
593 dcd_event_bus_reset(0, TUSB_SPEED_FULL, true);
594 return; // Don't do the rest of the things here; perhaps they've been cleared?
595 }
596
597 if (int_status & USB_ISTR_CTR)
598 {
599 /* servicing of the endpoint correct transfer interrupt */
600 /* clear of the CTR flag into the sub */
601 dcd_ep_ctr_handler();
602 }
603
604 if (int_status & USB_ISTR_WKUP)
605 {
606 reg16_clear_bits(&USB->CNTR, USB_CNTR_LPMODE);
607 reg16_clear_bits(&USB->CNTR, USB_CNTR_FSUSP);
608 clear_istr_bits(USB_ISTR_WKUP);
609 dcd_event_bus_signal(0, DCD_EVENT_RESUME, true);
610 }
611
612 if (int_status & USB_ISTR_SUSP)
613 {
614 /* Suspend is asserted for both suspend and unplug events. without Vbus monitoring,
615 * these events cannot be differentiated, so we only trigger suspend. */
616
617 /* Force low-power mode in the macrocell */
618 USB->CNTR |= USB_CNTR_FSUSP;
619 USB->CNTR |= USB_CNTR_LPMODE;
620
621 /* clear of the ISTR bit must be done after setting of CNTR_FSUSP */
622 clear_istr_bits(USB_ISTR_SUSP);
623 dcd_event_bus_signal(0, DCD_EVENT_SUSPEND, true);
624 }
625
626#if USE_SOF
627 if(int_status & USB_ISTR_SOF) {
628 clear_istr_bits(USB_ISTR_SOF);
629 dcd_event_bus_signal(0, DCD_EVENT_SOF, true);
630 }
631#endif
632
633 if(int_status & USB_ISTR_ESOF) {
634 if(remoteWakeCountdown == 1u)
635 {
636 USB->CNTR &= (uint16_t)(~USB_CNTR_RESUME);
637 }
638 if(remoteWakeCountdown > 0u)
639 {
640 remoteWakeCountdown--;
641 }
642 clear_istr_bits(USB_ISTR_ESOF);
643 }
644}

681/***
682 * Allocate a section of PMA
683 *
684 * If the EP number has already been allocated, and the new allocation
685 * is larger than the old allocation, then this will fail with a TU_ASSERT.
686 * (This is done to simplify the code. More complicated algorithms could be used)
687 *
688 * During failure, TU_ASSERT is used. If this happens, rework/reallocate memory manually.
689 */
690static uint16_t dcd_pma_alloc(uint8_t ep_addr, size_t length)
691{
692 uint8_t const epnum = tu_edpt_number(ep_addr);
693 uint8_t const dir = tu_edpt_dir(ep_addr);
694 xfer_ctl_t* epXferCtl = xfer_ctl_ptr(epnum,dir);
695
696 if(epXferCtl->pma_alloc_size != 0U)
697 {
698 //TU_LOG2("dcd_pma_alloc(%x,%x)=%x (cached)\r\n",ep_addr,length,epXferCtl->pma_ptr);
699 // Previously allocated
700 TU_ASSERT(length <= epXferCtl->pma_alloc_size, 0xFFFF); // Verify no larger than previous alloc
701 return epXferCtl->pma_ptr;
702 }
703
704 uint16_t addr = ep_buf_ptr;
705 ep_buf_ptr = (uint16_t)(ep_buf_ptr + length); // increment buffer pointer
706
707 // Verify no overflow
708 TU_ASSERT(ep_buf_ptr <= PMA_LENGTH, 0xFFFF);
709
710 epXferCtl->pma_ptr = addr;
711 epXferCtl->pma_alloc_size = length;
712 //TU_LOG2("dcd_pma_alloc(%x,%x)=%x\r\n",ep_addr,length,addr);
713
714 return addr;
715}

718 * Free a block of PMA space
719 */

721{
722 uint8_t const epnum = tu_edpt_number(ep_addr);
723 uint8_t const dir = tu_edpt_dir(ep_addr);
724
725 // Presently, this should never be called for EP0 IN/OUT

731

737 for(uint32_t i=1; i<MAX_EP_COUNT; i++)
738 {
739 xfer_ctl_ptr(i,TUSB_DIR_OUT)->pma_alloc_size = 0U;
740 xfer_ctl_ptr(i,TUSB_DIR_IN)->pma_alloc_size = 0U;

742 xfer_ctl_ptr(i,TUSB_DIR_IN)->pma_ptr = 0U;

979// Packet buffer access can only be 8- or 16-bit.
990static bool dcd_write_packet_memory(uint16_t dst, const void *__restrict src, size_t wNBytes)
991{
992 uint32_t n = ((uint32_t)wNBytes + 1U) >> 1U;
993 uint32_t i;
994 uint16_t temp1, temp2;
995 const uint8_t * srcVal;
996
997 // The GCC optimizer will combine access to 32-bit sizes if we let it. Force
998 // it volatile so that it won't do that.
999 __IO uint16_t *pdwVal;
1000
1001 srcVal = src;
1002 pdwVal = &pma[PMA_STRIDE*(dst>>1)];
1003
1004 for (i = n; i != 0; i--)
1005 {
1006 temp1 = (uint16_t) *srcVal;
1007 srcVal++;
1008 temp2 = temp1 | ((uint16_t)((uint16_t) ((*srcVal) << 8U))) ;
1009 *pdwVal = temp2;
1010 pdwVal += PMA_STRIDE;
1011 srcVal++;
1012 }
1013 return true;
1014}
