91//--------------------------------------------------------------------+
92// Control Endpoint API
93//--------------------------------------------------------------------+
94
95// Get current protocol: HID_PROTOCOL_BOOT (0) or HID_PROTOCOL_REPORT (1)
96// Note: Device will be initialized in Boot protocol for simplicity.
97// Application can use set_protocol() to switch back to Report protocol.
98uint8_t tuh_hid_get_protocol(uint8_t dev_addr, uint8_t idx);
99
100// Device by default is enumerated in Boot protocol for simplicity. Application
101// can use this to modify the default protocol for next enumeration.
102void tuh_hid_set_default_protocol(uint8_t protocol);
103
104// Set protocol to HID_PROTOCOL_BOOT (0) or HID_PROTOCOL_REPORT (1)
105// This function is only supported by Boot interface (tuh_n_hid_interface_protocol() != NONE)
106bool tuh_hid_set_protocol(uint8_t dev_addr, uint8_t idx, uint8_t protocol);
107
108// Get Report using control endpoint
109// report_type is either Input, Output or Feature, (value from hid_report_type_t)
110bool tuh_hid_get_report(uint8_t dev_addr, uint8_t idx, uint8_t report_id, uint8_t report_type, void* report, uint16_t len);
111
112// Set Report using control endpoint
113// report_type is either Input, Output or Feature, (value from hid_report_type_t)
114bool tuh_hid_set_report(uint8_t dev_addr, uint8_t idx, uint8_t report_id, uint8_t report_type,
115 void* report, uint16_t len);
116
117//--------------------------------------------------------------------+
118// Interrupt Endpoint API
119//--------------------------------------------------------------------+
120
121// Check if HID interface is ready to receive report
122bool tuh_hid_receive_ready(uint8_t dev_addr, uint8_t idx);
123
124// Try to receive next report on Interrupt Endpoint. Immediately return
125// - true If succeeded, tuh_hid_report_received_cb() callback will be invoked when report is available
126// - false if failed to queue the transfer e.g endpoint is busy
127bool tuh_hid_receive_report(uint8_t dev_addr, uint8_t idx);
128
129// Abort receiving report on Interrupt Endpoint
130bool tuh_hid_receive_abort(uint8_t dev_addr, uint8_t idx);
131
132// Check if HID interface is ready to send report
133bool tuh_hid_send_ready(uint8_t dev_addr, uint8_t idx);
134
135// Send report using interrupt endpoint
136// If report_id > 0 (composite), it will be sent as 1st byte, then report contents. Otherwise only report content is sent.
137bool tuh_hid_send_report(uint8_t dev_addr, uint8_t idx, uint8_t report_id, const void* report, uint16_t len);
138
139//--------------------------------------------------------------------+
140// Callbacks (Weak is optional)
141//--------------------------------------------------------------------+
142
143// Invoked when device with hid interface is mounted
144// Report descriptor is also available for use. tuh_hid_parse_report_descriptor()
145// can be used to parse common/simple enough descriptor.
146// Note: if report descriptor length > CFG_TUH_ENUMERATION_BUFSIZE, it will be skipped
147// therefore report_desc = NULL, desc_len = 0
148TU_ATTR_WEAK void tuh_hid_mount_cb(uint8_t dev_addr, uint8_t idx, uint8_t const* report_desc, uint16_t desc_len);
149
150// Invoked when device with hid interface is un-mounted
151TU_ATTR_WEAK void tuh_hid_umount_cb(uint8_t dev_addr, uint8_t idx);
152
153// Invoked when received report from device via interrupt endpoint
154// Note: if there is report ID (composite), it is 1st byte of report
155void tuh_hid_report_received_cb(uint8_t dev_addr, uint8_t idx, uint8_t const* report, uint16_t len);
156
157// Invoked when sent report to device successfully via interrupt endpoint
158TU_ATTR_WEAK void tuh_hid_report_sent_cb(uint8_t dev_addr, uint8_t idx, uint8_t const* report, uint16_t len);
159
160// Invoked when Get Report to device via either control endpoint
161// len = 0 indicate there is error in the transfer e.g stalled response
162TU_ATTR_WEAK void tuh_hid_get_report_complete_cb(uint8_t dev_addr, uint8_t idx, uint8_t report_id, uint8_t report_type, uint16_t len);
163
164// Invoked when Sent Report to device via either control endpoint
165// len = 0 indicate there is error in the transfer e.g stalled response
166TU_ATTR_WEAK void tuh_hid_set_report_complete_cb(uint8_t dev_addr, uint8_t idx, uint8_t report_id, uint8_t report_type, uint16_t len);
167
168// Invoked when Set Protocol request is complete
169TU_ATTR_WEAK void tuh_hid_set_protocol_complete_cb(uint8_t dev_addr, uint8_t idx, uint8_t protocol);
