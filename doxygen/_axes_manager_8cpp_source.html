101bool AxesManager::setAxisCount(int8_t count) {
102 if (!this->validAxisRange(count)) {
103 return false; // invalid number of axis
104 }
105 // Really need to use some form of mutex
106 Flash_Write(ADR_AXIS_COUNT, count);
107
108 while (count < axis_count) {
109 uint8_t pos = axis_count-1;
110 axes.erase(axes.begin()+pos);
111 axis_count--;
112 }
113 while (count > axis_count) {
114 axes.push_back( std::make_unique<Axis>('X'+axis_count, control) ); // Axis are indexed from 1-X to 3-Z
115 axis_count++;
116 }
117 control->update_disabled = false;
118
119 // Allocate axis value buffer
120 this->axisValues.reserve(axis_count);
121 //this->axisValues = std::vector<int32_t>(axis_count,0);
122
123 return true;
124}
