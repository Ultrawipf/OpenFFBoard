85/*
86If the metric is less than CP Offset - Dead Band, then the resulting force is given by the following formula:
87 force = Negative Coefficient * (q - (CP Offset â€“ Dead Band))
88Similarly, if the metric is greater than CP Offset + Dead Band, then the resulting force is given by the
89following formula:
90 force = Positive Coefficient * (q - (CP Offset + Dead Band))
91A spring condition uses axis position as the metric.
92A damper condition uses axis velocity as the metric.
93An inertia condition uses axis acceleration as the metric.
94
95If the number of Condition report blocks is equal to the number of axes for the effect, then the first report
96block applies to the first axis, the second applies to the second axis, and so on. For example, a two-axis
97spring condition with CP Offset set to zero in both Condition report blocks would have the same effect as
98the joystick self-centering spring. When a condition is defined for each axis in this way, the effect must
99not be rotated.
100
101If there is a single Condition report block for an effect with more than one axis, then the direction along
102which the parameters of the Condition report block are in effect is determined by the direction parameters
103passed in the Direction field of the Effect report block. For example, a friction condition rotated 45
104degrees (in polar coordinates) would resist joystick motion in the northeast-southwest direction but would
105have no effect on joystick motion in the northwest-southeast direction.
106 */

108/*
109 * Calculates the resulting torque for FFB effects
110 * Takes current position input scaled from -0x7fff to 0x7fff
111 * Outputs a torque value from -0x7fff to 0x7fff (not yet clipped)
112 */
113void EffectsCalculator::calculateEffects(std::vector<std::unique_ptr<Axis>> &axes)
114{
115 for (auto &axis : axes) {
116 axis->calculateAxisEffects(isActive());
117 }
118
119 if(!isActive()){
120 return;
121 }
122
123 int32_t forceX = 0;
124 int32_t forceY = 0;
125 int32_t forceVector = 0;
126 uint8_t axisCount = axes.size();
127 bool validY = axisCount > 1;
128#if MAX_AXIS == 3
129 int32_t forceZ = 0;
130 bool validZ = axisCount > 2;
131#endif
132
133 for (uint8_t i = 0; i < effects_stats.size(); i++)
134 {
135 effects_stats[i].current = 0; // Reset active effect forces
136 }
137
138
139 for (uint8_t i = 0; i < MAX_EFFECTS; i++)
140 {
141 FFB_Effect *effect = &effects[i];
142
143 // Effect activated and not infinite (0 or 0xffff)
144 if (effect->state != EFFECT_STATE_INACTIVE && effect->duration != FFB_EFFECT_DURATION_INFINITE && effect->duration != 0){
145 // Start delay not yet reached
146 if(HAL_GetTick() < effect->startTime){
147 continue;
148 }
149 // If effect has expired make inactive
150 if (HAL_GetTick() > effect->startTime + effect->duration)
151 {
152 effect->state = EFFECT_STATE_INACTIVE;
153 calcStatsEffectType(effect->type, 0); // record a 0 on the ended force
154 }
155 }
156
157 // Filter out inactive effects
158 if (effect->state == EFFECT_STATE_INACTIVE)
159 {
160 continue;
161 }
162
163
164 //if (effect->conditionsCount == 0) {
165 forceVector = calcNonConditionEffectForce(effect);
166 //}
167
168 uint8_t directionEnableMask = this->directionEnableMask ? this->directionEnableMask : DIRECTION_ENABLE(axisCount);
169
170 if ( (effect->enableAxis & directionEnableMask) || (effect->enableAxis & X_AXIS_ENABLE))
171 {
172 int32_t newEffectForce = calcComponentForce(effect, forceVector, axes, 0);
173 calcStatsEffectType(effect->type, newEffectForce);
174 forceX += newEffectForce;
175 forceX = clip<int32_t, int32_t>(forceX, -0x7fff, 0x7fff); // Clip
176 }
177 if (validY && ((effect->enableAxis & directionEnableMask) || (effect->enableAxis & Y_AXIS_ENABLE)))
178 {
179 int32_t newEffectForce = calcComponentForce(effect, forceVector, axes, 0);
180 calcStatsEffectType(effect->type, newEffectForce);
181 forceY += newEffectForce;
182 forceY = clip<int32_t, int32_t>(forceY, -0x7fff, 0x7fff); // Clip
183 }
184
185 }
186
187 axes[0]->setEffectTorque(forceX);
188 if (validY)
189 {
190 axes[1]->setEffectTorque(forceY);
191 }
192}

198int32_t EffectsCalculator::calcNonConditionEffectForce(FFB_Effect *effect) {
199 int32_t force_vector = 0;
200 int32_t magnitude = effect->magnitude;
201
202 // If using an envelope modulate the magnitude based on time
203 if(effect->useEnvelope){
204 magnitude = getEnvelopeMagnitude(effect);
205 }
206 switch (effect->type){
207
208 case FFB_EFFECT_CONSTANT:
209 { // Constant force is just the force
210 force_vector = (int32_t)magnitude;
211 break;
212 }
213
214 case FFB_EFFECT_RAMP:
215 {
216 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
217 int32_t duration = effect->duration;
218 force_vector = (int32_t)effect->startLevel + ((int32_t)elapsed_time * (effect->endLevel - effect->startLevel)) / duration;
219 break;
220 }
221
222 case FFB_EFFECT_SQUARE:
223 {
224 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
225 int32_t force = ((elapsed_time + effect->phase) % ((uint32_t)effect->period + 2)) < (uint32_t)(effect->period + 2) / 2 ? -magnitude : magnitude;
226 force_vector = force + effect->offset;
227 break;
228 }
229
230 case FFB_EFFECT_TRIANGLE:
231 {
232 int32_t force = 0;
233 int32_t offset = effect->offset;
234 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
235 uint32_t phase = effect->phase;
236 uint32_t period = effect->period;
237 float periodF = period;
238
239 int32_t maxMagnitude = offset + magnitude;
240 int32_t minMagnitude = offset - magnitude;
241 uint32_t phasetime = (phase * period) / 35999;
242 uint32_t timeTemp = elapsed_time + phasetime;
243 float remainder = timeTemp % period;
244 float slope = ((maxMagnitude - minMagnitude) * 2) / periodF;
245 if (remainder > (periodF / 2))
246 force = slope * (periodF - remainder);
247 else
248 force = slope * remainder;
249 force += minMagnitude;
250 force_vector = force;
251 break;
252 }
253
254 case FFB_EFFECT_SAWTOOTHUP:
255 {
256 float offset = effect->offset;
257 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
258 uint32_t phase = effect->phase;
259 uint32_t period = effect->period;
260 float periodF = effect->period;
261
262 float maxMagnitude = offset + magnitude;
263 float minMagnitude = offset - magnitude;
264 int32_t phasetime = (phase * period) / 35999;
265 uint32_t timeTemp = elapsed_time + phasetime;
266 float remainder = timeTemp % period;
267 float slope = (maxMagnitude - minMagnitude) / periodF;
268 force_vector = (int32_t)(minMagnitude + slope * (period - remainder));
269 break;
270 }
271
272 case FFB_EFFECT_SAWTOOTHDOWN:
273 {
274 float offset = effect->offset;
275 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
276 float phase = effect->phase;
277 uint32_t period = effect->period;
278 float periodF = effect->period;
279
280 float maxMagnitude = offset + magnitude;
281 float minMagnitude = offset - magnitude;
282 int32_t phasetime = (phase * period) / 35999;
283 uint32_t timeTemp = elapsed_time + phasetime;
284 float remainder = timeTemp % period;
285 float slope = (maxMagnitude - minMagnitude) / periodF;
286 force_vector = (int32_t)(minMagnitude + slope * (remainder)); // reverse time
287 break;
288 }
289
290 case FFB_EFFECT_SINE:
291 {
292 float t = HAL_GetTick() - effect->startTime;
293 float freq = 1.0f / (float)(std::max<uint16_t>(effect->period, 2));
294 float phase = (float)effect->phase / (float)35999; //degrees
295 float sine = sinf(2.0 * M_PI * (t * freq + phase)) * magnitude;
296 force_vector = (int32_t)(effect->offset + sine);
297 break;
298 }
299 default:
300 return 0;
301 break;
302 }
303
304 return (force_vector * effect->gain) / 255;
305}

323int32_t EffectsCalculator::calcComponentForce(FFB_Effect *effect, int32_t forceVector, std::vector<std::unique_ptr<Axis>> &axes, uint8_t axis)
324{
325 int32_t result_torque = 0;
326 uint16_t direction;
327 uint8_t con_idx = 0; // condition block index
328
329 metric_t *metrics = axes[axis]->getMetrics();
330 uint8_t axisCount = axes.size();
331 uint8_t directionEnableMask = this->directionEnableMask ? this->directionEnableMask : DIRECTION_ENABLE(axisCount);
332 if (effect->enableAxis & directionEnableMask)
333 {
334 direction = effect->directionX;
335 con_idx = axis;
336 }
337 else
338 {
339 direction = axis == 0 ? effect->directionX : effect->directionY;
340 con_idx = axis;
341 }
342
343 //bool useForceDirectionForConditionEffect = (effect->enableAxis == DIRECTION_ENABLE && axisCount > 1 && effect->conditionsCount == 1);
344 bool rotateConditionForce = (axisCount > 1); // && effect->conditionsCount < axisCount
345 float angle = ((float)direction * (2*M_PI) / 36000.0);
346 float angle_ratio = axis == 0 ? sin(angle) : -1 * cos(angle);
347 angle_ratio = rotateConditionForce ? angle_ratio : 1.0;
348
349 switch (effect->type)
350 {
351 case FFB_EFFECT_CONSTANT:
352 {
353 // Optional filtering to reduce spikes
354 if(effect->filter[con_idx] != nullptr) {
355 // if the filter is enabled we apply it
356 if (effect->filter[con_idx]->getFc() < 0.5 && effect->filter[0]->getFc() != 0.0)
357 {
358 forceVector = effect->filter[con_idx]->process(forceVector);
359 }
360 }
361 }
362 case FFB_EFFECT_RAMP:
363 case FFB_EFFECT_SQUARE:
364 case FFB_EFFECT_TRIANGLE:
365 case FFB_EFFECT_SAWTOOTHUP:
366 case FFB_EFFECT_SAWTOOTHDOWN:
367 case FFB_EFFECT_SINE:
368 {
369 result_torque = -forceVector * angle_ratio;
370 break;
371 }
372
373 case FFB_EFFECT_SPRING:
374 {
375 float pos = metrics->pos;
376 result_torque -= calcConditionEffectForce(effect, pos, gain.spring, con_idx, scaler.spring, angle_ratio);
377 break;
378 }
379
380
393 case FFB_EFFECT_FRICTION: // TODO sometimes unstable.
394 {
395 float speed = metrics->speed * INTERNAL_SCALER_FRICTION;
396
397 int16_t offset = effect->conditions[con_idx].cpOffset;
398 int16_t deadBand = effect->conditions[con_idx].deadBand;
399 int32_t force = 0;
400
401 float speedRampupCeil = speedRampupPct();
402
403 // Effect is only active outside deadband + offset
404 if (abs((int32_t)speed - offset) > deadBand){
405
406 // remove offset/deadband from metric to compute force
407 speed -= (offset + (deadBand * (speed < offset ? -1 : 1)) );
408
409 // check if speed is in the 0..x% to rampup, if is this range, apply a sinusoidale function to smooth the torque (slow near 0, slow around the X% rampup
410 float rampupFactor = 1.0;
411 if (fabs (speed) < speedRampupCeil) { // if speed in the range to rampup we apply a sinus curbe to ramup
412
413 float phaseRad = M_PI * ((fabs (speed) / speedRampupCeil) - 0.5);// we start to compute the normalized angle (speed / normalizedSpeed@5%) and translate it of -1/2PI to translate sin on 1/2 periode
414 rampupFactor = ( 1 + sin(phaseRad ) ) / 2; // sin value is -1..1 range, we translate it to 0..2 and we scale it by 2
415
416 }
417
418 int8_t sign = speed >= 0 ? 1 : -1;
419 uint16_t coeff = speed < 0 ? effect->conditions[con_idx].negativeCoefficient : effect->conditions[con_idx].positiveCoefficient;
420 force = coeff * rampupFactor * sign;
421
422 //if there is a saturation, used it to clip result
423 if (effect->conditions[con_idx].negativeSaturation !=0 || effect->conditions[con_idx].positiveSaturation !=0) {
424 force = clip<int32_t, int32_t>(force, -effect->conditions[con_idx].negativeSaturation, effect->conditions[con_idx].positiveSaturation);
425 }
426
427 result_torque -= effect->filter[con_idx]->process( (((gain.friction + 1) * force) >> 8) * angle_ratio * scaler.friction);
428 }
429
430 break;
431 }
432 case FFB_EFFECT_DAMPER:
433 {
434
435 float speed = metrics->speed * INTERNAL_SCALER_DAMPER;
436 result_torque -= effect->filter[con_idx]->process(calcConditionEffectForce(effect, speed, gain.damper, con_idx, scaler.damper, angle_ratio));
437
438 break;
439 }
440
441 case FFB_EFFECT_INERTIA:
442 {
443 float accel = metrics->accel * INTERNAL_SCALER_INERTIA;
444 result_torque -= effect->filter[con_idx]->process(calcConditionEffectForce(effect, accel, gain.inertia, con_idx, scaler.inertia, angle_ratio)); // Bump *60 the inertia feedback
445
446 break;
447 }
448
449 default:
450 // Unsupported effect
451 break;
452 }
453 return (result_torque * global_gain) / 255; // Apply global gain
454}

466int32_t EffectsCalculator::calcConditionEffectForce(FFB_Effect *effect, float metric, uint8_t gain,
467 uint8_t idx, float scale, float angle_ratio)
468{
469 int16_t offset = effect->conditions[idx].cpOffset;
470 int16_t deadBand = effect->conditions[idx].deadBand;
471 int32_t force = 0;
472 float gainfactor = (float)(gain+1) / 256.0;
473
474 // Effect is only active outside deadband + offset
475 if (abs(metric - offset) > deadBand){
476 float coefficient = effect->conditions[idx].negativeCoefficient;
477 if(metric > offset){
478 coefficient = effect->conditions[idx].positiveCoefficient;
479 }
480 coefficient /= 0x7fff; // rescale the coefficient of effect
481
482 // remove offset/deadband from metric to compute force
483 metric = metric - (offset + (deadBand * (metric < offset ? -1 : 1)) );
484
485 force = clip<int32_t, int32_t>((coefficient * gainfactor * scale * (float)(metric)),
486 -effect->conditions[idx].negativeSaturation,
487 effect->conditions[idx].positiveSaturation);
488 }
489
490
491 return force * angle_ratio;
492}

500int32_t EffectsCalculator::getEnvelopeMagnitude(FFB_Effect *effect)
501{
502 if(effect->duration == FFB_EFFECT_DURATION_INFINITE || effect->duration == 0){
503 return effect->magnitude; // Effect is infinite. envelope is invalid
504 }
505 int32_t scaler = abs(effect->magnitude);
506 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
507 if (elapsed_time < effect->attackTime && effect->attackTime != 0)
508 {
509 scaler = (scaler - effect->attackLevel) * elapsed_time;
510 scaler /= (int32_t)effect->attackTime;
511 scaler += effect->attackLevel;
512 }
513 if (elapsed_time > (effect->duration - effect->fadeTime) && effect->fadeTime != 0)
514 {
515 scaler = (scaler - effect->fadeLevel) * (effect->duration - elapsed_time); // Reversed
516 scaler /= (int32_t)effect->fadeTime;
517 scaler += effect->fadeLevel;
518 }
519 scaler = signbit(effect->magnitude) ? -scaler : scaler; // Follow original sign of magnitude because envelope has no sign (important for constant force)
520 return scaler;
521}

523void EffectsCalculator::setFilters(FFB_Effect *effect){
524
525 std::function<void(std::unique_ptr<Biquad> &)> fnptr = [=](std::unique_ptr<Biquad> &filter){};
526
527 switch (effect->type)
528 {
529 case FFB_EFFECT_DAMPER:
530 fnptr = [=](std::unique_ptr<Biquad> &filter){
531 if (filter != nullptr)
532 filter->setBiquad(BiquadType::lowpass, this->filter[filterProfileId].damper.freq/ (float)calcfrequency, this->filter[filterProfileId].damper.q * qfloatScaler , (float)0.0);
533 else
534 filter = std::make_unique<Biquad>(BiquadType::lowpass, this->filter[filterProfileId].damper.freq / (float)calcfrequency, this->filter[filterProfileId].damper.q * qfloatScaler, (float)0.0);
535 };
536 break;
537 case FFB_EFFECT_FRICTION:
538 fnptr = [=](std::unique_ptr<Biquad> &filter){
539 if (filter != nullptr)
540 filter->setBiquad(BiquadType::lowpass, this->filter[filterProfileId].friction.freq / (float)calcfrequency, this->filter[filterProfileId].friction.q * qfloatScaler, (float)0.0);
541 else
542 filter = std::make_unique<Biquad>(BiquadType::lowpass, this->filter[filterProfileId].friction.freq / (float)calcfrequency, this->filter[filterProfileId].friction.q * qfloatScaler, (float)0.0);
543 };
544 break;
545 case FFB_EFFECT_INERTIA:
546 fnptr = [=](std::unique_ptr<Biquad> &filter){
547 if (filter != nullptr)
548 filter->setBiquad(BiquadType::lowpass, this->filter[filterProfileId].inertia.freq / (float)calcfrequency, this->filter[filterProfileId].inertia.q * qfloatScaler, (float)0.0);
549 else
550 filter = std::make_unique<Biquad>(BiquadType::lowpass, this->filter[filterProfileId].inertia.freq / (float)calcfrequency, this->filter[filterProfileId].inertia.q * qfloatScaler, (float)0.0);
551 };
552 break;
553 case FFB_EFFECT_CONSTANT:
554 fnptr = [=](std::unique_ptr<Biquad> &filter){
555 if (filter != nullptr)
556 filter->setBiquad(BiquadType::lowpass, this->filter[0].constant.freq / (float)calcfrequency, this->filter[0].constant.q * qfloatScaler, (float)0.0);
557 else
558 filter = std::make_unique<Biquad>(BiquadType::lowpass, this->filter[0].constant.freq / (float)calcfrequency, this->filter[0].constant.q * qfloatScaler, (float)0.0);
559 };
560 break;
561 }
562
563
564 for (int i=0; i<MAX_AXIS; i++) {
565 fnptr(effect->filter[i]);
566 }
567}

587void EffectsCalculator::restoreFlash()
588{
589 uint16_t filterStorage;
590 if (Flash_Read(ADR_FFB_CF_FILTER, &filterStorage))
591 {
592 uint32_t freq = filterStorage & 0x1FF;
593 uint8_t q = (filterStorage >> 9) & 0x7F;
594 checkFilterCoeff(&(this->filter[0].constant), freq, q);
595 updateFilterSettingsForEffects(FFB_EFFECT_CONSTANT);
596 }
597
598 if (Flash_Read(ADR_FFB_FR_FILTER, &filterStorage))
599 {
600 uint32_t freq = filterStorage & 0x1FF;
601 uint8_t q = (filterStorage >> 9) & 0x7F;
602 checkFilterCoeff(&(this->filter[CUSTOM_PROFILE_ID].friction), freq, q);
603 updateFilterSettingsForEffects(FFB_EFFECT_FRICTION);
604 }
605
606 if (Flash_Read(ADR_FFB_DA_FILTER, &filterStorage))
607 {
608 uint32_t freq = filterStorage & 0x1FF;
609 uint8_t q = (filterStorage >> 9) & 0x7F;
610 checkFilterCoeff(&(this->filter[CUSTOM_PROFILE_ID].damper), freq, q);
611 updateFilterSettingsForEffects(FFB_EFFECT_DAMPER);
612 }
613
614 if (Flash_Read(ADR_FFB_IN_FILTER, &filterStorage))
615 {
616 uint32_t freq = filterStorage & 0x1FF;
617 uint8_t q = (filterStorage >> 9) & 0x7F;
618 checkFilterCoeff(&(this->filter[CUSTOM_PROFILE_ID].inertia), freq, q);
619 updateFilterSettingsForEffects(FFB_EFFECT_INERTIA);
620 }

638// Saves parameters to flash
639void EffectsCalculator::saveFlash()
640{
641 uint16_t filterStorage;
642
643 // save CF biquad
644 filterStorage = (uint16_t)filter[0].constant.freq & 0x1FF;
645 filterStorage |= ( (uint16_t)filter[0].constant.q & 0x7F ) << 9 ;
646 Flash_Write(ADR_FFB_CF_FILTER, filterStorage);
647
648 if(filterProfileId == CUSTOM_PROFILE_ID){ // Only attempt saving if custom profile active
649 // save Friction biquad
650 filterStorage = (uint16_t)filter[CUSTOM_PROFILE_ID].friction.freq & 0x1FF;
651 filterStorage |= ( (uint16_t)filter[CUSTOM_PROFILE_ID].friction.q & 0x7F ) << 9 ;
652 Flash_Write(ADR_FFB_FR_FILTER, filterStorage);
653
654 // save Damper biquad
655 filterStorage = (uint16_t)filter[CUSTOM_PROFILE_ID].damper.freq & 0x1FF;
656 filterStorage |= ( (uint16_t)filter[CUSTOM_PROFILE_ID].damper.q & 0x7F ) << 9 ;
657 Flash_Write(ADR_FFB_DA_FILTER, filterStorage);
658
659 // save Inertia biquad
660 filterStorage = (uint16_t)filter[CUSTOM_PROFILE_ID].inertia.freq & 0x1FF;
661 filterStorage |= ( (uint16_t)filter[CUSTOM_PROFILE_ID].inertia.q & 0x7F ) << 9 ;
662 Flash_Write(ADR_FFB_IN_FILTER, filterStorage);
663 }
664
665 // save the effect gain
666 uint16_t effects = gain.inertia | (gain.friction << 8);
667 Flash_Write(ADR_FFB_EFFECTS1, effects);
668
669 effects = gain.spring | (gain.damper << 8);
670 Flash_Write(ADR_FFB_EFFECTS2, effects);
671
672 // save the friction rampup zone
673 effects = frictionPctSpeedToRampup | (filterProfileId << 8);
674 Flash_Write(ADR_FFB_EFFECTS3, effects);
