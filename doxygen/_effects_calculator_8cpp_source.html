67/*
68If the metric is less than CP Offset - Dead Band, then the resulting force is given by the following formula:
69 force = Negative Coefficient * (q - (CP Offset â€“ Dead Band))
70Similarly, if the metric is greater than CP Offset + Dead Band, then the resulting force is given by the
71following formula:
72 force = Positive Coefficient * (q - (CP Offset + Dead Band))
73A spring condition uses axis position as the metric.
74A damper condition uses axis velocity as the metric.
75An inertia condition uses axis acceleration as the metric.
76
77If the number of Condition report blocks is equal to the number of axes for the effect, then the first report
78block applies to the first axis, the second applies to the second axis, and so on. For example, a two-axis
79spring condition with CP Offset set to zero in both Condition report blocks would have the same effect as
80the joystick self-centering spring. When a condition is defined for each axis in this way, the effect must
81not be rotated.
82
83If there is a single Condition report block for an effect with more than one axis, then the direction along
84which the parameters of the Condition report block are in effect is determined by the direction parameters
85passed in the Direction field of the Effect report block. For example, a friction condition rotated 45
86degrees (in polar coordinates) would resist joystick motion in the northeast-southwest direction but would
87have no effect on joystick motion in the northwest-southeast direction.
88 */

90/*
91 * Calculates the resulting torque for FFB effects
92 * Takes current position input scaled from -0x7fff to 0x7fff
93 * Outputs a torque value from -0x7fff to 0x7fff (not yet clipped)
94 */
95void EffectsCalculator::calculateEffects(std::vector<std::unique_ptr<Axis>> &axes)
96{
97 for (auto &axis : axes) {
98 axis->calculateAxisEffects(isActive());
99 }
100
101 if(!isActive()){
102 return;
103 }
104
105 int32_t forceX = 0;
106 int32_t forceY = 0;
107 int32_t forceVector = 0;
108 uint8_t axisCount = axes.size();
109 bool validY = axisCount > 1;
110#if MAX_AXIS == 3
111 int32_t forceZ = 0;
112 bool validZ = axisCount > 2;
113#endif
114
115 for (uint8_t i = 0; i < MAX_EFFECTS; i++)
116 {
117 FFB_Effect *effect = &effects[i];
118
119 // Effect activated and not infinite (0 or 0xffff)
120 if (effect->state != EFFECT_STATE_INACTIVE && effect->duration != FFB_EFFECT_DURATION_INFINITE && effect->duration != 0){
121 // Start delay not yet reached
122 if(HAL_GetTick() < effect->startTime){
123 continue;
124 }
125 // If effect has expired make inactive
126 if (HAL_GetTick() > effect->startTime + effect->duration)
127 {
128 effect->state = EFFECT_STATE_INACTIVE;
129 }
130 }
131
132 // Filter out inactive effects
133 if (effect->state == EFFECT_STATE_INACTIVE)
134 {
135 continue;
136 }
137
138
139 //if (effect->conditionsCount == 0) {
140 forceVector = calcNonConditionEffectForce(effect);
141 //}
142
143 uint8_t directionEnableMask = this->directionEnableMask ? this->directionEnableMask : DIRECTION_ENABLE(axisCount);
144
145 if (effect->enableAxis & directionEnableMask || (effect->enableAxis & X_AXIS_ENABLE))
146 {
147 forceX += calcComponentForce(effect, forceVector, axes, 0);
148 forceX = clip<int32_t, int32_t>(forceX, -0x7fff, 0x7fff); // Clip
149 }
150 if (validY && (effect->enableAxis & directionEnableMask || (effect->enableAxis & Y_AXIS_ENABLE)))
151 {
152 forceY += calcComponentForce(effect, forceVector, axes, 1);
153 forceY = clip<int32_t, int32_t>(forceY, -0x7fff, 0x7fff); // Clip
154 }
155
156 }
157
158 axes[0]->setEffectTorque(forceX);
159 if (validY)
160 {
161 axes[1]->setEffectTorque(forceY);
162 }
163}

169int32_t EffectsCalculator::calcNonConditionEffectForce(FFB_Effect *effect) {
170 int32_t force_vector = 0;
171 int32_t magnitude = effect->magnitude;
172
173 // If using an envelope modulate the magnitude based on time
174 if(effect->useEnvelope){
175 magnitude = getEnvelopeMagnitude(effect);
176 }
177 switch (effect->type){
178
179 case FFB_EFFECT_CONSTANT:
180 { // Constant force is just the force
181 force_vector = (int32_t)magnitude;
182 break;
183 }
184
185 case FFB_EFFECT_RAMP:
186 {
187 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
188 int32_t duration = effect->duration;
189 force_vector = (int32_t)effect->startLevel + ((int32_t)elapsed_time * (effect->endLevel - effect->startLevel)) / duration;
190 break;
191 }
192
193 case FFB_EFFECT_SQUARE:
194 {
195 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
196 int32_t force = ((elapsed_time + effect->phase) % ((uint32_t)effect->period + 2)) < (uint32_t)(effect->period + 2) / 2 ? -magnitude : magnitude;
197 force_vector = force + effect->offset;
198 break;
199 }
200
201 case FFB_EFFECT_TRIANGLE:
202 {
203 int32_t force = 0;
204 int32_t offset = effect->offset;
205 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
206 uint32_t phase = effect->phase;
207 uint32_t period = effect->period;
208 float periodF = period;
209
210 int32_t maxMagnitude = offset + magnitude;
211 int32_t minMagnitude = offset - magnitude;
212 uint32_t phasetime = (phase * period) / 35999;
213 uint32_t timeTemp = elapsed_time + phasetime;
214 float remainder = timeTemp % period;
215 float slope = ((maxMagnitude - minMagnitude) * 2) / periodF;
216 if (remainder > (periodF / 2))
217 force = slope * (periodF - remainder);
218 else
219 force = slope * remainder;
220 force += minMagnitude;
221 force_vector = force;
222 break;
223 }
224
225 case FFB_EFFECT_SAWTOOTHUP:
226 {
227 float offset = effect->offset;
228 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
229 uint32_t phase = effect->phase;
230 uint32_t period = effect->period;
231 float periodF = effect->period;
232
233 float maxMagnitude = offset + magnitude;
234 float minMagnitude = offset - magnitude;
235 int32_t phasetime = (phase * period) / 35999;
236 uint32_t timeTemp = elapsed_time + phasetime;
237 float remainder = timeTemp % period;
238 float slope = (maxMagnitude - minMagnitude) / periodF;
239 force_vector = (int32_t)(minMagnitude + slope * (period - remainder));
240 break;
241 }
242
243 case FFB_EFFECT_SAWTOOTHDOWN:
244 {
245 float offset = effect->offset;
246 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
247 float phase = effect->phase;
248 uint32_t period = effect->period;
249 float periodF = effect->period;
250
251 float maxMagnitude = offset + magnitude;
252 float minMagnitude = offset - magnitude;
253 int32_t phasetime = (phase * period) / 35999;
254 uint32_t timeTemp = elapsed_time + phasetime;
255 float remainder = timeTemp % period;
256 float slope = (maxMagnitude - minMagnitude) / periodF;
257 force_vector = (int32_t)(minMagnitude + slope * (remainder)); // reverse time
258 break;
259 }
260
261 case FFB_EFFECT_SINE:
262 {
263 float t = HAL_GetTick() - effect->startTime;
264 float freq = 1.0f / (float)(std::max<uint16_t>(effect->period, 2));
265 float phase = (float)effect->phase / (float)35999; //degrees
266 float sine = sinf(2.0 * M_PI * (t * freq + phase)) * magnitude;
267 force_vector = (int32_t)(effect->offset + sine);
268 break;
269 }
270 default:
271 return 0;
272 break;
273 }
274
275 return (force_vector * effect->gain) / 255;
276}

294int32_t EffectsCalculator::calcComponentForce(FFB_Effect *effect, int32_t forceVector, std::vector<std::unique_ptr<Axis>> &axes, uint8_t axis)
295{
296 int32_t result_torque = 0;
297 uint16_t direction;
298 uint8_t con_idx = 0; // condition block index
299
300 metric_t *metrics = axes[axis]->getMetrics();
301 uint8_t axisCount = axes.size();
302 float scaleSpeed = 40;//axes[axis]->getSpeedScalerNormalized(); // TODO decide if scalers are useful or not
303 float scaleAccel = 40;//axes[axis]->getAccelScalerNormalized();
304 uint8_t directionEnableMask = this->directionEnableMask ? this->directionEnableMask : DIRECTION_ENABLE(axisCount);
305 if (effect->enableAxis & directionEnableMask)
306 {
307 direction = effect->directionX;
308// if (effect->conditionsCount > 1)
309// {
310 con_idx = axis;
311// }
312 }
313 else
314 {
315 direction = axis == 0 ? effect->directionX : effect->directionY;
316 con_idx = axis;
317 }
318
319 //bool useForceDirectionForConditionEffect = (effect->enableAxis == DIRECTION_ENABLE && axisCount > 1 && effect->conditionsCount == 1);
320 bool rotateConditionForce = (axisCount > 1); // && effect->conditionsCount < axisCount
321 float angle = ((float)direction * (2*M_PI) / 36000.0);
322 float angle_ratio = axis == 0 ? sin(angle) : -1 * cos(angle);
323 angle_ratio = rotateConditionForce ? angle_ratio : 1.0;
324
325 switch (effect->type)
326 {
327 case FFB_EFFECT_CONSTANT:
328 {
329 // Optional filtering to reduce spikes
330 if (cfFilter_f < calcfrequency / 2 && cfFilter_f != 0 )
331 {
332 forceVector = effect->filter[con_idx]->process(forceVector);
333 }
334 }
335 case FFB_EFFECT_RAMP:
336 case FFB_EFFECT_SQUARE:
337 case FFB_EFFECT_TRIANGLE:
338 case FFB_EFFECT_SAWTOOTHUP:
339 case FFB_EFFECT_SAWTOOTHDOWN:
340 case FFB_EFFECT_SINE:
341 {
342 result_torque = -forceVector * angle_ratio;
343 break;
344 }
345
346 case FFB_EFFECT_SPRING:
347 {
348 float pos = metrics->pos;
349 result_torque -= calcConditionEffectForce(effect, pos, gain.spring, con_idx, spring_scaler, angle_ratio);
350 break;
351 }
352
353
366 case FFB_EFFECT_FRICTION: // TODO sometimes unstable.
367 {
368 float speed = metrics->speed * scaleSpeed;//effect->filter[con_idx]->process()
369
370 int16_t offset = effect->conditions[con_idx].cpOffset;
371 int16_t deadBand = effect->conditions[con_idx].deadBand;
372 int32_t force = 0;
373
374 // Effect is only active outside deadband + offset
375 if (abs((int32_t)speed - offset) > deadBand){
376
377 // remove offset/deadband from metric to compute force
378 speed -= (offset + (deadBand * (speed < offset ? -1 : 1)) );
379
380 // check if speed is in the 0..x% to rampup, if is this range, apply a sinusoidale function to smooth the torque (slow near 0, slow around the X% rampup
381 float rampupFactor = 1.0;
382 if (fabs (speed) < speedRampupPct) { // if speed in the range to rampup we apply a sinus curbe to ramup
383
384 float phaseRad = M_PI * ((fabs (speed) / speedRampupPct) - 0.5);// we start to compute the normalized angle (speed / normalizedSpeed@5%) and translate it of -1/2PI to translate sin on 1/2 periode
385 rampupFactor = ( 1 + sin(phaseRad ) ) / 2; // sin value is -1..1 range, we translate it to 0..2 and we scale it by 2
386
387 }
388
389 int8_t sign = speed >= 0 ? 1 : -1;
390 uint16_t coeff = speed < 0 ? effect->conditions[con_idx].negativeCoefficient : effect->conditions[con_idx].positiveCoefficient;
391 force = coeff * rampupFactor * sign;
392
393 //if there is a saturation, used it to clip result
394 if (effect->conditions[con_idx].negativeSaturation !=0 || effect->conditions[con_idx].positiveSaturation !=0) {
395 force = clip<int32_t, int32_t>(force, -effect->conditions[con_idx].negativeSaturation, effect->conditions[con_idx].positiveSaturation);
396 }
397
398// static int32_t last_force = 0;
399//
400// // if there is 2 successive torque with a different direction, we ignore the first one to remove oscillation
401// if (last_force * force >=0) {
402// force = ((gain.friction + 1) * force) >> 7;
403// result_torque -= force * angle_ratio;
404// }
405// last_force = force;
406 result_torque -= effect->filter[con_idx]->process( (((gain.friction + 1) * force) >> 8) * angle_ratio * friction_scaler);
407 }
408// float accel = metrics->accel * scaleAccel;
409// result_torque -= calcConditionEffectForce(effect, accel, gain.friction, con_idx, friction_scaler, angle_ratio);
410
411 break;
412 }
413 case FFB_EFFECT_DAMPER:
414 {
415
416 float speed = metrics->speed * scaleSpeed;
417 result_torque -= effect->filter[con_idx]->process(calcConditionEffectForce(effect, speed, gain.damper, con_idx, damper_scaler, angle_ratio));
418
419 break;
420 }
421
422 case FFB_EFFECT_INERTIA:
423 {
424 float accel = metrics->accel* scaleAccel;
425 result_torque -= effect->filter[con_idx]->process(calcConditionEffectForce(effect, accel, gain.inertia, con_idx, inertia_scaler, angle_ratio)); // Bump *60 the inertia feedback
426
427 break;
428 }
429
430 default:
431 // Unsupported effect
432 break;
433 }
434 return (result_torque * global_gain) / 255; // Apply global gain
435}

442int32_t EffectsCalculator::calcConditionEffectForce(FFB_Effect *effect, float metric, uint8_t gain,
443 uint8_t idx, float scale, float angle_ratio)
444{
445 int16_t offset = effect->conditions[idx].cpOffset;
446 int16_t deadBand = effect->conditions[idx].deadBand;
447 int32_t force = 0;
448 float gainfactor = (float)(gain+1) / 256.0;
449
450 // Effect is only active outside deadband + offset
451 if (abs(metric - offset) > deadBand){
452 float coefficient = effect->conditions[idx].negativeCoefficient;
453 if(metric > offset){
454 coefficient = effect->conditions[idx].positiveCoefficient;
455 }
456 coefficient /= 0x7fff; // rescale the coefficient of effect
457
458 // remove offset/deadband from metric to compute force
459 metric = metric - (offset + (deadBand * (metric < offset ? -1 : 1)) );
460
461 force = clip<int32_t, int32_t>((coefficient * gainfactor * scale * (float)(metric)),
462 -effect->conditions[idx].negativeSaturation,
463 effect->conditions[idx].positiveSaturation);
464 }
465
466
467 return force * angle_ratio;
468}

476int32_t EffectsCalculator::getEnvelopeMagnitude(FFB_Effect *effect)
477{
478 if(effect->duration == FFB_EFFECT_DURATION_INFINITE || effect->duration == 0){
479 return effect->magnitude; // Effect is infinite. envelope is invalid
480 }
481 int32_t scaler = abs(effect->magnitude);
482 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
483 if (elapsed_time < effect->attackTime && effect->attackTime != 0)
484 {
485 scaler = (scaler - effect->attackLevel) * elapsed_time;
486 scaler /= (int32_t)effect->attackTime;
487 scaler += effect->attackLevel;
488 }
489 if (elapsed_time > (effect->duration - effect->fadeTime) && effect->fadeTime != 0)
490 {
491 scaler = (scaler - effect->fadeLevel) * (effect->duration - elapsed_time); // Reversed
492 scaler /= (int32_t)effect->fadeTime;
493 scaler += effect->fadeLevel;
494 }
495 scaler = signbit(effect->magnitude) ? -scaler : scaler; // Follow original sign of magnitude because envelope has no sign (important for constant force)
496 return scaler;
497}

499void EffectsCalculator::setFilters(FFB_Effect *effect){
500
501 std::function<void(std::unique_ptr<Biquad> &)> fnptr = [=](std::unique_ptr<Biquad> &filter){};
502
503 switch (effect->type)
504 {
505 case FFB_EFFECT_DAMPER:
506 fnptr = [=](std::unique_ptr<Biquad> &filter){
507 if (filter != nullptr)
508 filter->setBiquad(BiquadType::lowpass, (float)damper_f / (float)calcfrequency, damper_q, (float)0.0);
509 else
510 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)damper_f / (float)calcfrequency, damper_q, (float)0.0);
511 };
512 break;
513 case FFB_EFFECT_FRICTION:
514 fnptr = [=](std::unique_ptr<Biquad> &filter){
515 if (filter != nullptr)
516 filter->setBiquad(BiquadType::lowpass, (float)friction_f / (float)calcfrequency, friction_q, (float)0.0);
517 else
518 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)friction_f / (float)calcfrequency, friction_q, (float)0.0);
519 };
520 break;
521 case FFB_EFFECT_INERTIA:
522 fnptr = [=](std::unique_ptr<Biquad> &filter){
523 if (filter != nullptr)
524 filter->setBiquad(BiquadType::lowpass, (float)inertia_f / (float)calcfrequency, inertia_q, (float)0.0);
525 else
526 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)inertia_f / (float)calcfrequency, inertia_q, (float)0.0);
527 };
528 break;
529 case FFB_EFFECT_CONSTANT:
530 fnptr = [=](std::unique_ptr<Biquad> &filter){
531 if (filter != nullptr)
532 filter->setBiquad(BiquadType::lowpass, (float)cfFilter_f / (float)calcfrequency, cfFilter_qfloatScaler * (cfFilter_q+1), (float)0.0);
533 else
534 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)cfFilter_f / (float)calcfrequency, cfFilter_qfloatScaler * (cfFilter_q+1), (float)0.0);
535 };
536 break;
537 }
538
539
540 for (int i=0; i<MAX_AXIS; i++) {
541 fnptr(effect->filter[i]);
542 }
543}

560/*
561 * Read parameters from flash and restore settings
562 */
563void EffectsCalculator::restoreFlash()
564{
565 uint16_t filter;
566 if (Flash_Read(ADR_CF_FILTER, &filter))
567 {
568 this->cfFilter_f = filter & 0x1FF;
569 this->cfFilter_q = (filter >> 9) & 0x7F;
570 }
571 setCfFilter(this->cfFilter_f,this->cfFilter_q);
572
573 uint16_t effects = 0;
574 if(Flash_Read(ADR_AXIS_EFFECTS1, &effects)){
575 gain.friction = (effects >> 8) & 0xff;
576 gain.inertia = (effects & 0xff);
577 }
578 if(Flash_Read(ADR_AXIS_EFFECTS2, &effects)){
579 gain.damper = (effects >> 8) & 0xff;
580 gain.spring = (effects & 0xff);
581 }
582
583}

585// Saves parameters to flash
586void EffectsCalculator::saveFlash()
587{
588 uint16_t cffilter = (cfFilter_f & 0x1FF) | ((cfFilter_q & 0x7F) << 9);
589 Flash_Write(ADR_CF_FILTER, cffilter);
590 uint16_t effects = gain.inertia | (gain.friction << 8);
591 Flash_Write(ADR_AXIS_EFFECTS1, effects);
592 effects = gain.spring | (gain.damper << 8);
593 Flash_Write(ADR_AXIS_EFFECTS2, effects);
594

650CommandStatus EffectsCalculator::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
651 switch(static_cast<EffectsCalculator_commands>(cmd.cmdId)){
652
653 case EffectsCalculator_commands::ffbfiltercf:
654 if (cmd.type == CMDtype::get)
655 {
656 replies.emplace_back(cfFilter_f);
657 }
658 else if (cmd.type == CMDtype::set)
659 {
660 setCfFilter(cmd.val,this->cfFilter_q);
661 }
662 break;
663
664 case EffectsCalculator_commands::ffbfiltercf_q:
665 if (cmd.type == CMDtype::get)
666 {
667 replies.emplace_back(cfFilter_q);
668 }
669 else if (cmd.type == CMDtype::set)
670 {
671 setCfFilter(this->cfFilter_f,cmd.val);
672 }
673
674 break;
675
676 case EffectsCalculator_commands::effects:
677 if (cmd.type == CMDtype::get)
678 {
679 replies.emplace_back(listEffectsUsed());
680 }
