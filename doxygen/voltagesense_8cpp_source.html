52void brakeCheck(){
53 if(maxVoltage == 0 || brake_failure){
54 return;
55 }
56 int32_t vint = getIntV();
57 int32_t vext = getExtV();
58
59 if(vint < minVoltage && vext < minVoltage){
60 return; // Do not enable if device is unpowered (just measuring usb leakage)
61 }
62
63 bool lastBrakingFlag = braking_flag;
64 braking_flag = (vint > vext + voltageDiffActivate) || vint > maxVoltage || (braking_flag && (vint > vext + voltageDiffDeactivate));
65 //(ADC_BUF[ADC_CHAN_VINT] > ADC_BUF[ADC_CHAN_VEXT]+400 || (ADC_BUF[ADC_CHAN_VINT] > 3000));
66 if(braking_flag != lastBrakingFlag){
67 if(braking_flag){
68 brakeActiveTime = HAL_GetTick();
69 }else{
70 brakeActiveTime = 0;
71 }
72 }
73 if(brakeActiveTime && HAL_GetTick() - brakeActiveTime > 5000){
74 // Brake resistor active over 5s. Shut down.
75 brake_failure = true;
76 HAL_GPIO_WritePin(DRV_BRAKE_GPIO_Port,DRV_BRAKE_Pin, GPIO_PIN_RESET);
77 // TODO: can not create error here in ISR. may cause faults
78 ErrorHandler::addError(resError);
79 }else{
80 HAL_GPIO_WritePin(DRV_BRAKE_GPIO_Port,DRV_BRAKE_Pin, braking_flag ? GPIO_PIN_SET:GPIO_PIN_RESET);
81 }
