25uint8_t TMCDebugBridge::checksum(std::vector<uint8_t> *buffer,uint8_t len){
26 uint16_t sum = 0;
27 for(uint8_t i = 0; i < len ;i++){
28 sum+= (*buffer)[i];
29 }
30 sum = sum % 256;
31 return sum;
32}

46CommandStatus TMCDebugBridge::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
47 switch(static_cast<TMCDebugBridge_commands>(cmd.cmdId)){
48 case TMCDebugBridge_commands::torque:
49 if(cmd.type == CMDtype::get){
50 replies.emplace_back(drv->getTorque());
51 }else if(cmd.type == CMDtype::set){
52 if(drv->getState() != TMC_ControlState::Running){
53 drv->startMotor();
54 }
55 drv->turn(cmd.val);
56 }else{
57 return CommandStatus::ERR;
58 }
59 break;
60
61 case TMCDebugBridge_commands::pos:
62 if(cmd.type == CMDtype::get){
63 replies.emplace_back(drv->getPos());
64 }else if(cmd.type == CMDtype::set){
65 drv->setTargetPos(cmd.val);
66 }else{
67 return CommandStatus::ERR;
68 }
69 break;
70
71 case TMCDebugBridge_commands::openloopspeed:
72 if(cmd.type == CMDtype::set){
73 drv->setOpenLoopSpeedAccel(cmd.val,100);
74 }else if(cmd.type == CMDtype::setat){
75 drv->runOpenLoop(cmd.adr,0,cmd.val,10);
76 }else{
77 return CommandStatus::ERR;
78 }
79 break;
80
81 case TMCDebugBridge_commands::velocity:
82 if(cmd.type == CMDtype::get){
83 replies.emplace_back(drv->getVelocity());
84 }else if(cmd.type == CMDtype::set){
85 drv->setTargetVelocity(cmd.val);
86 }else{
87 return CommandStatus::ERR;
88 }
89 break;
90
91 case TMCDebugBridge_commands::mode:
92 if(cmd.type == CMDtype::get){
93 replies.emplace_back((uint8_t)drv->getMotionMode());
94 }else if(cmd.type == CMDtype::set && cmd.val < (uint8_t)MotionMode::NONE){
95 drv->setMotionMode(MotionMode(cmd.val));
96 drv->startMotor();
97 }else{
98 replies.emplace_back("stop=0,torque=1,velocity=2,position=3,prbsflux=4,prbstorque=5,prbsvelocity=6,uqudext=8,encminimove=9");
99 }
100 break;
101
102 case TMCDebugBridge_commands::reg:
103 if(cmd.type == CMDtype::getat){
104 replies.emplace_back(drv->readReg(cmd.val));
105 }else if(cmd.type == CMDtype::setat){
106 drv->writeReg(cmd.adr,cmd.val);
107 }else{
108 return CommandStatus::ERR;
109 }
110 break;
111 default:
112 return CommandStatus::NOT_FOUND;
113 }
114 return CommandStatus::OK;
115}

122void TMCDebugBridge::cdcRcv(char* Buf, uint32_t *Len){
123 std::vector<uint8_t> dat;
124 dat.assign(Buf,Buf+*Len);
125 if(*Len == 9 && *Buf == 1){
126 uint8_t crc = checksum(&dat,*Len-1);
127
128 //uint8_t id = dat[0];
129 uint8_t cmd = dat[1];
130 uint8_t addr = dat[2];
131
132 if(dat[dat.size()-1] != crc){
133 //fail
134 return;
135 }
136
137 if(cmd == 148){ // read reg
138 uint8_t buf[4];
139 uint8_t rpl[4] = {2,1,0x64,0x94};
140// tmcReadRegRaw(addr, buf);
141 uint32_t bufI = drv->readReg(addr);
142 bufI = __REV(bufI);
143 memcpy(buf,&bufI,4);
144 std::vector<uint8_t> rx_data;
145
146 rx_data.insert(rx_data.begin(), &rpl[0],&rpl[4]);
147 rx_data.insert(rx_data.end(),buf, buf+4);
148 rx_data.push_back(checksum(&rx_data,8));
149 sendCdc((char*)rx_data.data(), 9);
150
151 }else if(cmd == 146){ // write
152
153 uint32_t ndat;
154 memcpy(&ndat,Buf+4,4);
155 drv->writeReg(addr,__REV(ndat));
156 std::vector<uint8_t> repl({2,1,0x64,0x92,dat[3],dat[4],dat[5],dat[6]});
157 repl.push_back(checksum(&repl,8));
158 sendCdc((char*)repl.data(), 9);
159 }else if(cmd == 143){
160 std::vector<uint8_t> repl(8,0);
161 if(addr == 3){
162 repl.assign({2,1,40,0x8f,0,0,0,2});
163 }else if(addr == 4){
164 repl.assign({2,1,40,0x8f,2,6,2,2});
165 }
166 repl.push_back(checksum(&repl,8));
167 sendCdc((char*)repl.data(), 9);
168 }else if(cmd == 10){ // Get global parameter.
169 std::vector<uint8_t> repl({2,1,64,0x0A,0,0,0,0});
170 if(addr == 5){
171 repl[7] = 2; // hwid
172 }else if(addr == 2){
173 repl[7] = 1; // active
174 }
175 repl.push_back(checksum(&repl,8));
176 sendCdc((char*)repl.data(), 9);
177 }else if(cmd == 0x0F){ // Get input
178 std::vector<uint8_t> repl({2,1,64,0x0F,0,0,0,0});
179 if(addr == 5){ // Voltage
180 uint16_t v = getIntV()/100;
181 repl[7] = v & 0xff;
182 repl[6] = (v>>8) & 0xff;
183 }
184 repl.push_back(checksum(&repl,8));
185 sendCdc((char*)repl.data(), 9);
186 }else if(cmd == 0x88){
187 char version[9] = {2,'0','0','1','5','V','3','0','7'}; // Version string
188 if(addr == 0){
189 sendCdc(version, 9);
190 }else if(addr == 1){ // Version binary
191 // module version high
192 uint32_t tmpVal = (uint8_t) version[1] - '0';
193 std::vector<uint8_t> repl = {2,0x01,0x64,0x88,0x00,0,0,0};
194 tmpVal *= 10;
195 tmpVal += (uint8_t) version[2] - '0';
196 repl[4] = tmpVal;
197
198 // module version low
199 tmpVal = (uint8_t) version[3] - '0';
200 tmpVal *= 10;
201 tmpVal += (uint8_t) version[4] - '0';
202 repl[5] = tmpVal;
203
204 // fw version high
205 repl[6] = (uint8_t) version[6] - '0';
206
207 // fw version low
208 tmpVal = (uint8_t) version[7] - '0';
209 tmpVal *= 10;
210 tmpVal += (uint8_t) version[8] - '0';
211 repl[7] = tmpVal;
212 repl.push_back(checksum(&repl,8));
213 sendCdc((char*)repl.data(), 9);
214 }
215
216 }
217 }else{
218 FFBoardMain::cdcRcv(Buf, Len);
219 }
220}
