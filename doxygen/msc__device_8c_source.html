78static inline uint32_t rdwr10_get_lba(uint8_t const command[])
79{
80 // read10 & write10 has the same format
81 scsi_write10_t* p_rdwr10 = (scsi_write10_t*) command;
82
83 // copy first to prevent mis-aligned access
84 uint32_t lba;
85 // use offsetof to avoid pointer to the odd/misaligned address
86 memcpy(&lba, (uint8_t*) p_rdwr10 + offsetof(scsi_write10_t, lba), 4);
87
88 // lba is in Big Endian format
89 return tu_ntohl(lba);
90}
91
92static inline uint16_t rdwr10_get_blockcount(uint8_t const command[])
93{
94 // read10 & write10 has the same format
95 scsi_write10_t* p_rdwr10 = (scsi_write10_t*) command;
96
97 // copy first to prevent mis-aligned access
98 uint16_t block_count;
99 // use offsetof to avoid pointer to the odd/misaligned address
100 memcpy(&block_count, (uint8_t*) p_rdwr10 + offsetof(scsi_write10_t, block_count), 2);
101
102 return tu_ntohs(block_count);
103}

161uint16_t mscd_open(uint8_t rhport, tusb_desc_interface_t const * itf_desc, uint16_t max_len)
162{
163 // only support SCSI's BOT protocol
164 TU_VERIFY(TUSB_CLASS_MSC == itf_desc->bInterfaceClass &&
165 MSC_SUBCLASS_SCSI == itf_desc->bInterfaceSubClass &&
166 MSC_PROTOCOL_BOT == itf_desc->bInterfaceProtocol, 0);
167
168 // msc driver length is fixed
169 uint16_t const drv_len = sizeof(tusb_desc_interface_t) + 2*sizeof(tusb_desc_endpoint_t);
170
171 // Max length mus be at least 1 interface + 2 endpoints
172 TU_ASSERT(max_len >= drv_len, 0);
173
174 mscd_interface_t * p_msc = &_mscd_itf;
175 p_msc->itf_num = itf_desc->bInterfaceNumber;
176
177 // Open endpoint pair
178 TU_ASSERT( usbd_open_edpt_pair(rhport, tu_desc_next(itf_desc), 2, TUSB_XFER_BULK, &p_msc->ep_out, &p_msc->ep_in), 0 );
179
180 // Prepare for Command Block Wrapper
181 if ( !usbd_edpt_xfer(rhport, p_msc->ep_out, (uint8_t*) &p_msc->cbw, sizeof(msc_cbw_t)) )
182 {
183 TU_LOG1_FAILED();
184 TU_BREAKPOINT();
185 }
186
187 return drv_len;
188}

190// Invoked when a control transfer occurred on an interface of this class
191// Driver response accordingly to the request and the transfer stage (setup/data/ack)
192// return false to stall control endpoint (e.g unsupported request)
193bool mscd_control_xfer_cb(uint8_t rhport, uint8_t stage, tusb_control_request_t const * p_request)
194{
195 // nothing to do with DATA & ACK stage
196 if (stage != CONTROL_STAGE_SETUP) return true;
197
198 // Handle class request only
199 TU_VERIFY(p_request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS);
200
201 switch ( p_request->bRequest )
202 {
203 case MSC_REQ_RESET:
204 // TODO: Actually reset interface.
205 tud_control_status(rhport, p_request);
206 break;
207
208 case MSC_REQ_GET_MAX_LUN:
209 {
210 uint8_t maxlun = 1;
211 if (tud_msc_get_maxlun_cb) maxlun = tud_msc_get_maxlun_cb();
212 TU_VERIFY(maxlun);
213
214 // MAX LUN is minus 1 by specs
215 maxlun--;
216
217 tud_control_xfer(rhport, p_request, &maxlun, 1);
218 }
219 break;
220
221 default: return false; // stall unsupported request
222 }
223
224 return true;
225}

227bool mscd_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t event, uint32_t xferred_bytes)
228{
229 mscd_interface_t* p_msc = &_mscd_itf;
230 msc_cbw_t const * p_cbw = &p_msc->cbw;
231 msc_csw_t * p_csw = &p_msc->csw;
232
233 switch (p_msc->stage)
234 {
235 case MSC_STAGE_CMD:
236 //------------- new CBW received -------------//
237 // Complete IN while waiting for CMD is usually Status of previous SCSI op, ignore it
238 if(ep_addr != p_msc->ep_out) return true;
239
240 TU_ASSERT( event == XFER_RESULT_SUCCESS &&
241 xferred_bytes == sizeof(msc_cbw_t) && p_cbw->signature == MSC_CBW_SIGNATURE );
242
243 TU_LOG2(" SCSI Command: %s\r\n", tu_lookup_find(&_msc_scsi_cmd_table, p_cbw->command[0]));
244 // TU_LOG2_MEM(p_cbw, xferred_bytes, 2);
245
246 p_csw->signature = MSC_CSW_SIGNATURE;
247 p_csw->tag = p_cbw->tag;
248 p_csw->data_residue = 0;
249
250 /*------------- Parse command and prepare DATA -------------*/
251 p_msc->stage = MSC_STAGE_DATA;
252 p_msc->total_len = p_cbw->total_bytes;
253 p_msc->xferred_len = 0;
254
255 if (SCSI_CMD_READ_10 == p_cbw->command[0])
256 {
257 proc_read10_cmd(rhport, p_msc);
258 }
259 else if (SCSI_CMD_WRITE_10 == p_cbw->command[0])
260 {
261 proc_write10_cmd(rhport, p_msc);
262 }
263 else
264 {
265 // For other SCSI commands
266 // 1. OUT : queue transfer (invoke app callback after done)
267 // 2. IN & Zero: Process if is built-in, else Invoke app callback. Skip DATA if zero length
268 if ( (p_cbw->total_bytes > 0 ) && !tu_bit_test(p_cbw->dir, 7) )
269 {
270 // queue transfer
271 TU_ASSERT( usbd_edpt_xfer(rhport, p_msc->ep_out, _mscd_buf, p_msc->total_len) );
272 }else
273 {
274 int32_t resplen;
275
276 // First process if it is a built-in commands
277 resplen = proc_builtin_scsi(p_cbw->lun, p_cbw->command, _mscd_buf, sizeof(_mscd_buf));
278
279 // Not built-in, invoke user callback
280 if ( (resplen < 0) && (p_msc->sense_key == 0) )
281 {
282 resplen = tud_msc_scsi_cb(p_cbw->lun, p_cbw->command, _mscd_buf, p_msc->total_len);
283 }
284
285 if ( resplen < 0 )
286 {
287 p_msc->total_len = 0;
288 p_csw->status = MSC_CSW_STATUS_FAILED;
289 p_msc->stage = MSC_STAGE_STATUS;
290
291 // failed but senskey is not set: default to Illegal Request
292 if ( p_msc->sense_key == 0 ) tud_msc_set_sense(p_cbw->lun, SCSI_SENSE_ILLEGAL_REQUEST, 0x20, 0x00);
293
294 // Stall bulk In if needed
295 if (p_cbw->total_bytes) usbd_edpt_stall(rhport, p_msc->ep_in);
296 }
297 else
298 {
299 p_msc->total_len = (uint32_t) resplen;
300 p_csw->status = MSC_CSW_STATUS_PASSED;
301
302 if (p_msc->total_len)
303 {
304 TU_ASSERT( p_cbw->total_bytes >= p_msc->total_len ); // cannot return more than host expect
305 TU_ASSERT( usbd_edpt_xfer(rhport, p_msc->ep_in, _mscd_buf, p_msc->total_len) );
306 }else
307 {
308 p_msc->stage = MSC_STAGE_STATUS;
309 }
310 }
311 }
312 }
313 break;
314
315 case MSC_STAGE_DATA:
316 TU_LOG2(" SCSI Data\r\n");
317 //TU_LOG2_MEM(_mscd_buf, xferred_bytes, 2);
318
319 // OUT transfer, invoke callback if needed
320 if ( !tu_bit_test(p_cbw->dir, 7) )
321 {
322 if ( SCSI_CMD_WRITE_10 != p_cbw->command[0] )
323 {
324 int32_t cb_result = tud_msc_scsi_cb(p_cbw->lun, p_cbw->command, _mscd_buf, p_msc->total_len);
325
326 if ( cb_result < 0 )
327 {
328 p_csw->status = MSC_CSW_STATUS_FAILED;
329 tud_msc_set_sense(p_cbw->lun, SCSI_SENSE_ILLEGAL_REQUEST, 0x20, 0x00); // Sense = Invalid Command Operation
330 }else
331 {
332 p_csw->status = MSC_CSW_STATUS_PASSED;
333 }
334 }
335 else
336 {
337 uint16_t const block_sz = p_cbw->total_bytes / rdwr10_get_blockcount(p_cbw->command);
338
339 // Adjust lba with transferred bytes
340 uint32_t const lba = rdwr10_get_lba(p_cbw->command) + (p_msc->xferred_len / block_sz);
341
342 // Application can consume smaller bytes
343 int32_t nbytes = tud_msc_write10_cb(p_cbw->lun, lba, p_msc->xferred_len % block_sz, _mscd_buf, xferred_bytes);
344
345 if ( nbytes < 0 )
346 {
347 // negative means error -> skip to status phase, status in CSW set to failed
348 p_csw->data_residue = p_cbw->total_bytes - p_msc->xferred_len;
349 p_csw->status = MSC_CSW_STATUS_FAILED;
350 p_msc->stage = MSC_STAGE_STATUS;
351
352 tud_msc_set_sense(p_cbw->lun, SCSI_SENSE_ILLEGAL_REQUEST, 0x20, 0x00); // Sense = Invalid Command Operation
353 break;
354 }else
355 {
356 // Application consume less than what we got (including zero)
357 if ( nbytes < (int32_t) xferred_bytes )
358 {
359 if ( nbytes > 0 )
360 {
361 p_msc->xferred_len += nbytes;
362 memmove(_mscd_buf, _mscd_buf+nbytes, xferred_bytes-nbytes);
363 }
364
365 // simulate an transfer complete with adjusted parameters --> this driver callback will fired again
366 dcd_event_xfer_complete(rhport, p_msc->ep_out, xferred_bytes-nbytes, XFER_RESULT_SUCCESS, false);
367
368 return true; // skip the rest
369 }
370 else
371 {
372 // Application consume all bytes in our buffer. Nothing to do, process with normal flow
373 }
374 }
375 }
376 }
377
378 // Accumulate data so far
379 p_msc->xferred_len += xferred_bytes;
380
381 if ( p_msc->xferred_len >= p_msc->total_len )
382 {
383 // Data Stage is complete
384 p_msc->stage = MSC_STAGE_STATUS;
385 }
386 else
387 {
388 // READ10 & WRITE10 Can be executed with large bulk of data e.g write 8K bytes (several flash write)
389 // We break it into multiple smaller command whose data size is up to CFG_TUD_MSC_EP_BUFSIZE
390 if (SCSI_CMD_READ_10 == p_cbw->command[0])
391 {
392 proc_read10_cmd(rhport, p_msc);
393 }
394 else if (SCSI_CMD_WRITE_10 == p_cbw->command[0])
395 {
396 proc_write10_cmd(rhport, p_msc);
397 }else
398 {
399 // No other command take more than one transfer yet -> unlikely error
400 TU_BREAKPOINT();
401 }
402 }
403 break;
404
405 case MSC_STAGE_STATUS:
406 // processed immediately after this switch, supposedly to be empty
407 break;
408
409 case MSC_STAGE_STATUS_SENT:
410 // Wait for the Status phase to complete
411 if( (ep_addr == p_msc->ep_in) && (xferred_bytes == sizeof(msc_csw_t)) )
412 {
413 TU_LOG2(" SCSI Status: %u\r\n", p_csw->status);
414 // TU_LOG2_MEM(p_csw, xferred_bytes, 2);
415
416 // Invoke complete callback if defined
417 // Note: There is racing issue with samd51 + qspi flash testing with arduino
418 // if complete_cb() is invoked after queuing the status.
419 switch(p_cbw->command[0])
420 {
421 case SCSI_CMD_READ_10:
422 if ( tud_msc_read10_complete_cb ) tud_msc_read10_complete_cb(p_cbw->lun);
423 break;
424
425 case SCSI_CMD_WRITE_10:
426 if ( tud_msc_write10_complete_cb ) tud_msc_write10_complete_cb(p_cbw->lun);
427 break;
428
429 default:
430 if ( tud_msc_scsi_complete_cb ) tud_msc_scsi_complete_cb(p_cbw->lun, p_cbw->command);
431 break;
432 }
433
434 // Move to default CMD stage
435 p_msc->stage = MSC_STAGE_CMD;
436
437 // Queue for the next CBW
438 TU_ASSERT( usbd_edpt_xfer(rhport, p_msc->ep_out, (uint8_t*) &p_msc->cbw, sizeof(msc_cbw_t)) );
439 }
440 break;
441
442 default : break;
443 }
444
445 if ( p_msc->stage == MSC_STAGE_STATUS )
446 {
447 // Either endpoints is stalled, need to wait until it is cleared by host
448 if ( usbd_edpt_stalled(rhport, p_msc->ep_in) || usbd_edpt_stalled(rhport, p_msc->ep_out) )
449 {
450 // simulate an transfer complete with adjusted parameters --> this driver callback will fired again
451 // and response with status phase after halted endpoints are cleared.
452 // note: use ep_out to prevent confusing with STATUS complete
453 dcd_event_xfer_complete(rhport, p_msc->ep_out, 0, XFER_RESULT_SUCCESS, false);
454 }
455 else
456 {
457 // Move to Status Sent stage
458 p_msc->stage = MSC_STAGE_STATUS_SENT;
459
460 // Send SCSI Status
461 TU_ASSERT(usbd_edpt_xfer(rhport, p_msc->ep_in , (uint8_t*) &p_msc->csw, sizeof(msc_csw_t)));
462 }
463 }
464
465 return true;
466}

472// return response's length (copied to buffer). Negative if it is not an built-in command or indicate Failed status (CSW)
473// In case of a failed status, sense key must be set for reason of failure
474static int32_t proc_builtin_scsi(uint8_t lun, uint8_t const scsi_cmd[16], uint8_t* buffer, uint32_t bufsize)
475{
476 (void) bufsize; // TODO refractor later
477 int32_t resplen;
478
479 switch ( scsi_cmd[0] )
480 {
481 case SCSI_CMD_TEST_UNIT_READY:
482 resplen = 0;
483 if ( !tud_msc_test_unit_ready_cb(lun) )
484 {
485 // Failed status response
486 resplen = - 1;
487
488 // If sense key is not set by callback, default to Logical Unit Not Ready, Cause Not Reportable
489 if ( _mscd_itf.sense_key == 0 ) tud_msc_set_sense(lun, SCSI_SENSE_NOT_READY, 0x04, 0x00);
490 }
491 break;
492
493 case SCSI_CMD_START_STOP_UNIT:
494 resplen = 0;
495
496 if (tud_msc_start_stop_cb)
497 {
498 scsi_start_stop_unit_t const * start_stop = (scsi_start_stop_unit_t const *) scsi_cmd;
499 if ( !tud_msc_start_stop_cb(lun, start_stop->power_condition, start_stop->start, start_stop->load_eject) )
500 {
501 // Failed status response
502 resplen = - 1;
503
504 // If sense key is not set by callback, default to Logical Unit Not Ready, Cause Not Reportable
505 if ( _mscd_itf.sense_key == 0 ) tud_msc_set_sense(lun, SCSI_SENSE_NOT_READY, 0x04, 0x00);
506 }
507 }
508 break;
509
510 case SCSI_CMD_READ_CAPACITY_10:
511 {
512 uint32_t block_count;
513 uint32_t block_size;
514 uint16_t block_size_u16;
515
516 tud_msc_capacity_cb(lun, &block_count, &block_size_u16);
517 block_size = (uint32_t) block_size_u16;
518
519 // Invalid block size/count from callback, possibly unit is not ready
520 // stall this request, set sense key to NOT READY
521 if (block_count == 0 || block_size == 0)
522 {
523 resplen = -1;
524
525 // If sense key is not set by callback, default to Logical Unit Not Ready, Cause Not Reportable
526 if ( _mscd_itf.sense_key == 0 ) tud_msc_set_sense(lun, SCSI_SENSE_NOT_READY, 0x04, 0x00);
527 }else
528 {
529 scsi_read_capacity10_resp_t read_capa10;
530
531 read_capa10.last_lba = tu_htonl(block_count-1);
532 read_capa10.block_size = tu_htonl(block_size);
533
534 resplen = sizeof(read_capa10);
535 memcpy(buffer, &read_capa10, resplen);
536 }
537 }
538 break;
539
540 case SCSI_CMD_READ_FORMAT_CAPACITY:
541 {
542 scsi_read_format_capacity_data_t read_fmt_capa =
543 {
544 .list_length = 8,
545 .block_num = 0,
546 .descriptor_type = 2, // formatted media
547 .block_size_u16 = 0
548 };
549
550 uint32_t block_count;
551 uint16_t block_size;
552
553 tud_msc_capacity_cb(lun, &block_count, &block_size);
554
555 // Invalid block size/count from callback, possibly unit is not ready
556 // stall this request, set sense key to NOT READY
557 if (block_count == 0 || block_size == 0)
558 {
559 resplen = -1;
560
561 // If sense key is not set by callback, default to Logical Unit Not Ready, Cause Not Reportable
562 if ( _mscd_itf.sense_key == 0 ) tud_msc_set_sense(lun, SCSI_SENSE_NOT_READY, 0x04, 0x00);
563 }else
564 {
565 read_fmt_capa.block_num = tu_htonl(block_count);
566 read_fmt_capa.block_size_u16 = tu_htons(block_size);
567
568 resplen = sizeof(read_fmt_capa);
569 memcpy(buffer, &read_fmt_capa, resplen);
570 }
571 }
572 break;
573
574 case SCSI_CMD_INQUIRY:
575 {
576 scsi_inquiry_resp_t inquiry_rsp =
577 {
578 .is_removable = 1,
579 .version = 2,
580 .response_data_format = 2,
581 };
582
583 // vendor_id, product_id, product_rev is space padded string
584 memset(inquiry_rsp.vendor_id , ' ', sizeof(inquiry_rsp.vendor_id));
585 memset(inquiry_rsp.product_id , ' ', sizeof(inquiry_rsp.product_id));
586 memset(inquiry_rsp.product_rev, ' ', sizeof(inquiry_rsp.product_rev));
587
588 tud_msc_inquiry_cb(lun, inquiry_rsp.vendor_id, inquiry_rsp.product_id, inquiry_rsp.product_rev);
589
590 resplen = sizeof(inquiry_rsp);
591 memcpy(buffer, &inquiry_rsp, resplen);
592 }
593 break;
594
595 case SCSI_CMD_MODE_SENSE_6:
596 {
597 scsi_mode_sense6_resp_t mode_resp =
598 {
599 .data_len = 3,
600 .medium_type = 0,
601 .write_protected = false,
602 .reserved = 0,
603 .block_descriptor_len = 0 // no block descriptor are included
604 };
605
606 bool writable = true;
607 if (tud_msc_is_writable_cb) {
608 writable = tud_msc_is_writable_cb(lun);
609 }
610 mode_resp.write_protected = !writable;
611
612 resplen = sizeof(mode_resp);
613 memcpy(buffer, &mode_resp, resplen);
614 }
615 break;
616
617 case SCSI_CMD_REQUEST_SENSE:
618 {

620 {
621 .response_code = 0x70,
622 .valid = 1
623 };
624

628 sense_rsp.add_sense_code = _mscd_itf.add_sense_code;

645static void proc_read10_cmd(uint8_t rhport, mscd_interface_t* p_msc)
646{
647 msc_cbw_t const * p_cbw = &p_msc->cbw;
648 msc_csw_t * p_csw = &p_msc->csw;
649
650 uint16_t const block_cnt = rdwr10_get_blockcount(p_cbw->command);
651 TU_ASSERT(block_cnt, ); // prevent div by zero
652
653 uint16_t const block_sz = p_cbw->total_bytes / block_cnt;
654 TU_ASSERT(block_sz, ); // prevent div by zero
655
656 // Adjust lba with transferred bytes
657 uint32_t const lba = rdwr10_get_lba(p_cbw->command) + (p_msc->xferred_len / block_sz);
658
659 // remaining bytes capped at class buffer
660 int32_t nbytes = (int32_t) tu_min32(sizeof(_mscd_buf), p_cbw->total_bytes-p_msc->xferred_len);
661
662 // Application can consume smaller bytes
663 nbytes = tud_msc_read10_cb(p_cbw->lun, lba, p_msc->xferred_len % block_sz, _mscd_buf, (uint32_t) nbytes);
664
665 if ( nbytes < 0 )
666 {
667 // negative means error -> pipe is stalled & status in CSW set to failed
668 p_csw->data_residue = p_cbw->total_bytes - p_msc->xferred_len;
669 p_csw->status = MSC_CSW_STATUS_FAILED;
670
671 tud_msc_set_sense(p_cbw->lun, SCSI_SENSE_ILLEGAL_REQUEST, 0x20, 0x00); // Sense = Invalid Command Operation
672 usbd_edpt_stall(rhport, p_msc->ep_in);
673 }
674 else if ( nbytes == 0 )
675 {
676 // zero means not ready -> simulate an transfer complete so that this driver callback will fired again
677 dcd_event_xfer_complete(rhport, p_msc->ep_in, 0, XFER_RESULT_SUCCESS, false);
678 }
679 else
680 {
681 TU_ASSERT( usbd_edpt_xfer(rhport, p_msc->ep_in, _mscd_buf, nbytes), );
682 }
683}

685static void proc_write10_cmd(uint8_t rhport, mscd_interface_t* p_msc)
686{
687 msc_cbw_t const * p_cbw = &p_msc->cbw;
688 bool writable = true;
689 if (tud_msc_is_writable_cb) {
690 writable = tud_msc_is_writable_cb(p_cbw->lun);
691 }
692 if (!writable) {
693 msc_csw_t* p_csw = &p_msc->csw;
694 p_csw->data_residue = p_cbw->total_bytes;
695 p_csw->status = MSC_CSW_STATUS_FAILED;
696
697 tud_msc_set_sense(p_cbw->lun, SCSI_SENSE_DATA_PROTECT, 0x27, 0x00); // Sense = Write protected
698 usbd_edpt_stall(rhport, p_msc->ep_out);
699 return;
700 }
701
702 // remaining bytes capped at class buffer
703 int32_t nbytes = (int32_t) tu_min32(sizeof(_mscd_buf), p_cbw->total_bytes-p_msc->xferred_len);
704
705 // Write10 callback will be called later when usb transfer complete
706 TU_ASSERT( usbd_edpt_xfer(rhport, p_msc->ep_out, _mscd_buf, nbytes), );
707}
