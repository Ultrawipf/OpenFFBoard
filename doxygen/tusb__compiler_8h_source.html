61//--------------------------------------------------------------------+
62// Compiler porting with Attribute and Endian
63//--------------------------------------------------------------------+
64#if defined(__GNUC__)
65 #define TU_ATTR_ALIGNED(Bytes) __attribute__ ((aligned(Bytes)))
66 #define TU_ATTR_SECTION(sec_name) __attribute__ ((section(#sec_name)))
67 #define TU_ATTR_PACKED __attribute__ ((packed))
68 #define TU_ATTR_PREPACKED
69 #define TU_ATTR_WEAK __attribute__ ((weak))
70 #define TU_ATTR_DEPRECATED(mess) __attribute__ ((deprecated(mess))) // warn if function with this attribute is used
71 #define TU_ATTR_UNUSED __attribute__ ((unused)) // Function/Variable is meant to be possibly unused
72 #define TU_ATTR_USED __attribute__ ((used)) // Function/Variable is meant to be used
73
74 // Endian conversion use well-known host to network (big endian) naming
75 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
76 #define TU_BYTE_ORDER TU_LITTLE_ENDIAN
77 #else
78 #define TU_BYTE_ORDER TU_BIG_ENDIAN
79 #endif
80
81 #define TU_BSWAP16(u16) (__builtin_bswap16(u16))
82 #define TU_BSWAP32(u32) (__builtin_bswap32(u32))
83
84#elif defined(__TI_COMPILER_VERSION__)
85 #define TU_ATTR_ALIGNED(Bytes) __attribute__ ((aligned(Bytes)))
86 #define TU_ATTR_SECTION(sec_name) __attribute__ ((section(#sec_name)))
87 #define TU_ATTR_PACKED __attribute__ ((packed))
88 #define TU_ATTR_PREPACKED
89 #define TU_ATTR_WEAK __attribute__ ((weak))
90 #define TU_ATTR_DEPRECATED(mess) __attribute__ ((deprecated(mess))) // warn if function with this attribute is used
91 #define TU_ATTR_UNUSED __attribute__ ((unused)) // Function/Variable is meant to be possibly unused
92 #define TU_ATTR_USED __attribute__ ((used))
93
94 // __BYTE_ORDER is defined in the TI ARM compiler, but not MSP430 (which is little endian)
95 #if ((__BYTE_ORDER__) == (__ORDER_LITTLE_ENDIAN__)) || defined(__MSP430__)
96 #define TU_BYTE_ORDER TU_LITTLE_ENDIAN
97 #else
98 #define TU_BYTE_ORDER TU_BIG_ENDIAN
99 #endif
100
101 #define TU_BSWAP16(u16) (__builtin_bswap16(u16))
102 #define TU_BSWAP32(u32) (__builtin_bswap32(u32))
103
104#elif defined(__ICCARM__)
105 #include <intrinsics.h>
106 #define TU_ATTR_ALIGNED(Bytes) __attribute__ ((aligned(Bytes)))
107 #define TU_ATTR_SECTION(sec_name) __attribute__ ((section(#sec_name)))
108 #define TU_ATTR_PACKED __attribute__ ((packed))
109 #define TU_ATTR_PREPACKED
110 #define TU_ATTR_WEAK __attribute__ ((weak))
111 #define TU_ATTR_DEPRECATED(mess) __attribute__ ((deprecated(mess))) // warn if function with this attribute is used
112 #define TU_ATTR_UNUSED __attribute__ ((unused)) // Function/Variable is meant to be possibly unused
113 #define TU_ATTR_USED __attribute__ ((used)) // Function/Variable is meant to be used
114
115 // Endian conversion use well-known host to network (big endian) naming
116 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
117 #define TU_BYTE_ORDER TU_LITTLE_ENDIAN
118 #else
119 #define TU_BYTE_ORDER TU_BIG_ENDIAN
120 #endif
121
122 #define TU_BSWAP16(u16) (__iar_builtin_REV16(u16))
123 #define TU_BSWAP32(u32) (__iar_builtin_REV(u32))
124#else
125 #error "Compiler attribute porting is required"
126#endif

128#if (TU_BYTE_ORDER == TU_LITTLE_ENDIAN)
