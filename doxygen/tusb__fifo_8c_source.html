65bool tu_fifo_config(tu_fifo_t *f, void* buffer, uint16_t depth, uint16_t item_size, bool overwritable)
66{
67 if (depth > 0x8000) return false; // Maximum depth is 2^15 items
68
69 tu_fifo_lock(f);
70
71 f->buffer = (uint8_t*) buffer;
72 f->depth = depth;
73 f->item_size = item_size;
74 f->overwritable = overwritable;
75
76 f->max_pointer_idx = 2*depth - 1; // Limit index space to 2*depth - this allows for a fast "modulo" calculation but limits the maximum depth to 2^16/2 = 2^15 and buffer overflows are detectable only if overflow happens once (important for unsupervised DMA applications)
77 f->non_used_index_space = UINT16_MAX - f->max_pointer_idx;
78
79 f->rd_idx = f->wr_idx = 0;
80
81 tu_fifo_unlock(f);
82
83 return true;
84}

88static inline uint16_t _ff_mod(uint16_t idx, uint16_t depth)
89{
90 while ( idx >= depth) idx -= depth;
91 return idx;
92}

94// send one item to FIFO WITHOUT updating write pointer
95static inline void _ff_push(tu_fifo_t* f, void const * data, uint16_t wRel)
96{
97 memcpy(f->buffer + (wRel * f->item_size), data, f->item_size);
98}
99
100// send n items to FIFO WITHOUT updating write pointer
101static void _ff_push_n(tu_fifo_t* f, void const * data, uint16_t n, uint16_t wRel)
102{
103 if(wRel + n <= f->depth) // Linear mode only
104 {
105 memcpy(f->buffer + (wRel * f->item_size), data, n*f->item_size);
106 }
107 else // Wrap around
108 {
109 uint16_t nLin = f->depth - wRel;
110
111 // Write data to linear part of buffer
112 memcpy(f->buffer + (wRel * f->item_size), data, nLin*f->item_size);
113
114 // Write data wrapped around
115 memcpy(f->buffer, ((uint8_t const*) data) + nLin*f->item_size, (n - nLin) * f->item_size);
116 }
117}
118
119// get one item from FIFO WITHOUT updating read pointer
120static inline void _ff_pull(tu_fifo_t* f, void * p_buffer, uint16_t rRel)
121{
122 memcpy(p_buffer, f->buffer + (rRel * f->item_size), f->item_size);
123}
124
125// get n items from FIFO WITHOUT updating read pointer
126static void _ff_pull_n(tu_fifo_t* f, void * p_buffer, uint16_t n, uint16_t rRel)
127{
128 if(rRel + n <= f->depth) // Linear mode only
129 {
130 memcpy(p_buffer, f->buffer + (rRel * f->item_size), n*f->item_size);
131 }
132 else // Wrap around
133 {
134 uint16_t nLin = f->depth - rRel;
135
136 // Read data from linear part of buffer
137 memcpy(p_buffer, f->buffer + (rRel * f->item_size), nLin*f->item_size);
138
139 // Read data wrapped part
140 memcpy((uint8_t*)p_buffer + nLin*f->item_size, f->buffer, (n - nLin) * f->item_size);
141 }
142}

144// Advance an absolute pointer
145static uint16_t advance_pointer(tu_fifo_t* f, uint16_t p, uint16_t offset)
146{
147 // We limit the index space of p such that a correct wrap around happens
148 // Check for a wrap around or if we are in unused index space - This has to be checked first!! We are exploiting the wrap around to the correct index
149 if ((p > p + offset) || (p + offset > f->max_pointer_idx))
150 {
151 p = (p + offset) + f->non_used_index_space;
152 }
153 else
154 {
155 p += offset;
156 }
157 return p;
158}
159
160// Backward an absolute pointer
161static uint16_t backward_pointer(tu_fifo_t* f, uint16_t p, uint16_t offset)
162{
163 // We limit the index space of p such that a correct wrap around happens
164 // Check for a wrap around or if we are in unused index space - This has to be checked first!! We are exploiting the wrap around to the correct index
165 if ((p < p - offset) || (p - offset > f->max_pointer_idx))
166 {
167 p = (p - offset) - f->non_used_index_space;
168 }
169 else
170 {
171 p -= offset;
172 }
173 return p;
174}
175
176// get relative from absolute pointer
177static uint16_t get_relative_pointer(tu_fifo_t* f, uint16_t p, uint16_t offset)
178{
179 return _ff_mod(advance_pointer(f, p, offset), f->depth);
180}

182// Works on local copies of w and r
183static inline uint16_t _tu_fifo_count(tu_fifo_t* f, uint16_t wAbs, uint16_t rAbs)
184{
185 uint16_t cnt = wAbs-rAbs;
186
187 // In case we have non-power of two depth we need a further modification
188 if (rAbs > wAbs) cnt -= f->non_used_index_space;
189
190 return cnt;
191}

193// Works on local copies of w and r
194static inline bool _tu_fifo_empty(uint16_t wAbs, uint16_t rAbs)
195{
196 return wAbs == rAbs;
197}

199// Works on local copies of w and r
200static inline bool _tu_fifo_full(tu_fifo_t* f, uint16_t wAbs, uint16_t rAbs)
201{
202 return (_tu_fifo_count(f, wAbs, rAbs) == f->depth);
203}
204
205// Works on local copies of w and r
206// BE AWARE - THIS FUNCTION MIGHT NOT GIVE A CORRECT ANSWERE IN CASE WRITE POINTER "OVERFLOWS"
207// Only one overflow is allowed for this function to work e.g. if depth = 100, you must not
208// write more than 2*depth-1 items in one rush without updating write pointer. Otherwise
209// write pointer wraps and you pointer states are messed up. This can only happen if you
210// use DMAs, write functions do not allow such an error.
211static inline bool _tu_fifo_overflowed(tu_fifo_t* f, uint16_t wAbs, uint16_t rAbs)
212{
213 return (_tu_fifo_count(f, wAbs, rAbs) > f->depth);
214}
215
216// Works on local copies of w
217// For more details see _tu_fifo_overflow()!
218static inline void _tu_fifo_correct_read_pointer(tu_fifo_t* f, uint16_t wAbs)
219{
220 f->rd_idx = backward_pointer(f, wAbs, f->depth);
221}

223// Works on local copies of w and r
224// Must be protected by mutexes since in case of an overflow read pointer gets modified
225static bool _tu_fifo_peek_at(tu_fifo_t* f, uint16_t offset, void * p_buffer, uint16_t wAbs, uint16_t rAbs)
226{
227 uint16_t cnt = _tu_fifo_count(f, wAbs, rAbs);
228
229 // Check overflow and correct if required
230 if (cnt > f->depth)
231 {
232 _tu_fifo_correct_read_pointer(f, wAbs);
233 cnt = f->depth;
234 }
235
236 // Skip beginning of buffer
237 if (cnt == 0 || offset >= cnt) return false;
238
239 uint16_t rRel = get_relative_pointer(f, rAbs, offset);
240
241 // Peek data
242 _ff_pull(f, p_buffer, rRel);
243
244 return true;
245}

247// Works on local copies of w and r
248// Must be protected by mutexes since in case of an overflow read pointer gets modified
249static uint16_t _tu_fifo_peek_at_n(tu_fifo_t* f, uint16_t offset, void * p_buffer, uint16_t n, uint16_t wAbs, uint16_t rAbs)
250{
251 uint16_t cnt = _tu_fifo_count(f, wAbs, rAbs);
252
253 // Check overflow and correct if required
254 if (cnt > f->depth)
255 {
256 _tu_fifo_correct_read_pointer(f, wAbs);
257 rAbs = f->rd_idx;
258 cnt = f->depth;
259 }
260
261 // Skip beginning of buffer
262 if (cnt == 0 || offset >= cnt) return 0;
263
264 // Check if we can read something at and after offset - if too less is available we read what remains
265 cnt -= offset;
266 if (cnt < n) {
267 if (cnt == 0) return 0;
268 n = cnt;
269 }
270
271 uint16_t rRel = get_relative_pointer(f, rAbs, offset);
272
273 // Peek data
274 _ff_pull_n(f, p_buffer, n, rRel);
275
276 return n;
277}

279// Works on local copies of w and r
280static inline uint16_t _tu_fifo_remaining(tu_fifo_t* f, uint16_t wAbs, uint16_t rAbs)
281{
282 return f->depth - _tu_fifo_count(f, wAbs, rAbs);

499/******************************************************************************/
514/******************************************************************************/
515bool tu_fifo_write(tu_fifo_t* f, const void * data)
516{
517 tu_fifo_lock(f);
518
519 uint16_t w = f->wr_idx;
520
521 if ( _tu_fifo_full(f, w, f->rd_idx) && !f->overwritable ) return false;
522
523 uint16_t wRel = get_relative_pointer(f, w, 0);
524
525 // Write data
526 _ff_push(f, data, wRel);
527
528 // Advance pointer
529 f->wr_idx = advance_pointer(f, w, 1);
530
531 tu_fifo_unlock(f);
532
533 return true;
534}

536/******************************************************************************/
549/******************************************************************************/
550uint16_t tu_fifo_write_n(tu_fifo_t* f, const void * data, uint16_t count)
551{
552 if ( count == 0 ) return 0;
553
554 tu_fifo_lock(f);
555
556 uint16_t w = f->wr_idx, r = f->rd_idx;
557 uint8_t const* buf8 = (uint8_t const*) data;
558
559 if (!f->overwritable)
560 {
561 // Not overwritable limit up to full
562 count = tu_min16(count, _tu_fifo_remaining(f, w, r));
563 }
564 else if (count > f->depth)
565 {
566 // Only copy last part
567 buf8 = buf8 + (count - f->depth) * f->item_size;
568 count = f->depth;
569
570 // We start writing at the read pointer's position since we fill the complete
571 // buffer and we do not want to modify the read pointer within a write function!
572 // This would end up in a race condition with read functions!
573 f->wr_idx = r;
574 }
575
576 uint16_t wRel = get_relative_pointer(f, w, 0);
577
578 // Write data
579 _ff_push_n(f, buf8, count, wRel);
580
581 // Advance pointer
582 f->wr_idx = advance_pointer(f, w, count);
583
584 tu_fifo_unlock(f);
585
586 return count;
587}
