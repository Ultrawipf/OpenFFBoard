118inline void setHidReportAxis(reportHID_t *report, uint8_t idx, int16_t val){
119 switch(idx){

121 report->X = val;

123 case 1:

125 break;
126 case 2:
127 report->Z = val;
128 break;
129 case 3:

132 case 4:
133 report->RY = val;

136 report->RZ = val;
137 break;

143 break;
144 default:
145 return;
146 }
