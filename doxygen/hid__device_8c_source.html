79bool tud_hid_n_report(uint8_t itf, uint8_t report_id, void const* report, uint8_t len)
80{
81 uint8_t const rhport = 0;
82 hidd_interface_t * p_hid = &_hidd_itf[itf];
83
84 // claim endpoint
85 TU_VERIFY( usbd_edpt_claim(rhport, p_hid->ep_in) );
86
87 // prepare data
88 if (report_id)
89 {
90 len = tu_min8(len, CFG_TUD_HID_EP_BUFSIZE-1);
91
92 p_hid->epin_buf[0] = report_id;
93 memcpy(p_hid->epin_buf+1, report, len);
94 len++;
95 }else
96 {
97 // If report id = 0, skip ID field
98 len = tu_min8(len, CFG_TUD_HID_EP_BUFSIZE);
99 memcpy(p_hid->epin_buf, report, len);
100 }
101
102 return usbd_edpt_xfer(TUD_OPT_RHPORT, p_hid->ep_in, p_hid->epin_buf, len);
103}

174uint16_t hidd_open(uint8_t rhport, tusb_desc_interface_t const * desc_itf, uint16_t max_len)
175{
176 TU_VERIFY(TUSB_CLASS_HID == desc_itf->bInterfaceClass, 0);
177
178 // len = interface + hid + n*endpoints
179 uint16_t const drv_len = sizeof(tusb_desc_interface_t) + sizeof(tusb_hid_descriptor_hid_t) + desc_itf->bNumEndpoints*sizeof(tusb_desc_endpoint_t);
180 TU_ASSERT(max_len >= drv_len, 0);
181
182 // Find available interface
183 hidd_interface_t * p_hid = NULL;
184 uint8_t hid_id;
185 for(hid_id=0; hid_id<CFG_TUD_HID; hid_id++)
186 {
187 if ( _hidd_itf[hid_id].ep_in == 0 )
188 {
189 p_hid = &_hidd_itf[hid_id];
190 break;
191 }
192 }
193 TU_ASSERT(p_hid, 0);
194
195 uint8_t const *p_desc = (uint8_t const *) desc_itf;
196
197 //------------- HID descriptor -------------//
198 p_desc = tu_desc_next(p_desc);
199 p_hid->hid_descriptor = (tusb_hid_descriptor_hid_t const *) p_desc;
200 TU_ASSERT(HID_DESC_TYPE_HID == p_hid->hid_descriptor->bDescriptorType, 0);
201
202 //------------- Endpoint Descriptor -------------//
203 p_desc = tu_desc_next(p_desc);
204 TU_ASSERT(usbd_open_edpt_pair(rhport, p_desc, desc_itf->bNumEndpoints, TUSB_XFER_INTERRUPT, &p_hid->ep_out, &p_hid->ep_in), 0);
205
206 if ( desc_itf->bInterfaceSubClass == HID_SUBCLASS_BOOT ) p_hid->boot_protocol = desc_itf->bInterfaceProtocol;
207
208 p_hid->boot_mode = false; // default mode is REPORT
209 p_hid->itf_num = desc_itf->bInterfaceNumber;
210
211 // Use offsetof to avoid pointer to the odd/misaligned address
212 memcpy(&p_hid->report_desc_len, (uint8_t*) p_hid->hid_descriptor + offsetof(tusb_hid_descriptor_hid_t, wReportLength), 2);
213
214 // Prepare for output endpoint
215 if (p_hid->ep_out)
216 {
217 if ( !usbd_edpt_xfer(rhport, p_hid->ep_out, p_hid->epout_buf, sizeof(p_hid->epout_buf)) )
218 {
219 TU_LOG1_FAILED();
220 TU_BREAKPOINT();
221 }
222 }
223
224 return drv_len;
225}

227// Invoked when a control transfer occurred on an interface of this class
228// Driver response accordingly to the request and the transfer stage (setup/data/ack)
229// return false to stall control endpoint (e.g unsupported request)
230bool hidd_control_xfer_cb (uint8_t rhport, uint8_t stage, tusb_control_request_t const * request)
231{
232 TU_VERIFY(request->bmRequestType_bit.recipient == TUSB_REQ_RCPT_INTERFACE);
233
234 uint8_t const hid_itf = get_index_by_itfnum((uint8_t) request->wIndex);
235 TU_VERIFY(hid_itf < CFG_TUD_HID);
236
237 hidd_interface_t* p_hid = &_hidd_itf[hid_itf];
238
239 if (request->bmRequestType_bit.type == TUSB_REQ_TYPE_STANDARD)
240 {
241 //------------- STD Request -------------//
242 if ( stage == CONTROL_STAGE_SETUP )
243 {
244 uint8_t const desc_type = tu_u16_high(request->wValue);
245 //uint8_t const desc_index = tu_u16_low (request->wValue);
246
247 if (request->bRequest == TUSB_REQ_GET_DESCRIPTOR && desc_type == HID_DESC_TYPE_HID)
248 {
249 TU_VERIFY(p_hid->hid_descriptor != NULL);
250 TU_VERIFY(tud_control_xfer(rhport, request, (void*) p_hid->hid_descriptor, p_hid->hid_descriptor->bLength));
251 }
252 else if (request->bRequest == TUSB_REQ_GET_DESCRIPTOR && desc_type == HID_DESC_TYPE_REPORT)
253 {
254 uint8_t const * desc_report = tud_hid_descriptor_report_cb(hid_itf);
255 tud_control_xfer(rhport, request, (void*) desc_report, p_hid->report_desc_len);
256 }
257 else
258 {
259 return false; // stall unsupported request
260 }
261 }
262 }
263 else if (request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS)
264 {
265 //------------- Class Specific Request -------------//
266 switch( request->bRequest )
267 {
268 case HID_REQ_CONTROL_GET_REPORT:
269 if ( stage == CONTROL_STAGE_SETUP )
270 {
271 uint8_t const report_type = tu_u16_high(request->wValue);
272 uint8_t const report_id = tu_u16_low(request->wValue);
273
274 uint16_t xferlen = tud_hid_get_report_cb(hid_itf, report_id, (hid_report_type_t) report_type, p_hid->epin_buf, request->wLength);
275 TU_ASSERT( xferlen > 0 );
276
277 tud_control_xfer(rhport, request, p_hid->epin_buf, xferlen);
278 }
279 break;
280
281 case HID_REQ_CONTROL_SET_REPORT:
282 if ( stage == CONTROL_STAGE_SETUP )
283 {
284 TU_VERIFY(request->wLength <= sizeof(p_hid->epout_buf));
285 tud_control_xfer(rhport, request, p_hid->epout_buf, request->wLength);
286 }
287 else if ( stage == CONTROL_STAGE_ACK )
288 {
289 uint8_t const report_type = tu_u16_high(request->wValue);
290 uint8_t const report_id = tu_u16_low(request->wValue);
291
292 tud_hid_set_report_cb(hid_itf, report_id, (hid_report_type_t) report_type, p_hid->epout_buf, request->wLength);
293 }
294 break;
295
296 case HID_REQ_CONTROL_SET_IDLE:
297 if ( stage == CONTROL_STAGE_SETUP )
298 {
299 p_hid->idle_rate = tu_u16_high(request->wValue);
300 if ( tud_hid_set_idle_cb )
301 {
302 // stall request if callback return false
303 TU_VERIFY( tud_hid_set_idle_cb( hid_itf, p_hid->idle_rate) );
304 }
305
306 tud_control_status(rhport, request);
307 }
308 break;
309
310 case HID_REQ_CONTROL_GET_IDLE:
311 if ( stage == CONTROL_STAGE_SETUP )
312 {
313 // TODO idle rate of report
314 tud_control_xfer(rhport, request, &p_hid->idle_rate, 1);
315 }
316 break;
317
318 case HID_REQ_CONTROL_GET_PROTOCOL:
319 if ( stage == CONTROL_STAGE_SETUP )
320 {
321 // 0 is Boot, 1 is Report protocol
322 uint8_t protocol = (uint8_t)(1-p_hid->boot_mode);
323 tud_control_xfer(rhport, request, &protocol, 1);
324 }
325 break;
326
327 case HID_REQ_CONTROL_SET_PROTOCOL:
328 if ( stage == CONTROL_STAGE_SETUP )
329 {
330 // 0 is Boot, 1 is Report protocol
331 p_hid->boot_mode = 1 - request->wValue;
332 tud_control_status(rhport, request);
333 }
334 else if ( stage == CONTROL_STAGE_ACK )
335 {
336 if (tud_hid_boot_mode_cb)
337 {
338 tud_hid_boot_mode_cb(hid_itf, p_hid->boot_mode);
339 }
340 }
341 break;
342
343 default: return false; // stall unsupported request
344 }
345 }else
346 {
347 return false; // stall unsupported request
348 }
349
350 return true;
351}

353bool hidd_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
354{
355 (void) result;
356
357 uint8_t itf = 0;
358 hidd_interface_t * p_hid = _hidd_itf;
359
360 // Identify which interface to use
361 for (itf = 0; itf < CFG_TUD_HID; itf++)
362 {
363 p_hid = &_hidd_itf[itf];
364 if ( (ep_addr == p_hid->ep_out) || (ep_addr == p_hid->ep_in) ) break;
365 }
366 TU_ASSERT(itf < CFG_TUD_HID);
367
368 // Sent report successfully
369 if (ep_addr == p_hid->ep_in)
370 {
371 if (tud_hid_report_complete_cb)
372 {
373 tud_hid_report_complete_cb(itf, p_hid->epin_buf, (uint8_t) xferred_bytes);
374 }
375 }
376 // Received report
377 else if (ep_addr == p_hid->ep_out)
378 {
379 tud_hid_set_report_cb(itf, 0, HID_REPORT_TYPE_INVALID, p_hid->epout_buf, xferred_bytes);
380 TU_ASSERT(usbd_edpt_xfer(rhport, p_hid->ep_out, p_hid->epout_buf, sizeof(p_hid->epout_buf)));
381 }
382
383 return true;
384}
