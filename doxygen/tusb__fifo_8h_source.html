34// Due to the use of unmasked pointers, this FIFO does not suffer from loosing
35// one item slice. Furthermore, write and read operations are completely
36// decoupled as write and read functions do not modify a common state. Henceforth,
37// writing or reading from the FIFO within an ISR is safe as long as no other
38// process (thread or ISR) interferes.
39// Also, this FIFO is ready to be used in combination with a DMA as the write and
40// read pointers can be updated from within a DMA ISR. Overflows are detectable
41// within a certain number (see tu_fifo_overflow()).
42
43// mutex is only needed for RTOS
44// for OS None, we don't get preempted
45#define CFG_FIFO_MUTEX (CFG_TUSB_OS != OPT_OS_NONE)

81#define TU_FIFO_INIT(_buffer, _depth, _type, _overwritable) \
82{ \
83 .buffer = _buffer, \
84 .depth = _depth, \
85 .item_size = sizeof(_type), \
86 .overwritable = _overwritable, \
87 .max_pointer_idx = 2*(_depth)-1, \
88 .non_used_index_space = UINT16_MAX - (2*(_depth)-1) \
89}

91#define TU_FIFO_DEF(_name, _depth, _type, _overwritable) \
