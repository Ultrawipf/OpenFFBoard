34void LocalButtons::setMask(uint32_t mask){
35 this->mask = mask;
36
37 this->btnnum = 0;

39 if(mask & (1 << i)){
40 this->btnnum++;
41 }
42 }

45uint8_t LocalButtons::readButtons(uint64_t* buf){
46 uint8_t cur_btn = 0;
47 for(uint8_t i = 0;i<this->maxButtons;i++){
48 if(mask & (0x1 << i)){
49 bool b{readButton(i)};
50 if(this->polarity){ // positive polarity
51 *buf |= b << cur_btn++;
52 }else{ // Negative polarity (normal)
53 *buf |= !b << cur_btn++;
54 }
55 }
56 }
57 return this->btnnum;
