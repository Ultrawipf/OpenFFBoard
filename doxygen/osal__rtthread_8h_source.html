48typedef struct rt_semaphore osal_semaphore_def_t;
49typedef rt_sem_t osal_semaphore_t;
50
51TU_ATTR_ALWAYS_INLINE static inline
52osal_semaphore_t osal_semaphore_create(osal_semaphore_def_t *semdef) {
53 rt_sem_init(semdef, "tusb", 0, RT_IPC_FLAG_PRIO);
54 return semdef;

57TU_ATTR_ALWAYS_INLINE static inline bool osal_semaphore_delete(osal_semaphore_t semd_hdl) {
58 return RT_EOK == rt_sem_detach(semd_hdl);
59}
60
61TU_ATTR_ALWAYS_INLINE static inline bool osal_semaphore_post(osal_semaphore_t sem_hdl, bool in_isr) {
62 (void) in_isr;
63 return rt_sem_release(sem_hdl) == RT_EOK;
64}
65
66TU_ATTR_ALWAYS_INLINE static inline bool osal_semaphore_wait(osal_semaphore_t sem_hdl, uint32_t msec) {
67 return rt_sem_take(sem_hdl, rt_tick_from_millisecond((rt_int32_t) msec)) == RT_EOK;
68}
69
70TU_ATTR_ALWAYS_INLINE static inline void osal_semaphore_reset(osal_semaphore_t const sem_hdl) {
71 rt_sem_control(sem_hdl, RT_IPC_CMD_RESET, 0);
72}

101// role device/host is used by OS NONE for mutex (disable usb isr) only
102#define OSAL_QUEUE_DEF(_int_set, _name, _depth, _type) \
103 static _type _name##_##buf[_depth]; \
104 osal_queue_def_t _name = { .depth = _depth, .item_sz = sizeof(_type), .buf = _name##_##buf };
105
106typedef struct {
107 uint16_t depth;
108 uint16_t item_sz;
109 void *buf;
110
111 struct rt_messagequeue sq;
112} osal_queue_def_t;
113
114typedef rt_mq_t osal_queue_t;
115
116TU_ATTR_ALWAYS_INLINE static inline osal_queue_t osal_queue_create(osal_queue_def_t *qdef) {
117 rt_mq_init(&(qdef->sq), "tusb", qdef->buf, qdef->item_sz,
118 qdef->item_sz * qdef->depth, RT_IPC_FLAG_PRIO);
119 return &(qdef->sq);
120}
121
122TU_ATTR_ALWAYS_INLINE static inline bool osal_queue_delete(osal_queue_t qhdl) {
123 return RT_EOK == rt_mq_detach(qhdl);
124}
125
126TU_ATTR_ALWAYS_INLINE static inline bool osal_queue_receive(osal_queue_t qhdl, void *data, uint32_t msec) {
127 rt_tick_t tick = rt_tick_from_millisecond((rt_int32_t) msec);
128#if RT_VERSION_MAJOR >= 5
129 return rt_mq_recv(qhdl, data, qhdl->msg_size, tick) > 0;
130#else
131 return rt_mq_recv(qhdl, data, qhdl->msg_size, tick) == RT_EOK;
132#endif /* RT_VERSION_MAJOR >= 5 */
133}
134
135TU_ATTR_ALWAYS_INLINE static inline bool osal_queue_send(osal_queue_t qhdl, void const *data, bool in_isr) {
136 (void) in_isr;
137 return rt_mq_send(qhdl, (void *)data, qhdl->msg_size) == RT_EOK;
138}
139
140TU_ATTR_ALWAYS_INLINE static inline bool osal_queue_empty(osal_queue_t qhdl) {
141 return (qhdl->entry) == 0;
142}
