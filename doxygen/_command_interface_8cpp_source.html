111void StringCommandInterface::formatReply(std::string& reply,const std::vector<CommandResult>& results,const bool formatWriteAsRead){
112 //uint16_t lastId = 0xFFFF;
113 for(const CommandResult& result : results){ // All commands processed this batch
114 if(formatWriteAsRead && !(result.originalCommand.type == CMDtype::set || result.originalCommand.type == CMDtype::setat)){
115 return; // Ignore commands that should be formatted as read if they are not set commands
116 }
117
118 reply += "["; // Start marker
119 reply += StringCommandInterface::formatOriginalCommandFromResult(result.originalCommand,result.commandHandler, formatWriteAsRead);
120 reply += "|"; // Separator
121
122 if(formatWriteAsRead){
123 std::string tstr;
124 StringCommandInterface::generateReplyFromCmd(tstr,result.originalCommand);
125 reply += tstr;
126 }else{
127 if(result.type == CommandStatus::NOT_FOUND){
128 reply += "NOT_FOUND";
129 //ErrorHandler::addError(CommandInterface::cmdNotFoundError);
130 }else if(result.type == CommandStatus::OK || result.type == CommandStatus::BROADCAST){
131 if(result.reply.empty()){
132 reply += "OK";
133 }else{
134 uint16_t repliesRemaining = result.reply.size();
135 for(const CommandReply& cmdReply : result.reply){ // For all entries of this command. Normally just one
136 std::string tstr;
137 //TODO Long replies fail HERE while copying reply strings
138 StringCommandInterface::generateReplyValueString(tstr,cmdReply);
139 reply += tstr;
140 if(--repliesRemaining > 0){
141 reply += "\n"; // Separate replies with newlines
142 }
143 }
144 }
145 }else if(result.type == CommandStatus::ERR){
146 reply += "ERR";
147 }else{
148 reply += "None";
149 }
150 }
151
152 reply += "]\n"; // end marker
153 }
154}

159std::string StringCommandInterface::formatOriginalCommandFromResult(const ParsedCommand& originalCommand,CommandHandler* commandHandler,const bool formatWriteAsRead){
160
161// ClassIdentifier info = commandHandler->getInfo();
162 std::string cmdstring = commandHandler->getCommandHandlerInfo()->clsname;
163 if(originalCommand.instance != 0xFF){
164 cmdstring += "." + std::to_string(originalCommand.instance);
165 }
166 cmdstring += ".";
167
168
169 CmdHandlerCommanddef* cmdDef = commandHandler->getCommandFromId(originalCommand.cmdId); // CMD name
170 if(!cmdDef){
171 return ""; // can not find cmd. should never happen
172 }
173 cmdstring += std::string(cmdDef->cmd);
174 if(originalCommand.type == CMDtype::get || (formatWriteAsRead && originalCommand.type == CMDtype::set)){
175 cmdstring += "?";
176
177 }else if(originalCommand.type == CMDtype::getat || (formatWriteAsRead && originalCommand.type == CMDtype::setat)){ // cls.inst.cmd?
178 cmdstring += "?" + std::to_string(originalCommand.adr);
179
180 }else if(originalCommand.type == CMDtype::set){ // cls.inst.cmd?
181 cmdstring += "=" + std::to_string(originalCommand.val);
182
183 }else if(originalCommand.type == CMDtype::setat){ // cls.inst.cmd?x=y
184 cmdstring += "=" + std::to_string(originalCommand.val) + "?" + std::to_string(originalCommand.adr);
185
186 }else if(originalCommand.type == CMDtype::info){
187 cmdstring += "!";
188 }
189
190 return cmdstring;

197void StringCommandInterface::generateReplyValueString(std::string& replyPart,const CommandReply& reply){
198 //std::string replyPart;
199 if(reply.type == CommandReplyType::STRING || reply.type == CommandReplyType::STRING_OR_INT || reply.type == CommandReplyType::STRING_OR_DOUBLEINT){
200 replyPart.assign(reply.reply);
201 }else if(reply.type == CommandReplyType::INT){
202 replyPart = std::to_string(reply.val);
203 }else if(reply.type == CommandReplyType::DOUBLEINTS){
204 replyPart = std::to_string(reply.val) + ":" + std::to_string(reply.adr);
205 }else if(reply.type == CommandReplyType::ACK){
206 replyPart = "OK";
207 }
208}

252void CDC_CommandInterface::sendReplies(const std::vector<CommandResult>& results,CommandInterface* originalInterface){
253
254 if(HAL_GetTick() - lastSendTime > parserTimeout){
255 resultsBuffer.clear(); // Empty buffer because we were not able to reply in time to prevent the full buffer from blocking future commands
256 //CDCcomm::clearRemainingBuffer(0);
257 }
258
259 if( (!enableBroadcastFromOtherInterfaces && originalInterface != this) ){
260 return;
261 }
262 lastSendTime = HAL_GetTick();
263 resultsBuffer.assign(results.begin(), results.end());
264 resultsBuffer.shrink_to_fit();
265 nextFormat = originalInterface != this && originalInterface != nullptr;
266 Notify(); // Resume
267}

272bool CDC_CommandInterface::readyToSend(){
273 if(HAL_GetTick() - lastSendTime > parserTimeout && CDCcomm::remainingData(0) == 0){
274 return true;
275 }
276 return CDCcomm::remainingData(0) == 0 && resultsBuffer.empty();
277}
