212void FFBHIDMain::send_report(){
213 // Check if HID command interface wants to send something and allow that if we did not skip too many reports
214 if(!tud_hid_n_ready(0) || ((reportSendCounter++ < usb_report_rate*2) && this->hidCommands->waitingToSend())){
215 return;
216 }
217 //Try semaphore
218 if(!sourcesSem.Take(10)){
219 return;
220 }
221 // Read buttons
222 reportHID.buttons = 0; // Reset buttons
223 uint8_t shift = 0;
224 if(btns.size() != 0){
225 for(auto &btn : btns){
226 uint64_t buf = 0;
227 uint8_t amount = btn->readButtons(&buf);
228 reportHID.buttons |= buf << shift;
229 shift += amount;
230 }
231 }
232
233
234 // Encoder
235 //axes_manager->addAxesToReport(analogAxesReport, &count);
236
237 std::vector<int32_t>* axes = axes_manager->getAxisValues();
238 uint8_t count = 0;
239 for(auto val : *axes){
240 setHidReportAxis(&reportHID,count++,val);
241 }
242
243 // Fill remaining values with analog inputs
244 for(auto &ain : analog_inputs){
245 std::vector<int32_t>* axes = ain->getAxes();
246 for(int32_t val : *axes){
247 if(count >= analogAxisCount)
248 break;
249 setHidReportAxis(&reportHID,count++,val);
250 }
251 }
252 sourcesSem.Give();
253 // Fill rest
254 for(;count<analogAxisCount; count++){
255 setHidReportAxis(&reportHID,count,0);
256 }
257
258
259 /*
260 * Only send a new report if actually changed since last time or timeout and hid is ready
261 */
262 if( (reportSendCounter > 100/usb_report_rate || (memcmp(&lastReportHID,&reportHID,sizeof(reportHID_t)) != 0) ))
263 {
264
265
266 tud_hid_report(0, reinterpret_cast<uint8_t*>(&reportHID), sizeof(reportHID_t));
267 lastReportHID = reportHID;
268 reportSendCounter = 0;
269
270 }
