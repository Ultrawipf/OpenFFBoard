74FFBHIDMain::FFBHIDMain(uint8_t axisCount) :
75 btn_chooser(button_sources),analog_chooser(analog_sources)
76{
77 // Creates the required no of axis (Default 1)
78 effects_calc = std::make_unique<EffectsCalculator>();
79 axes_manager = std::make_unique<AxesManager>(&control);
80 axes_manager->setEffectsCalculator(effects_calc.get());
81// Create the USB effects handler & pass in the effects calculator
82 this->ffb = std::make_unique<HidFFB>();
83 this->ffb->setEffectsCalculator(effects_calc.get());
84
85 axes_manager->setAxisCount(axisCount);
86 restoreFlash(); // Load parameters

88
89

92 if(estopState){ // Estop pressed at startup
93 emergencyStop(!estopState);
94 lastEstop = HAL_GetTick();

141void FFBHIDMain::update(){
142 if(control.request_update_disabled) {
143 //logSerial("request update disabled");
144 control.update_disabled = true;
145 control.request_update_disabled = false;
146
147 }
148 if(control.update_disabled){
149 //logSerial("Update disabled");
150 return;
151 }
152
153 if(control.resetEncoder){
154 control.resetEncoder = false;
155 axes_manager->resetPosZero();
156 }
157
158
159
160 // TODO Emulate a SOF timer...
161 if(HAL_GetTick() - lastUsbReportTick > 0 && !control.usb_disabled){
162 lastUsbReportTick = HAL_GetTick();
163 control.usb_update_flag = true;
164 }
165
166 // If either usb or timer triggered
167 if(control.usb_update_flag || control.update_flag){
168 //debugpin.set();
169 uint32_t prio = uxTaskPriorityGet((TaskHandle_t)defaultTaskHandle);
170 vTaskPrioritySet((TaskHandle_t)defaultTaskHandle,34); // Increase priority
171 axes_manager->update();
172 control.update_flag = false;
173 if(control.usb_update_flag){
174 control.usb_update_flag = false;
175 if(++report_rate_cnt >= usb_report_rate){
176 report_rate_cnt = 0;
177 this->send_report();
178 }
179 }
180 if(!control.emergency){
181 axes_manager->updateTorque();
182
183 }else{
184 pulseClipLed();
185 }
186 //debugpin.reset();
187 vTaskPrioritySet((TaskHandle_t)defaultTaskHandle,prio); // reset priority
188 }

199void FFBHIDMain::setBtnTypes(uint16_t btntypes){
200 sourcesSem.Take();
201 this->btnsources = btntypes;
202 clearBtnTypes();
203 for(uint8_t id = 0;id<16;id++){

205 // Matching flag

207 if(btn!=nullptr)

209 }
210 }
211 sourcesSem.Give();

268void FFBHIDMain::send_report(){
269 if(!sourcesSem.Take(10)){
270 return;
271 }
272 // Read buttons
273 reportHID.buttons = 0; // Reset buttons
274 uint8_t shift = 0;
275 if(btns.size() != 0){
276 for(auto &btn : btns){
277 uint64_t buf = 0;
278 uint8_t amount = btn->readButtons(&buf);
279 reportHID.buttons |= buf << shift;
280 shift += amount;
281 }
282 }
283
284
285 // Encoder
286 //axes_manager->addAxesToReport(analogAxesReport, &count);
287
288 std::vector<int32_t>* axes = axes_manager->getAxisValues();
289 uint8_t count = 0;
290 for(auto val : *axes){
291 setHidReportAxis(&reportHID,count++,val);
292 }
293
294 // Fill remaining values with analog inputs
295 for(auto &ain : analog_inputs){
296 std::vector<int32_t>* axes = ain->getAxes();
297 for(int32_t val : *axes){
298 if(count >= analogAxisCount)
299 break;
300 setHidReportAxis(&reportHID,count++,val);
301 }
302 }
303 sourcesSem.Give();
304 // Fill rest
305 for(;count<analogAxisCount; count++){
306 setHidReportAxis(&reportHID,count,0);
307 }
308
309
310 /*
311 * Only send a new report if actually changed since last time or timeout and hid is ready
312 */
313 if( (reportSendCounter++ > 100/usb_report_rate || (memcmp(&lastReportHID,&reportHID,sizeof(reportHID_t)) != 0) )
314 && tud_hid_n_ready(0)
315 && !(reportSendCounter < usb_report_rate*2 && this->hidCommands->waitingToSend())) // Check if HID command interface wants to send something and allow that if we did not skip the last report
316 {
317
318
319 tud_hid_report(0, reinterpret_cast<uint8_t*>(&reportHID), sizeof(reportHID_t));
320 lastReportHID = reportHID;
321 reportSendCounter = 0;
322
323 }
