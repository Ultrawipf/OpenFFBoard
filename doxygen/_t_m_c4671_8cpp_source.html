142 uint16_t mconfint;
143 uint16_t abncpr = 0;
144
145 // Read flash
146 if(Flash_Read(flashAddrs.mconf, &mconfint))
147 this->conf.motconf = TMC4671::decodeMotFromInt(mconfint);

150 setCpr(abncpr);

158 Flash_Read(flashAddrs.encOffset,&encofs);
159 this->abnconf.posOffsetFromIndex = (int16_t)encofs;
160
161 Flash_Read(flashAddrs.offsetFlux, (uint16_t*)&this->maxOffsetFlux);
162
163 // Restore ADC settings
164 if(Flash_Read(flashAddrs.ADC_i0_ofs,&conf.adc_I0_offset) && Flash_Read(flashAddrs.ADC_i1_ofs,&conf.adc_I1_offset)){
165 adcSettingsStored = true; // Previous adc settings restored
166 }else{
167 recalibrationRequired = true; // Never stored

169
170// abnconf.phiEoffset = Flash_ReadDefault(flashAddrs.phieOffset,0);
171// if(abnconf.phiEoffset != 0){
172// phiErestored = true;
173// }
174
175 uint16_t miscval;
176 if(Flash_Read(flashAddrs.encA, &miscval)){
177 restoreEncHallMisc(miscval);
178 encHallRestored = true;
179 }
180 uint16_t filterval;

240bool TMC4671::initialize(){
241// active = true;
242// if(state == TMC_ControlState::uninitialized){
243// state = TMC_ControlState::Init_wait;
244// }
245 // Check if a TMC4671 is active and replies correctly
246 if(!pingDriver()){
247 ErrorHandler::addError(communicationError);
248 return false;
249 }
250
251 writeReg(1, 1);
252 if(readReg(0) == 0x00010000 && allowSlowSPI){
253 /* Slow down SPI if old TMC engineering sample is detected
254 * The first version has a high chance of glitches of the MSB
255 * when high spi speeds are used.
256 * This can cause problems for some operations.
257 */
258 pulseClipLed();
259
260 this->spiConfig.peripheral.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_64;
261 spiPort.configurePort(&this->spiConfig.peripheral);
262 ES_TMCdetected = true;
263 }
264
265 if(!ES_TMCdetected){
266 this->setPidPrecision(pidPrecision);
267 }
268
269 // Detect if tmc was previously uninitialized
270 if(startupType == TMC_StartupType::NONE){
271 if(getMotionMode() != MotionMode::stop){
272 startupType = TMC_StartupType::warmStart;
273 }else{
274 startupType = TMC_StartupType::coldStart;
275 }
276 }
277
278 // Write main constants
279
280 writeReg(0x64, 0); // No flux/torque
281 setPwm(0,conf.pwmcnt,conf.bbmL,conf.bbmH); // Set FOC @ 25khz but turn off pwm for now
282 setMotorType(conf.motconf.motor_type,conf.motconf.pole_pairs);
283 setPhiEtype(conf.motconf.phiEsource);
284 setup_HALL(hallconf); // Enables hall filter and masking
285
286 initAdc(conf.mdecA,conf.mdecB,conf.mclkA,conf.mclkB);
287 setAdcOffset(conf.adc_I0_offset, conf.adc_I1_offset);
288 setAdcScale(conf.adc_I0_scale, conf.adc_I1_scale);
289 setTorqueFilter(torqueFilterConf);
290
291 // Initial adc calibration and check without PWM if power off to get basic offsets. PWM is off!
292 if(!hasPower()){
293 if(!calibrateAdcOffset(150)){
294 changeState(TMC_ControlState::HardError); // ADC or shunt amp is broken!
295 enablePin.reset();
296 return false;
297 }
298 }
299 // brake res failsafe.
300// /*
301// * Single ended input raw value
302// * 0V = 0x7fff
303// * 4.7k / (360k+4.7k) Divider on old board.
304// * 1.5k / (71.5k+1.5k) 16.121 counts 60V new. 100V VM => 2V
305// * 13106 counts/V input.
306// */
307 setBrakeLimits(this->conf.hwconf.brakeLimLow,this->conf.hwconf.brakeLimHigh); // update limit from previously loaded constants or defaults
308
309 // Status mask
310 if(ES_TMCdetected){
311 setStatusMask(0); // ES Version status output is broken
312 }else{
313 /*
314 * Enable adc clipping and pll errors
315 */
316 statusMask.asInt = 0;
317 statusMask.flags.adc_i_clipped = 1;
318 statusMask.flags.not_PLL_locked = 1;
319 setStatusMask(statusMask);
320 }
321
322 setPids(curPids); // Write basic pids
323
324// if(hasPower()){
325// enablePin.set();
326// setPwm(TMC_PwmMode::PWM_FOC);
327// calibrateAdcOffset(400); // Calibrate ADC again with power
328// motorEnabledRequested = true;
329// }
330 //setEncoderType(conf.motconf.enctype);
331
332 // Update flags
333 readFlags(false); // Read all flags
334
335 initialized = true;
336 initTime = HAL_GetTick();
337 return initialized;
338}

344float TMC4671::getTemp(){
345 if(!this->conf.hwconf.temperatureEnabled){
346 return 0;
347 }
348 TMC4671HardwareTypeConf* hwconf = &conf.hwconf;
349
350 writeReg(0x03, 2);
351 int32_t adcval = ((readReg(0x02)) & 0xffff) - 0x7fff; // Center offset
352 adcval -= hwconf->adcOffset;
353 if(adcval <= 0){
354 return 0.0;
355 }
356 float r = hwconf->thermistor_R2 * (((float)43252 / (float)adcval)); //43252 equivalent ADC count if it was 3.3V and not 2.5V
357
358 // Beta
359 r = (1.0 / 298.15) + log(r / hwconf->thermistor_R) / hwconf->thermistor_Beta;
360 r = 1.0 / r;
361 r -= 273.15;
362 return r;
363

380void TMC4671::initializeWithPower(){
381 if(powerInitialized){
382 return;
383 }
384 powerInitialized = true;
385 // Load ADC settings
386 if(Flash_Read(flashAddrs.ADC_i0_ofs,&conf.adc_I0_offset) && Flash_Read(flashAddrs.ADC_i1_ofs,&conf.adc_I1_offset)){
387 adcSettingsStored = true; // Previous adc settings restored
388 setAdcOffset(conf.adc_I0_offset, conf.adc_I1_offset);

390
391 if(adcSettingsStored && checkAdc()){
392 adcCalibrated = true;
393 }else{
394 if(!calibrateAdcOffset(300)){
395 powerInitialized = false;
396 return; // Abort
397 }
398 }
399
400 // got power long enough. proceed to set up encoder
401 // if encoder not set up
402 enablePin.set();
403 setPwm(TMC_PwmMode::PWM_FOC); // enable foc
404 if(!encoderAligned){
405 setEncoderType(conf.motconf.enctype);
406 }else{
407 //last state
408 if(!emergency){
409 allowStateChange = true;
410 changeState(requestedState);
411 setMotionMode(lastMotionMode,true);
412 ErrorHandler::clearError(ErrorCode::undervoltage);
413 }

422void TMC4671::Run(){
423 // Main state machine
424 while(1){
425
426 if(!initialized && state != TMC_ControlState::HardError){
427 changeState(TMC_ControlState::uninitialized,true);
428 }
429
430 // check if we are in a privileged state otherwise use requested state as new state
431 if(allowStateChange){
432 state = requestedState;
433 }
434
435 switch(this->state){
436
437 case TMC_ControlState::uninitialized:
438 allowStateChange = false;
439 // check communication and write constants
440 if(!pingDriver()){ // driver not available
441 initialized = false; // Assume driver is not initialized if we can not detect it
442 Delay(250);
443 break;
444 }
445 // Driver available. Write constants and go to next step
446 if(!initialized){
447 initialize();
448 }
449 changeState(TMC_ControlState::waitPower,true);
450 break;
451
452 case TMC_ControlState::waitPower:
453 {
454 allowStateChange = false;
455 pulseClipLed(); // blink led
456 static uint8_t powerCheckCounter = 0;
457 // if powered check ADCs and go to encoder calibration
458 if(!hasPower()){
459 powerCheckCounter = 0;
460 Delay(250);
461 break;
462 }
463 if(++powerCheckCounter > 5 && !powerInitialized){
464 initializeWithPower();
465 }
466 if(powerInitialized){
467 allowStateChange = true;
468 }
469 Delay(100);
470 break;
471 }
472
473 case TMC_ControlState::FullCalibration:
474 {
475 fullCalibrationInProgress = true;
476 /*
477 * Wait for power (OK)
478 * Calibrate ADC offsets (OK)
479 * Measure motor response
480 * depending on encoder do encoder parameter estimation
481 * align and store phiE for single phase AENC or indexed ABN enc
482 *
483 * If at any point external movement is detected abort
484 */
485 // Wait for Power
486 while(!hasPower()){
487 Delay(100);
488 }
489
490 // Calibrate ADC
491 enablePin.set();
492 setPwm(TMC_PwmMode::PWM_FOC); // enable foc to calibrate adc
493 Delay(50);
494 if(calibrateAdcOffset(500)){
495 saveAdcParams();
496 }else{
497 calibFailCb();
498 break;
499 }
500
501 // Encoder
502 calibrateEncoder();
503 setEncoderType(conf.motconf.enctype);
504 recalibrationRequired = false;
505 break;
506 }
507
508 case TMC_ControlState::IndexSearch:
509 autohome();
510 changeState(laststate);
511
512 break;
513
514 case TMC_ControlState::Running:
515 {
516 // Check status, Temps, Everything alright?
517 uint32_t tick = HAL_GetTick();
518 if(tick - lastStatTime > 2000){ // Every 2s
519 lastStatTime = tick;
520 statusCheck();
521 // Get enable input. If tmc does not reply the result will read 0 or 0xffffffff (not possible normally)
522 uint32_t pins = readReg(0x76);
523 bool tmc_en = ((pins >> 15) & 0x01) && pins != 0xffffffff;
524 if(!tmc_en && motorEnabledRequested){ // Hardware emergency.
525 this->estopTriggered = true;
526 this->emergencyStop(false);
527 ErrorHandler::addError(estopError);
528 //changeState(TMC_ControlState::HardError);
529 }
530
531 // Temperature sense
532 if(conf.hwconf.temperatureEnabled){
533 float temp = getTemp();
534 if(temp > conf.hwconf.temp_limit){
535 changeState(TMC_ControlState::OverTemp);
536 pulseErrLed();
537 }
538 }
539
540 }
541 Delay(200);
542 }
543 break;
544
545 case TMC_ControlState::Shutdown:
546 Delay(100);
547 if(estopTriggered){
548 uint32_t pins = readReg(0x76);
549 bool tmc_en = ((pins >> 15) & 0x01) && pins != 0xffffffff;
550 if(tmc_en){
551 // Emergency stop reset
552 ErrorHandler::clearError(estopError);
553 this->estopTriggered = false; // TODO resume correctly
554 changeState(TMC_ControlState::uninitialized,true);
555 }
556 }
557 break;
558
559 case TMC_ControlState::EncoderInit:
560 if(powerInitialized && hasPower())
561 encoderInit();
562 break;
563
564 case TMC_ControlState::ExternalEncoderInit:
565 if(powerInitialized && hasPower() && drvEncoder != nullptr)
566 encoderInit();
567 break;
568
569 case TMC_ControlState::HardError:
570
571 break; // Broken
572
573 case TMC_ControlState::OverTemp:
574 this->stopMotor();
575 changeState(TMC_ControlState::HardError); // Block
576 break;
577
578 case TMC_ControlState::EncoderFinished: // Startup sequence done
579 //setEncoderIndexFlagEnabled(false); // TODO
580 encoderAligned = true;
581 if(motorEnabledRequested && isSetUp()){
582 startMotor();
583 changeState(TMC_ControlState::Running);
584 }else{
585 stopMotor();
586 laststate = TMC_ControlState::Running; // Go to running when starting again
587 }
588
589 if(fullCalibrationInProgress){
590 fullCalibrationInProgress = false;
591 Flash_Write(flashAddrs.encA,encodeEncHallMisc()); // Save encoder settings
592 }
593
594
595 break;
596
597 default:
598
599 break;
600 }
601
602
603 // Optional update methods for safety
604
605 if(!hasPower() && state != TMC_ControlState::waitPower && initialized && powerInitialized){ // low voltage or overvoltage
606 lastMotionMode = curMotionMode;
607 requestedState = state;
608 ErrorHandler::addError(lowVoltageError);
609 setMotionMode(MotionMode::stop,true); // Disable tmc
610 changeState(TMC_ControlState::waitPower,true);
611 allowStateChange = false;
612 }
613
614 if(flagCheckInProgress){ // cause some delay until reenabling the status interrupt checking
615 setStatusFlags(0);
616 flagCheckInProgress = false;
617 }
618 Delay(10);
619 } // End while
620}

677void TMC4671::zeroAbnUsingPhiM(bool offsetPhiE){
678 int32_t npos = (int32_t)readReg(0x28); // raw encoder counts at index hit
679 int32_t npos_M = (npos * 0xffff) / abnconf.cpr; // Scaled encoder angle at index
680 abnconf.phiMoffset = -npos_M;
681 if(offsetPhiE){
682 abnconf.phiEoffset += npos_M*conf.motconf.pole_pairs;
683 // change index to zero phiM
684 uint32_t phiEphiM = readReg(0x29);
685 int16_t phiE = ((phiEphiM >> 16) & 0xffff); // Write back phiE offset
686 int16_t phiM = phiEphiM & 0xffff;
687 //updateReg(0x29, abnconf.phiMoffset, 0xffff, 0);
688 writeReg(0x29,(phiE << 16) | phiM);
689 }else{
690 updateReg(0x29, abnconf.phiMoffset, 0xffff, 0);
691 }
692 setTmcPos(getPosAbs()); // Set position to absolute position = ~zero
693}

1045bool TMC4671::checkEncoder(){
1046 if(this->conf.motconf.motor_type != MotorType::STEPPER && this->conf.motconf.motor_type != MotorType::BLDC &&
1047 conf.motconf.enctype != EncoderType_TMC::uvw && conf.motconf.enctype != EncoderType_TMC::sincos && conf.motconf.enctype != EncoderType_TMC::abn && conf.motconf.enctype != EncoderType_TMC::ext)
1048 { // If not stepper or bldc return
1049 return true;
1050 }
1051 blinkClipLed(150, 0);
1052
1053 const uint16_t maxcount = 50; // Allowed reversals
1054 const uint16_t maxfail = 10; // Allowed fails
1055 const int16_t startAngle = getPhiE_Enc(); // Start angle offsets all angles later so there is no jump if angle is already properly aligned
1056 const int16_t targetAngle = 0x3FFF;
1057
1058 bool result = true;
1059 PhiE lastphie = getPhiEtype();
1060 MotionMode lastmode = getMotionMode();
1061 setFluxTorque(0, 0);
1062 setPhiEtype(PhiE::ext);
1063
1064 setPhiE_ext(startAngle);
1065 // Ramp up flux
1066 for(int16_t flux = 0; flux <= 2*bangInitPower/3; flux+=20){
1067 setFluxTorque(flux, 0);
1068 Delay(2);
1069 }
1070 //Forward
1071 int16_t phiE_enc = 0;
1072 uint16_t failcount = 0;
1073 int16_t revCount = 0;
1074 for(int16_t angle = 0;angle<targetAngle;angle+=0x00ff){
1075 uint16_t c = 0;
1076 setPhiE_ext(angle+startAngle);
1077 Delay(5);
1078 phiE_enc = getPhiE_Enc() - startAngle;
1079 int16_t err = abs(phiE_enc - angle);
1080 int16_t nErr = abs(phiE_enc + angle);
1081 // Wait more until encoder settles a bit
1082 while(err > 2000 && nErr > 2000 && c++ < 50){
1083 phiE_enc = getPhiE_Enc() - startAngle;
1084 err = abs(phiE_enc - angle);
1085 nErr = abs(angle - phiE_enc);
1086 Delay(10);
1087 }
1088 if(err > nErr){
1089 revCount++;
1090 }
1091 if(c >= maxcount){
1092 failcount++;
1093 if(failcount > maxfail){
1094 result = false;
1095 break;
1096 }
1097 }
1098 }
1099 /* If we are still at the start angle the encoder did not move at all.
1100 * Possible issues:
1101 * Encoder connection wrong
1102 * Wrong encoder selection
1103 * No motor movement
1104 * No encoder power
1105 */
1106 if(startAngle == getPhiE_Enc()){
1107 ErrorHandler::addError(Error(ErrorCode::encoderAlignmentFailed,ErrorType::critical,"Encoder did not move during alignment"));
1108 this->changeState(TMC_ControlState::HardError, true);
1109 result = false;
1110 }
1111
1112 // Backward
1113
1114 if(result){ // Only if not already failed
1115 for(int16_t angle = targetAngle;angle>0;angle -= 0x00ff){
1116 uint16_t c = 0;
1117 setPhiE_ext(angle+startAngle);
1118 Delay(5);
1119 phiE_enc = getPhiE_Enc() - startAngle;
1120 int16_t err = abs(phiE_enc - angle);
1121 int16_t nErr = abs(phiE_enc + angle);
1122 // Wait more
1123 while(err > 2500 && nErr > 2500 && c++ < 50){
1124 phiE_enc = getPhiE_Enc() - startAngle;
1125 err = abs(phiE_enc - angle);
1126 nErr = abs(angle - phiE_enc);
1127 Delay(10);
1128 }
1129 if(err > nErr){
1130 revCount++;
1131 }
1132 if(c >= maxcount){
1133 failcount++;
1134 if(failcount > maxfail){
1135 result = false;
1136 break;
1137 }
1138 }
1139 }
1140 }
1141
1142 // TODO check if we want that
1143 if(revCount > maxcount){ // Encoder seems reversed
1144 // reverse encoder
1145 if(this->conf.motconf.enctype == EncoderType_TMC::abn){
1146 this->abnconf.rdir = !this->abnconf.rdir;
1147 setup_ABN_Enc(abnconf);
1148 }else if(this->conf.motconf.enctype == EncoderType_TMC::sincos || this->conf.motconf.enctype == EncoderType_TMC::uvw){
1149 this->aencconf.rdir = !this->aencconf.rdir;
1150 setup_AENC(aencconf);
1151 }else if(this->conf.motconf.enctype == EncoderType_TMC::ext){
1152 this->conf.encoderReversed = !this->conf.encoderReversed;
1153 }
1154 ErrorHandler::addError(Error(ErrorCode::encoderReversed,ErrorType::warning,"Encoder direction reversed during check"));
1155 }
1156
1157 setFluxTorque(0, 0);
1158 setPhiE_ext(0);
1159 setPhiEtype(lastphie);
1160 setMotionMode(lastmode,true);
1161
1162 if(result){
1163 encoderAligned = true;
1164 }
1165 blinkClipLed(0, 0);
1166 return result;
1167}

1309void TMC4671::encoderInit(){
1310
1311 if(!powerInitialized || !hasPower()){
1312 changeState(TMC_ControlState::waitPower);
1313 requestedState = TMC_ControlState::EncoderInit;
1314 return;
1315 }
1316
1317 // Initializes encoder
1318 if(conf.motconf.enctype == EncoderType_TMC::abn){
1319 setPosSel(PosSelection::PhiM_abn); // Mechanical Angle
1320 setVelSel(VelSelection::PhiM_abn); // Mechanical Angle (RPM)
1321 //setup_ABN_Enc(abnconf);
1322 if(!encHallRestored){
1323 estimateABNparams(); // If not saved try to estimate parameters
1324 recalibrationRequired = true;
1325 }
1326 }else if(conf.motconf.enctype == EncoderType_TMC::sincos || conf.motconf.enctype == EncoderType_TMC::uvw){
1327 setPosSel(PosSelection::PhiM_aenc); // Mechanical Angle
1328 setVelSel(VelSelection::PhiM_aenc); // Mechanical Angle (RPM)
1329 //setup_AENC(aencconf);
1330 calibrateAenc();
1331 }
1332
1333 // find index
1334
1335 if(conf.motconf.enctype == EncoderType_TMC::abn && abnconf.useIndex && !encoderIndexHitFlag){ // TODO changing direction might invalidate phiE offset because of index pulse width
1336 findEncoderIndex(abnconf.posOffsetFromIndex < 0 ? 10 : -10,bangInitPower/2,true,true); // Go to index and zero encoder
1337 setPhiEtype(PhiE::openloop); // Openloop used in last step. Use for aligning too
1338 }else{
1339 setPhiE_ext(getPhiE());
1340 setPhiEtype(PhiE::ext);
1341 }
1342
1343 // Align encoder
1344 // TODO handle absolute external encoders
1345 bangInitEnc(bangInitPower);
1346
1347 // Check encoder
1348 if(!checkEncoder()){
1349 if(++enc_retry > enc_retry_max){
1350 encoderAligned = false;
1351 Error err = Error(ErrorCode::encoderAlignmentFailed,ErrorType::critical,"Encoder alignment failed");
1352 ErrorHandler::addError(err);
1353 stopMotor();
1354 allowStateChange = false;
1355 changeState(TMC_ControlState::HardError,true);
1356 }
1357
1358 if(manualEncAlign){
1359 manualEncAlign = false;
1360 CommandHandler::broadcastCommandReply(CommandReply("Error during check",1), (uint32_t)TMC4671_commands::encalign, CMDtype::get);
1361 }
1362 return;
1363 }
1364 encoderAligned = true;
1365
1366
1367
1368 if(conf.motconf.enctype == EncoderType_TMC::abn && abnconf.useIndex && encoderIndexHitFlag)
1369 setTmcPos(getPosAbs() - abnconf.posOffsetFromIndex); // Load stored position
1370
1371 if(manualEncAlign){
1372 manualEncAlign = false;
1373 CommandHandler::broadcastCommandReply(CommandReply("Aligned successfully",1), (uint32_t)TMC4671_commands::encalign, CMDtype::get);
1374 }
1375 changeState(TMC_ControlState::EncoderFinished);
1376
1377 if(conf.motconf.enctype == EncoderType_TMC::abn){
1378 setPhiEtype(PhiE::abn);
1379 }else if(conf.motconf.enctype == EncoderType_TMC::sincos || conf.motconf.enctype == EncoderType_TMC::uvw){
1380 setPhiEtype(PhiE::aenc);
1381 }else if(usingExternalEncoder()){
1382// setPosSel(PosSelection::PhiE_ext);
1383// setVelSel(VelSelection::PhiE_ext); // Mechanical Angle (RPM)

1385 }
