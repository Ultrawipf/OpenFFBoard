55void MotorPWM::turn(int16_t power){
56 if(!active)
57 return;
58
59 /*
60 * Generates a classic RC 20ms 1000-2000µs signal
61 * Centered at 1500µs for bidirectional RC ESCs and similiar stuff
62 */
63 if(mode == ModePWM_DRV::RC_PWM){
64 int32_t pval = power;
65
66 float val = ((pval * 1000)/0x7fff)*tFreq;
67 val = clip((1500*tFreq)-val,1000*tFreq, 2000*tFreq);
68 setPWM(val,ccr_1);
69
70 /*
71 * Generates a 0-100% PWM signal
72 * and outputs complementary direction signals.
73 * Can be used with cheap halfbridge modules and DC motors
74 */
75 }else if(mode == ModePWM_DRV::PWM_DIR){
76 if(power < 0){
77 setPWM(0,ccr_3);
78 setPWM(0xffff,ccr_4);
79 }else{
80 setPWM(0,ccr_4);
81 setPWM(0xffff,ccr_3);
82 }
83 int32_t val = (uint32_t)((abs(power) * period)/0x7fff);
84 setPWM(val,ccr_1);
85
86 }else if(mode == ModePWM_DRV::CENTERED_PWM){
87 int32_t pval = 0x7fff+power;
88 int32_t val = (pval * period)/0xffff;
89 setPWM(val,ccr_1);
90
91 }else if(mode == ModePWM_DRV::PWM_DUAL){
92 int32_t val = (uint32_t)((abs(power) * period)/0x7fff);
93 if(power < 0){
94 setPWM(0,ccr_1);
95 setPWM(val,ccr_2);
96 }else{
97 setPWM(0,ccr_2);
98 setPWM(val,ccr_1);
99 }
100 }

106void MotorPWM::setPwmSpeed(SpeedPWM_DRV spd){
107 bool ok = true;
108 switch(spd){
109
110 case SpeedPWM_DRV::LOW:
111 if(mode == ModePWM_DRV::RC_PWM){
112 period = 40000; //20ms (40000/Sysclock)
113 prescaler = TIM_PWM_FREQ/2000000;
114 }else{
115 period = TIM_PWM_FREQ/3000; // Check if timer can count high enough for very high clock speeds!
116 prescaler = 0;
117 }
118
119 break;
120 case SpeedPWM_DRV::MID:
121 if(mode == ModePWM_DRV::RC_PWM){
122 period = 30000;//15ms(30000/47)
123 prescaler = TIM_PWM_FREQ/2000000;
124 }else{
125 period = TIM_PWM_FREQ/9000;
126 prescaler = 0;
127 }
128
129 break;
130 case SpeedPWM_DRV::HIGH:
131 if(mode == ModePWM_DRV::RC_PWM){
132 period = 20000; //10ms (20000/47)
133 prescaler = TIM_PWM_FREQ/2000000;
134 }else{
135 period = TIM_PWM_FREQ/17000;
136 prescaler = 0;
137 }
138 break;
139 case SpeedPWM_DRV::VERYHIGH:
140 if(mode == ModePWM_DRV::RC_PWM){
141 period = 10000; //5ms (20000/23)
142 prescaler = TIM_PWM_FREQ/2000000;
143 }else{
144 period = TIM_PWM_FREQ/24000;
145 prescaler = 0;
146 }
147 break;
148 default:
149 ok = false;
150 }
151
152 if(ok){
153 this->pwmspeed = spd;
154 tFreq = (float)(TIM_PWM_FREQ/1000000)/(float)(prescaler+1);
155
156 pwmInitTimer(timer, channel_1,period,prescaler);
157 pwmInitTimer(timer, channel_2,period,prescaler);
158 pwmInitTimer(timer, channel_3,period,prescaler);
159 pwmInitTimer(timer, channel_4,period,prescaler);
160 HAL_TIM_MspPostInit(timer);
161// setPWM_HAL(0, timer, channel_1, period);
162// pwmInitTimer(timer, channel_2,period,prescaler);
163// setPWM_HAL(0, timer, channel_2, period);
164 turn(0);
165 }
166}

191void MotorPWM::saveFlash(){
192 // 0-3: mode
193 // 4-6: speed
194 uint16_t var = (uint8_t)this->mode & 0xf;
195 var |= ((uint8_t)this->pwmspeed & 0x7) << 4;
196 Flash_Write(ADR_PWM_MODE, var);
197}
198void MotorPWM::restoreFlash(){
199 uint16_t var = 0;
200 if(Flash_Read(ADR_PWM_MODE, &var)){
201 uint8_t m = var & 0xf;
202 this->setMode(ModePWM_DRV(m));
203
204 uint8_t s = (var >> 4) & 0x7;
205 this->setPwmSpeed(SpeedPWM_DRV(s));
206 }
207}

232CommandStatus MotorPWM::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
233
234 switch(static_cast<MotorPWM_commands>(cmd.cmdId)){
235
236 case MotorPWM_commands::freq:
237 {
238 if(cmd.type == CMDtype::set){
239 this->setPwmSpeed((SpeedPWM_DRV)cmd.val);
240 }else if(cmd.type == CMDtype::get){
241 replies.push_back(CommandReply((uint8_t)this->getPwmSpeed()));
242 }else if(cmd.type == CMDtype::info){
243 std::vector<std::string> names = PWM_SpeedNames;
244 if(this->mode == ModePWM_DRV::RC_PWM){
245 names = RC_SpeedNames;
246 }else{
247 names = PWM_SpeedNames;
248 }
249 for(uint8_t i = 0; i<names.size();i++){
250 replies.push_back(CommandReply(names[i] + ":" + std::to_string(i)+"\n"));
251 }
252 }
253 break;
254 }
255 case MotorPWM_commands::mode:
256 {
257 if(cmd.type == CMDtype::set){
258 this->setMode((ModePWM_DRV)cmd.val);
259 }else if(cmd.type == CMDtype::get){
260 replies.push_back(CommandReply((uint8_t)this->getMode()));
261 }else if(cmd.type == CMDtype::info){
262 for(uint8_t i = 0; i<PwmModeNames.size();i++){
263 replies.push_back(CommandReply(PwmModeNames[i] + ":" + std::to_string(i)+"\n"));
264 }
265 }
266 break;
267 }
268 default:
269 return CommandStatus::NOT_FOUND;
270 }
271
272 return CommandStatus::OK;
273
274}
