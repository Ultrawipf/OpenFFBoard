140//--------------------------------------------------------------------
141// RootHub Mode Configuration
142// CFG_TUSB_RHPORTx_MODE contains operation mode and speed for that port
143//--------------------------------------------------------------------
144
145// Lower 4-bit is operational mode
146#define OPT_MODE_NONE 0x00
147#define OPT_MODE_DEVICE 0x01
148#define OPT_MODE_HOST 0x02
149
150// Higher 4-bit is max operational speed (corresponding to tusb_speed_t)
151#define OPT_MODE_FULL_SPEED 0x00
152#define OPT_MODE_LOW_SPEED 0x10
153#define OPT_MODE_HIGH_SPEED 0x20
154
155
156#ifndef CFG_TUSB_RHPORT0_MODE
157 #define CFG_TUSB_RHPORT0_MODE OPT_MODE_NONE
158#endif
159
160
161#ifndef CFG_TUSB_RHPORT1_MODE
162 #define CFG_TUSB_RHPORT1_MODE OPT_MODE_NONE
163#endif
164
165#if (((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_HOST ) && ((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_HOST )) || \
166 (((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_DEVICE) && ((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_DEVICE))
167 #error "TinyUSB currently does not support same modes on more than 1 roothub port"
168#endif
169
170// Which roothub port is configured as host
171#define TUH_OPT_RHPORT ( ((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_HOST) ? 0 : (((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_HOST) ? 1 : -1) )
172#define TUSB_OPT_HOST_ENABLED ( TUH_OPT_RHPORT >= 0 )
173
174// Which roothub port is configured as device
175#define TUD_OPT_RHPORT ( ((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_DEVICE) ? 0 : (((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_DEVICE) ? 1 : -1) )
176
177#if TUD_OPT_RHPORT == 0
178#define TUD_OPT_HIGH_SPEED ( (CFG_TUSB_RHPORT0_MODE) & OPT_MODE_HIGH_SPEED )
179#else
180#define TUD_OPT_HIGH_SPEED ( (CFG_TUSB_RHPORT1_MODE) & OPT_MODE_HIGH_SPEED )
181#endif
182
183#define TUSB_OPT_DEVICE_ENABLED ( TUD_OPT_RHPORT >= 0 )

280// Port Options
281// TUP for TinyUSB Port (can be renamed)
282//--------------------------------------------------------------------+
283
284// TUP_ARCH_STRICT_ALIGN if arch cannot access unaligned memory
285
286
287// ARMv7+ (M3-M7, M23-M33) can access unaligned memory
288#if (defined(__ARM_ARCH) && (__ARM_ARCH >= 7))
289 #define TUP_ARCH_STRICT_ALIGN 0
290#else
291 #define TUP_ARCH_STRICT_ALIGN 1
292#endif
293
294// TUP_MCU_STRICT_ALIGN will overwrite TUP_ARCH_STRICT_ALIGN.
295// In case TUP_MCU_STRICT_ALIGN = 1 and TUP_ARCH_STRICT_ALIGN =0, we will not reply on compiler
296// to generate unaligned access code.
297// LPC_IP3511 Highspeed cannot access unaligned memory on USB_RAM
298#if TUD_OPT_HIGH_SPEED && (CFG_TUSB_MCU == OPT_MCU_LPC54XXX || CFG_TUSB_MCU == OPT_MCU_LPC55XX)
299 #define TUP_MCU_STRICT_ALIGN 1
300#else
301 #define TUP_MCU_STRICT_ALIGN 0
302#endif
