161static void const* _find_desc(void const *beg, void const *end, uint_fast8_t desc_type)
162{
163 void const *cur = beg;
164 while ((cur < end) && (desc_type != tu_desc_type(cur))) {
165 cur = tu_desc_next(cur);
166 }
167 return cur;
168}

180static void const* _find_desc_3(void const *beg, void const *end,
181 uint_fast8_t desc_type,
182 uint_fast8_t element_0,
183 uint_fast8_t element_1)
184{
185 for (void const *cur = beg; cur < end; cur = _find_desc(cur, end, desc_type)) {
186 uint8_t const *p = (uint8_t const *)cur;
187 if ((p[2] == element_0) && (p[3] == element_1)) {
188 return cur;
189 }
190 cur = tu_desc_next(cur);
191 }
192 return end;
193}

203{

208 cur = _find_desc(tu_desc_next(cur), end, TUSB_DESC_INTERFACE);

210 return cur;

236static void const* _find_desc_ep(void const *beg, void const *end)
237{
238 for (void const *cur = beg; cur < end; cur = tu_desc_next(cur)) {
239 uint_fast8_t desc_type = tu_desc_type(cur);
240 if (TUSB_DESC_ENDPOINT == desc_type) return cur;
241 if (TUSB_DESC_INTERFACE == desc_type) break;
242 }
243 return end;

266 cur = tu_desc_next(cur);
267 }

295static bool _update_streaming_parameters(videod_streaming_interface_t const *stm,
296 video_probe_and_commit_control_t *param)
297{
298 tusb_desc_vs_itf_t const *vs = _get_desc_vs(stm);
299 uint_fast8_t fmtnum = param->bFormatIndex;
300 TU_ASSERT(fmtnum <= vs->stm.bNumFormats);
301 if (!fmtnum) {
302 if (1 < vs->stm.bNumFormats) return true; /* Need to negotiate all variables. */
303 fmtnum = 1;
304 param->bFormatIndex = 1;
305 }
306
307 /* Set the parameters determined by the format */
308 param->wKeyFrameRate = 1;
309 param->wPFrameRate = 0;
310 param->wCompQuality = 1; /* 1 to 10000 */
311 param->wCompWindowSize = 1; /* GOP size? */
312 param->wDelay = 0; /* milliseconds */
313 param->dwClockFrequency = 27000000; /* same as MPEG-2 system time clock */
314 param->bmFramingInfo = 0x3; /* enables FrameID and EndOfFrame */
315 param->bPreferedVersion = 1;
316 param->bMinVersion = 1;
317 param->bMaxVersion = 1;
318 param->bUsage = 0;
319 param->bBitDepthLuma = 8;
320
321 void const *end = _end_of_streaming_descriptor(vs);
322 tusb_desc_cs_video_fmt_uncompressed_t const *fmt = _find_desc_format(tu_desc_next(vs), end, fmtnum);
323 TU_ASSERT(fmt != end);
324 uint_fast8_t frmnum = param->bFrameIndex;
325 TU_ASSERT(frmnum <= fmt->bNumFrameDescriptors);
326 if (!frmnum) {
327 if (1 < fmt->bNumFrameDescriptors) return true;
328 frmnum = 1;
329 param->bFrameIndex = 1;
330 }
331 tusb_desc_cs_video_frm_uncompressed_t const *frm = _find_desc_frame(tu_desc_next(fmt), end, frmnum);
332 TU_ASSERT(frm != end);
333
334 /* Set the parameters determined by the frame */
335 uint_fast32_t frame_size = param->dwMaxVideoFrameSize;
336 if (!frame_size) {
337 frame_size = (uint_fast32_t)frm->wWidth * frm->wHeight * fmt->bBitsPerPixel / 8;
338 param->dwMaxVideoFrameSize = frame_size;
339 }
340
341 uint_fast32_t interval = param->dwFrameInterval;
342 if (!interval) {
343 if ((1 < frm->bFrameIntervalType) ||
344 ((0 == frm->bFrameIntervalType) && (frm->dwFrameInterval[1] != frm->dwFrameInterval[0]))) {
345 return true;
346 }
347 interval = frm->dwFrameInterval[0];
348 param->dwFrameInterval = interval;
349 }
350 uint_fast32_t interval_ms = interval / 10000;
351 TU_ASSERT(interval_ms);
352 uint_fast32_t payload_size = (frame_size + interval_ms - 1) / interval_ms + 2;
353 param->dwMaxPayloadTransferSize = payload_size;
354 return true;
355}

362static bool _negotiate_streaming_parameters(videod_streaming_interface_t const *stm, uint_fast8_t request,
363 video_probe_and_commit_control_t *param)
364{
365 uint_fast8_t const fmtnum = param->bFormatIndex;
366 if (!fmtnum) {
367 switch (request) {
368 case VIDEO_REQUEST_GET_MAX:
369 param->bFormatIndex = _get_desc_vs(stm)->stm.bNumFormats;
370 break;
371 case VIDEO_REQUEST_GET_MIN:
372 case VIDEO_REQUEST_GET_DEF:
373 param->bFormatIndex = 1;
374 break;
375 default: return false;
376 }
377 /* Set the parameters determined by the format */
378 param->wKeyFrameRate = 1;
379 param->wPFrameRate = 0;
380 param->wCompQuality = 1; /* 1 to 10000 */
381 param->wCompWindowSize = 1; /* GOP size? */
382 param->wDelay = 0; /* milliseconds */
383 param->dwClockFrequency = 27000000; /* same as MPEG-2 system time clock */
384 param->bmFramingInfo = 0x3; /* enables FrameID and EndOfFrame */
385 param->bPreferedVersion = 1;
386 param->bMinVersion = 1;
387 param->bMaxVersion = 1;
388 param->bUsage = 0;
389 param->bBitDepthLuma = 8;
390 return true;
391 }
392
393 uint_fast8_t frmnum = param->bFrameIndex;
394 if (!frmnum) {
395 tusb_desc_vs_itf_t const *vs = _get_desc_vs(stm);
396 void const *end = _end_of_streaming_descriptor(vs);
397 tusb_desc_cs_video_fmt_uncompressed_t const *fmt = _find_desc_format(tu_desc_next(vs), end, fmtnum);
398 switch (request) {
399 case VIDEO_REQUEST_GET_MAX:
400 frmnum = fmt->bNumFrameDescriptors;
401 break;
402 case VIDEO_REQUEST_GET_MIN:
403 frmnum = 1;
404 break;
405 case VIDEO_REQUEST_GET_DEF:
406 frmnum = fmt->bDefaultFrameIndex;
407 break;
408 default: return false;
409 }
410 param->bFrameIndex = frmnum;
411 /* Set the parameters determined by the frame */
412 tusb_desc_cs_video_frm_uncompressed_t const *frm = _find_desc_frame(tu_desc_next(fmt), end, frmnum);
413 param->dwMaxVideoFrameSize = frm->wWidth * frm->wHeight * fmt->bBitsPerPixel / 8;
414 return true;
415 }
416
417 if (!param->dwFrameInterval) {
418 tusb_desc_vs_itf_t const *vs = _get_desc_vs(stm);
419 void const *end = _end_of_streaming_descriptor(vs);
420 tusb_desc_cs_video_fmt_uncompressed_t const *fmt = _find_desc_format(tu_desc_next(vs), end, fmtnum);
421 tusb_desc_cs_video_frm_uncompressed_t const *frm = _find_desc_frame(tu_desc_next(fmt), end, frmnum);
422
423 uint_fast32_t interval, interval_ms;
424 switch (request) {
425 case VIDEO_REQUEST_GET_MAX:
426 {
427 uint_fast32_t min_interval, max_interval;
428 uint_fast8_t num_intervals = frm->bFrameIntervalType;
429 max_interval = num_intervals ? frm->dwFrameInterval[num_intervals - 1]: frm->dwFrameInterval[1];
430 min_interval = frm->dwFrameInterval[0];
431 interval = max_interval;
432 interval_ms = min_interval / 10000;
433 }
434 break;
435 case VIDEO_REQUEST_GET_MIN:
436 {
437 uint_fast32_t min_interval, max_interval;
438 uint_fast8_t num_intervals = frm->bFrameIntervalType;
439 max_interval = num_intervals ? frm->dwFrameInterval[num_intervals - 1]: frm->dwFrameInterval[1];
440 min_interval = frm->dwFrameInterval[0];
441 interval = min_interval;
442 interval_ms = max_interval / 10000;
443 }
444 break;
445 case VIDEO_REQUEST_GET_DEF:
446 interval = frm->dwDefaultFrameInterval;
447 interval_ms = interval / 10000;
448 break;
449 case VIDEO_REQUEST_GET_RES:
450 {
451 uint_fast8_t num_intervals = frm->bFrameIntervalType;
452 if (num_intervals) {
453 interval = 0;
454 } else {
455 interval = frm->dwFrameInterval[2];
456 interval_ms = interval / 10000;
457 }
458 }
459 break;
460 default: return false;
461 }
462 param->dwFrameInterval = interval;
463 if (!interval) {
464 param->dwMaxPayloadTransferSize = 0;
465 } else {
466 uint_fast32_t frame_size = param->dwMaxVideoFrameSize;
467 if (!interval_ms) {
468 param->dwMaxPayloadTransferSize = frame_size + 2;
469 } else {
470 param->dwMaxPayloadTransferSize = (frame_size + interval_ms - 1) / interval_ms + 2;
471 }
472 }
473 return true;
474 }
475 return true;
476}

482static bool _close_vc_itf(uint8_t rhport, videod_interface_t *self)
483{
484 tusb_desc_vc_itf_t const *vc = _get_desc_vc(self);
485 /* The next descriptor after the class-specific VC interface header descriptor. */
486 void const *cur = (void const*)vc + vc->std.bLength + vc->ctl.bLength;
487 /* The end of the video control interface descriptor. */
488 void const *end = (void const*)vc + vc->std.bLength + vc->ctl.wTotalLength;
489 if (vc->std.bNumEndpoints) {
490 /* Find the notification endpoint descriptor. */
491 cur = _find_desc(cur, end, TUSB_DESC_ENDPOINT);
492 TU_ASSERT(cur < end);
493 tusb_desc_endpoint_t const *notif = (tusb_desc_endpoint_t const *)cur;
494 usbd_edpt_close(rhport, notif->bEndpointAddress);
495 }
496 self->cur = 0;
497 return true;
498}

504static bool _open_vc_itf(uint8_t rhport, videod_interface_t *self, uint_fast8_t altnum)
505{
506 TU_LOG2(" open VC %d\n", altnum);
507 void const *beg = self->beg;
508 void const *end = beg + self->len;
509 /* The first descriptor is a video control interface descriptor. */
510 void const *cur = _find_desc_itf(beg, end, _desc_itfnum(beg), altnum);
511 TU_LOG2(" cur %d\n", cur - beg);
512 TU_VERIFY(cur < end);
513
514 tusb_desc_vc_itf_t const *vc = (tusb_desc_vc_itf_t const *)cur;
515 TU_LOG2(" bInCollection %d\n", vc->ctl.bInCollection);
516 /* Support for up to 2 streaming interfaces only. */
517 TU_ASSERT(vc->ctl.bInCollection <= CFG_TUD_VIDEO_STREAMING);
518
519 /* Update to point the end of the video control interface descriptor. */
520 end = cur + vc->std.bLength + vc->ctl.wTotalLength;
521 /* Advance to the next descriptor after the class-specific VC interface header descriptor. */
522 cur += vc->std.bLength + vc->ctl.bLength;
523 TU_LOG2(" bNumEndpoints %d\n", vc->std.bNumEndpoints);
524 /* Open the notification endpoint if it exist. */
525 if (vc->std.bNumEndpoints) {
526 /* Support for 1 endpoint only. */
527 TU_VERIFY(1 == vc->std.bNumEndpoints);
528 /* Find the notification endpoint descriptor. */
529 cur = _find_desc(cur, end, TUSB_DESC_ENDPOINT);
530 TU_VERIFY(cur < end);
531 tusb_desc_endpoint_t const *notif = (tusb_desc_endpoint_t const *)cur;
532 /* Open the notification endpoint */
533 TU_ASSERT(usbd_edpt_open(rhport, notif));
534 }
535 self->cur = (void const*)vc - beg;
536 return true;
537}

543static bool _open_vs_itf(uint8_t rhport, videod_streaming_interface_t *stm, uint_fast8_t altnum)
544{
545 uint_fast8_t i;
546 TU_LOG2(" reopen VS %d\n", altnum);
547 void const *desc = _videod_itf[stm->index_vc].beg;
548
549 /* Close endpoints of previous settings. */
550 for (i = 0; i < TU_ARRAY_SIZE(stm->desc.ep); ++i) {
551 uint_fast16_t ofs_ep = stm->desc.ep[i];
552 if (!ofs_ep) break;
553 uint_fast8_t ep_adr = _desc_ep_addr(desc + ofs_ep);
554 usbd_edpt_close(rhport, ep_adr);
555 stm->desc.ep[i] = 0;
556 TU_LOG2(" close EP%02x\n", ep_adr);
557 }
558 /* clear transfer management information */
559 stm->buffer = NULL;
560 stm->bufsize = 0;
561 stm->offset = 0;
562
563 /* Find a alternate interface */
564 void const *beg = desc + stm->desc.beg;
565 void const *end = desc + stm->desc.end;
566 void const *cur = _find_desc_itf(beg, end, _desc_itfnum(beg), altnum);
567 TU_VERIFY(cur < end);
568 uint_fast8_t numeps = ((tusb_desc_interface_t const *)cur)->bNumEndpoints;
569 TU_ASSERT(numeps <= TU_ARRAY_SIZE(stm->desc.ep));
570 stm->desc.cur = cur - desc; /* Save the offset of the new settings */
571 if (!altnum) {
572 /* initialize streaming settings */
573 stm->max_payload_transfer_size = 0;
574 video_probe_and_commit_control_t *param =
575 (video_probe_and_commit_control_t *)&stm->ep_buf;
576 tu_memclr(param, sizeof(*param));
577 return _update_streaming_parameters(stm, param);
578 }
579 /* Open endpoints of the new settings. */
580 for (i = 0, cur = tu_desc_next(cur); i < numeps; ++i, cur = tu_desc_next(cur)) {
581 cur = _find_desc_ep(cur, end);
582 TU_ASSERT(cur < end);
583 tusb_desc_endpoint_t const *ep = (tusb_desc_endpoint_t const*)cur;
584 if (!stm->max_payload_transfer_size) {
585 video_probe_and_commit_control_t const *param = (video_probe_and_commit_control_t const*)&stm->ep_buf;
586 uint_fast32_t max_size = param->dwMaxPayloadTransferSize;
587 if ((TUSB_XFER_ISOCHRONOUS == ep->bmAttributes.xfer) &&
588 (tu_edpt_packet_size(ep) < max_size))
589 {
590 /* FS must be less than or equal to max packet size */
591 return false;
592 }
593 /* Set the negotiated value */
594 stm->max_payload_transfer_size = max_size;
595 }
596 TU_ASSERT(usbd_edpt_open(rhport, ep));
597 stm->desc.ep[i] = cur - desc;
598 TU_LOG2(" open EP%02x\n", _desc_ep_addr(cur));
599 }
600 /* initialize payload header */
601 tusb_video_payload_header_t *hdr = (tusb_video_payload_header_t*)stm->ep_buf;
602 hdr->bHeaderLength = sizeof(*hdr);
603 hdr->bmHeaderInfo = 0;
604
605 return true;
606}

609static uint_fast16_t _prepare_in_payload(videod_streaming_interface_t *stm)
610{
611 uint_fast16_t remaining = stm->bufsize - stm->offset;
612 uint_fast16_t hdr_len = stm->ep_buf[0];
613 uint_fast16_t pkt_len = stm->max_payload_transfer_size;
614 if (hdr_len + remaining < pkt_len) {
615 pkt_len = hdr_len + remaining;
616 }
617 uint_fast16_t data_len = pkt_len - hdr_len;
618 memcpy(&stm->ep_buf[hdr_len], stm->buffer + stm->offset, data_len);
619 stm->offset += data_len;
620 remaining -= data_len;
621 if (!remaining) {
622 tusb_video_payload_header_t *hdr = (tusb_video_payload_header_t*)stm->ep_buf;
623 hdr->EndOfFrame = 1;
624 }
625 return hdr_len + data_len;
626}

629static int handle_video_ctl_std_req(uint8_t rhport, uint8_t stage,
630 tusb_control_request_t const *request,
631 uint_fast8_t ctl_idx)
632{
633 switch (request->bRequest) {
634 case TUSB_REQ_GET_INTERFACE:
635 if (stage == CONTROL_STAGE_SETUP)
636 {
637 TU_VERIFY(1 == request->wLength, VIDEO_ERROR_UNKNOWN);
638 tusb_desc_vc_itf_t const *vc = _get_desc_vc(&_videod_itf[ctl_idx]);
639 TU_VERIFY(vc, VIDEO_ERROR_UNKNOWN);
640
641 uint8_t alt_num = vc->std.bAlternateSetting;
642
643 TU_VERIFY(tud_control_xfer(rhport, request, &alt_num, sizeof(alt_num)), VIDEO_ERROR_UNKNOWN);
644 }
645 return VIDEO_ERROR_NONE;
646
647 case TUSB_REQ_SET_INTERFACE:
648 if (stage == CONTROL_STAGE_SETUP)
649 {
650 TU_VERIFY(0 == request->wLength, VIDEO_ERROR_UNKNOWN);

970{
971 TU_ASSERT(ctl_idx < CFG_TUD_VIDEO);
972 TU_ASSERT(stm_idx < CFG_TUD_VIDEO_STREAMING);
973 if (!buffer || !bufsize) return false;

976

980 for (uint_fast8_t i = 0; i < CFG_TUD_VIDEO_STREAMING; ++i) {
981 uint_fast16_t ofs_ep = stm->desc.ep[i];
982 if (!ofs_ep) continue;
983 ep_addr = _desc_ep_addr(desc + ofs_ep);
984 break;
985 }
986 if (!ep_addr) return false;
987
988 TU_VERIFY( usbd_edpt_claim(0, ep_addr));
989 /* update the packet header */
990 tusb_video_payload_header_t *hdr = (tusb_video_payload_header_t*)stm->ep_buf;
991 hdr->FrameID ^= 1;
992 hdr->EndOfFrame = 0;
993 /* update the packet data */
994 stm->buffer = (uint8_t*)buffer;
995 stm->bufsize = bufsize;
996 uint_fast16_t pkt_len = _prepare_in_payload(stm);
997 TU_ASSERT( usbd_edpt_xfer(0, ep_addr, stm->ep_buf, pkt_len), 0);
998 return true;
999}

1029uint16_t videod_open(uint8_t rhport, tusb_desc_interface_t const * itf_desc, uint16_t max_len)
1030{

1032 (VIDEO_SUBCLASS_CONTROL == itf_desc->bInterfaceSubClass) &&
1033 (VIDEO_ITF_PROTOCOL_15 == itf_desc->bInterfaceProtocol), 0);
1034
1035 /* Find available interface */
1036 videod_interface_t *self = NULL;
1037 uint_fast8_t ctl_idx;
1038 for (ctl_idx = 0; ctl_idx < CFG_TUD_VIDEO; ++ctl_idx) {
1039 if (_videod_itf[ctl_idx].beg) continue;
1040 self = &_videod_itf[ctl_idx];
1041 break;
1042 }
1043 TU_ASSERT(ctl_idx < CFG_TUD_VIDEO, 0);
1044
1045 void const *end = (void const*)itf_desc + max_len;
1046 self->beg = itf_desc;
1047 self->len = max_len;
1048 /*------------- Video Control Interface -------------*/
1049 TU_VERIFY(_open_vc_itf(rhport, self, 0), 0);
1050 tusb_desc_vc_itf_t const *vc = _get_desc_vc(self);
1051 uint_fast8_t bInCollection = vc->ctl.bInCollection;
1052 /* Find the end of the video interface descriptor */
1053 void const *cur = _next_desc_itf(itf_desc, end);
1054 for (uint_fast8_t stm_idx = 0; stm_idx < bInCollection; ++stm_idx) {
1055 videod_streaming_interface_t *stm = NULL;
1056 /* find free streaming interface handle */
1057 for (uint_fast8_t i = 0; i < CFG_TUD_VIDEO_STREAMING; ++i) {
1058 if (_videod_streaming_itf[i].desc.beg) continue;
1059 stm = &_videod_streaming_itf[i];
1060 self->stm[stm_idx] = i;
1061 break;
1062 }
1063 TU_ASSERT(stm, 0);
1064 stm->index_vc = ctl_idx;
1065 stm->index_vs = stm_idx;
1066 stm->desc.beg = (uintptr_t)cur - (uintptr_t)itf_desc;
1067 cur = _next_desc_itf(cur, end);
1068 stm->desc.end = (uintptr_t)cur - (uintptr_t)itf_desc;
1069 }
1070 self->len = (uintptr_t)cur - (uintptr_t)itf_desc;
1071 return (uintptr_t)cur - (uintptr_t)itf_desc;
1072}

1096 }
1097
1098 /* Identify which streaming interface to use */

1100 videod_streaming_interface_t *stm = &_videod_streaming_itf[itf];
1101 if (!stm->desc.beg) continue;
1102 void const *desc = _videod_itf[stm->index_vc].beg;
1103 if (itfnum == _desc_itfnum(desc + stm->desc.beg)) break;
1104 }

1107 err = handle_video_stm_req(rhport, stage, request, itf);
