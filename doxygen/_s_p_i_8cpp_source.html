44bool SPIPort::freeCsPin(OutputPin pin){
45 // is in pins
46 if(std::find(csPins.begin(),csPins.end(), pin) != csPins.end()){
47 // is not in free pins
48 if(std::find(freePins.begin(),freePins.end(), pin) == freePins.end()){
