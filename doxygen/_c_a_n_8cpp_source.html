267bool CANPort::sendMessage(CAN_TxHeaderTypeDef *pHeader, uint8_t aData[],uint32_t *pTxMailbox){
268 if(this->silent)

275 if (HAL_CAN_AddTxMessage(this->hcan, pHeader, aData, pTxMailbox) != HAL_OK)

277 /* Transmission request Error */
278 giveSemaphore();
279 return false;
280 }
281 giveSemaphore();

292int32_t CANPort::addCanFilter(CAN_FilterTypeDef sFilterConfig){
293 takeSemaphore();
294 int32_t lowestId = sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? 0 : slaveFilterStart;
295 int32_t highestId = sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? slaveFilterStart : 29;
296 int32_t foundId = -1;
297 for(uint8_t id = lowestId; id < highestId ; id++ ){
298 for(CAN_FilterTypeDef filter : canFilters){
299 if(filter.FilterFIFOAssignment == sFilterConfig.FilterFIFOAssignment && id == filter.FilterBank){
300 break;
301 }
302 }
303 foundId = id;
304 break;
305 }
306 if(foundId < highestId){
307 if (HAL_CAN_ConfigFilter(this->hcan, &sFilterConfig) == HAL_OK){
308 canFilters.push_back(sFilterConfig);
309 }
310 }
311 giveSemaphore();
312 return foundId;
313}

319void CANPort::removeCanFilter(uint8_t filterId){
320 semaphore.Take();
321 for (uint8_t i = 0; i < canFilters.size(); i++){
322 if(canFilters[i].FilterBank == filterId){
323 canFilters[i].FilterActivation = false;
324 HAL_CAN_ConfigFilter(this->hcan, &canFilters[i]);
325 canFilters.erase(canFilters.begin()+i);
326 break;
327 }
328 }
329 semaphore.Give();
330}

333CommandStatus CANPort::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
334
335 switch(static_cast<CanPort_commands>(cmd.cmdId)){
336
337 case CanPort_commands::speed:
338 if(cmd.type == CMDtype::get){
339 replies.emplace_back(this->speedPreset);
340 }else if(cmd.type == CMDtype::set){
341 setSpeedPreset(cmd.val);
342 }else if(cmd.type == CMDtype::info){
343 for(uint8_t i = 0; i<SpeedNames.size();i++){
344 replies.emplace_back(SpeedNames[i] + ":" + std::to_string(i));
345 }
346 }
347 break;
348
349 case CanPort_commands::send:
350 {
351 if(cmd.type == CMDtype::setat){
352 if(!active){
353 start(); // If port is not used activate port at first use
354 }
355 CAN_tx_msg msg;
356 memcpy(msg.data,&cmd.val,8);
357 header.StdId = cmd.adr;
358 msg.header = header;
359 sendMessage(msg);
360 }else{
361 return CommandStatus::NOT_FOUND;
362 }
363 break;
364 }
365 case CanPort_commands::len:
366 handleGetSet(cmd, replies, header.DLC);
367 header.DLC = std::min<uint32_t>(header.DLC,8);
368 break;
369 default:
370 return CommandStatus::NOT_FOUND;
371 }
372 return CommandStatus::OK;
373}
