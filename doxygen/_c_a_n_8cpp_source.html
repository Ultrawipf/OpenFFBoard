186int32_t CANPort::addCanFilter(CAN_FilterTypeDef sFilterConfig){
187 takeSemaphore();
188 int32_t lowestId = sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? 0 : slaveFilterStart;
189 int32_t highestId = sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? slaveFilterStart : 29;
190 int32_t foundId = -1;
191 for(uint8_t id = lowestId; id < highestId ; id++ ){
192 for(CAN_FilterTypeDef filter : canFilters){
193 if(filter.FilterFIFOAssignment == sFilterConfig.FilterFIFOAssignment && id == filter.FilterBank){
194 break;
195 }
196 }
197 foundId = id;
198 break;
199 }
200 if(foundId < highestId){
201 if (HAL_CAN_ConfigFilter(this->hcan, &sFilterConfig) == HAL_OK){
202 canFilters.push_back(sFilterConfig);
203 }
204 }
205 giveSemaphore();
206 return foundId;
207}

213void CANPort::removeCanFilter(uint8_t filterId){
214 semaphore.Take();
215 for (uint8_t i = 0; i < canFilters.size(); i++){
216 if(canFilters[i].FilterBank == filterId){
217 canFilters[i].FilterActivation = false;
218 HAL_CAN_ConfigFilter(this->hcan, &canFilters[i]);
219 canFilters.erase(canFilters.begin()+i);
220 break;
221 }
222 }
223 semaphore.Give();
224}
