54std::vector<int32_t>* LocalAnalog::getAxes(){
55 uint8_t chans = 0;
56 this->buf.clear();
57 volatile uint32_t* adcbuf = getAnalogBuffer(&AIN_HADC,&chans);
58
59 uint8_t axes = std::min<uint8_t>(chans-ADC_CHAN_FPIN,numPins);
60
61 for(uint8_t i = 0;i<axes;i++){
62 if(!(aconf.analogmask & 0x01 << i))
63 continue;
64
65 int32_t val = ((adcbuf[i+ADC_CHAN_FPIN] & 0xFFF) << 4)-0x7fff;
66
67 if(aconf.autorange){
68 minMaxVals[i].max = std::max(minMaxVals[i].max,val);
69 minMaxVals[i].min = std::min(minMaxVals[i].min,val);
70
71 int32_t range = (minMaxVals[i].max - minMaxVals[i].min);
72 if(range > 1){
73 float scaler = ((float)0xffff / (float)range);
74 val *= scaler;
75 val = val - ((scaler*(float)minMaxVals[i].min) + 0x7fff);
76 val = clip(val,-0x7fff,0x7fff); // Clip if slightly out of range because of inaccuracy
77 }
78 }
79
80
81 this->buf.push_back(val);
82 }
83 return &this->buf;
84}

86CommandStatus LocalAnalog::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
87
88 switch(static_cast<LocalAnaloc_commands>(cmd.cmdId)){
89 case LocalAnaloc_commands::pinmask:
90 return handleGetSet(cmd, replies, this->aconf.analogmask);
91 break;
92
93 case LocalAnaloc_commands::autocal:
94 if(cmd.type == CMDtype::get){
95 replies.push_back(CommandReply(aconf.autorange));
96 }else if(cmd.type == CMDtype::set){
97 setAutorange(cmd.val != 0);
98 }
99 break;
100 case LocalAnaloc_commands::pins:
101 if(cmd.type == CMDtype::get){
102 replies.push_back(CommandReply(numPins));
103 }else{
104 return CommandStatus::ERR;
105 }
106 break;
107
108 default:
109 return CommandStatus::NOT_FOUND;
110 }
111
112 return CommandStatus::OK;
113}
