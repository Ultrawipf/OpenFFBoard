149void Axis::restoreFlash(){
150 //NormalizedAxis::restoreFlash();
151 // read all constants
152 uint16_t value;
153 if (Flash_Read(flashAddrs.config, &value)){
154 this->conf = Axis::decodeConfFromInt(value);
155 }else{
156 pulseErrLed();
157 }
158
159 setDrvType(this->conf.drvtype);
160 setEncType(this->conf.enctype);
161
162// if (Flash_Read(flashAddrs.maxSpeed, &value)){
163// this->maxSpeedDegS = value;
164// }else{
165// pulseErrLed();
166// }
167//
168// if (Flash_Read(flashAddrs.maxAccel, &value)){
169// this->maxTorqueRateMS = value;
170// }else{
171// pulseErrLed();
172// }
173
174
175 uint16_t esval, power;
176 if(Flash_Read(flashAddrs.endstop, &esval)) {
177 fx_ratio_i = esval & 0xff;
178 endstopStrength = (esval >> 8) & 0xff;
179 }
180
181
182 if(Flash_Read(flashAddrs.power, &power)){
183 setPower(power);
184 }
185 uint16_t deg_t;
186 if(Flash_Read(flashAddrs.degrees, &deg_t)){
187 this->degreesOfRotation = deg_t & 0x7fff;
188 this->invertAxis = (deg_t >> 15) & 0x1;
189 setDegrees(degreesOfRotation);
190 }
191
192
193 uint16_t effects;
194 if(Flash_Read(flashAddrs.effects1, &effects)){
195 setIdleSpringStrength(effects & 0xff);
196 setDamperStrength((effects >> 8) & 0xff);
197 }
198
199 uint16_t ratio;
200 if(Flash_Read(flashAddrs.encoderRatio, &ratio)){
201 setGearRatio(ratio & 0xff, (ratio >> 8) & 0xff);
202 }
203
204 uint16_t filterStorage;
205 if (Flash_Read(flashAddrs.speedAccelFilter, &filterStorage))
206 {
207 uint8_t profile = filterStorage & 0xFF;
208 this->filterProfileId = profile;
209 speedFilter.setFc(filterSpeedCst[this->filterProfileId].freq / filter_f);
210 speedFilter.setQ(filterSpeedCst[this->filterProfileId].q / 100.0);
211 accelFilter.setFc(filterAccelCst[this->filterProfileId].freq / filter_f);
212 accelFilter.setQ(filterAccelCst[this->filterProfileId].q / 100.0);
213 }

217void Axis::saveFlash(){
218 //NormalizedAxis::saveFlash();
219 Flash_Write(flashAddrs.config, Axis::encodeConfToInt(this->conf));
220// Flash_Write(flashAddrs.maxSpeed, this->maxSpeedDegS);
221// Flash_Write(flashAddrs.maxAccel, (uint16_t)(this->maxTorqueRateMS));
222
223 Flash_Write(flashAddrs.endstop, fx_ratio_i | (endstopStrength << 8));
224 Flash_Write(flashAddrs.power, power);
225 Flash_Write(flashAddrs.degrees, (degreesOfRotation & 0x7fff) | (invertAxis << 15));
226 Flash_Write(flashAddrs.effects1, idlespringstrength | (damperIntensity << 8));
227 Flash_Write(flashAddrs.encoderRatio, gearRatio.numerator | (gearRatio.denominator << 8));
228
229 // save CF biquad
230 uint16_t filterStorage = (uint16_t)this->filterProfileId & 0xFF;
231 Flash_Write(flashAddrs.speedAccelFilter, filterStorage);
232}

265void Axis::prepareForUpdate(){
266 if (drv == nullptr){
267 pulseErrLed();
268 return;
269 }
270
271 //if (!drv->motorReady()) return;
272
273 float angle = getEncAngle(this->drv->getEncoder());
274
275 // Scale encoder value to set rotation range
276 // Update a change of range only when new range is within valid range
277 // if degree change, compute the SpeedScaler, it depends on degreesOfRotation
278 if (nextDegreesOfRotation != degreesOfRotation && abs(scaleEncValue(angle, nextDegreesOfRotation)) < 0x7fff){
279 degreesOfRotation = nextDegreesOfRotation;
280 }
281
282
283 // scaledEnc now gets inverted if necessary in updateMetrics
284 int32_t scaledEnc = scaleEncValue(angle, degreesOfRotation);
285
286 if (abs(scaledEnc) > 0xffff && drv->motorReady()){
287 // We are way off. Shut down
288 drv->stopMotor();
289 pulseErrLed();
290 if(!outOfBounds){
291 outOfBounds = true;
292 ErrorHandler::addError(outOfBoundsError);
293 }
294
295 }else if(abs(scaledEnc) <= 0x7fff) {
296 outOfBounds = false;
297 //ErrorHandler::clearError(outOfBoundsError);
298 }
299
300 this->updateMetrics(angle);
301
302}

565void Axis::resetMetrics(float new_pos= 0) { // pos is degrees
566 metric.current = metric_t();
567 metric.current.posDegrees = new_pos;
568 metric.current.pos = scaleEncValue(new_pos, degreesOfRotation);
569 metric.previous = metric_t();
570 // Reset filters
571 speedFilter.calcBiquad();
572 accelFilter.calcBiquad();
573}

578void Axis::updateMetrics(float new_pos) { // pos is degrees
579 // store old value for next metric's computing
580 metric.previous = metric.current;
581
582 metric.current.posDegrees = new_pos;
583 int32_t scaled_pos = scaleEncValue(new_pos, degreesOfRotation);
584 metric.current.pos = scaled_pos;
585
586
587 // compute speed and accel from raw instant speed normalized
588 float currentSpeed = (new_pos - metric.previous.posDegrees) * 1000.0; // deg/s
589 metric.current.speed = speedFilter.process(currentSpeed);
590 metric.current.accel = accelFilter.process((currentSpeed - _lastSpeed))* 1000.0; // deg/s/s
591 _lastSpeed = currentSpeed;
592
593 metric.current.torque = 0;
594}

621int16_t Axis::updateEndstop(){
622 int8_t clipdir = cliptest<int32_t,int32_t>(metric.current.pos, -0x7fff, 0x7fff);
623 if(clipdir == 0){
624 return 0;
625 }
626 float addtorque = clipdir*metric.current.posDegrees - (float)this->degreesOfRotation/2.0; // degress of rotation counts total range so multiply by 2
627 addtorque *= (float)endstopStrength * endstopGain; // Apply endstop gain for stiffness.
628 addtorque *= -clipdir;
629
630 return clip<int32_t,int32_t>(addtorque,-0x7fff,0x7fff);
631}

641bool Axis::updateTorque(int32_t* totalTorque) {
642
643 if(abs(effectTorque) >= 0x7fff){
644 pulseClipLed();
645 }
646
647 // Scale effect torque
648 int32_t torque = effectTorque; // Game effects
649 torque *= effect_margin_scaler;
650 torque += axisEffectTorque; // Independent effects
651 torque += updateEndstop();
652 torque *= torqueScaler; // Scale to power
653
654 // TODO speed and accel limiters
655 if(maxSpeedDegS > 0){
656
657 float torqueSign = torque > 0 ? 1 : -1; // Used to prevent metrics against the force to go into the limiter
658 // Speed. Mostly tuned...
659 spdlimiterAvg.addValue(metric.current.speed);
660 float speedreducer = (float)((spdlimiterAvg.getAverage()*torqueSign) - (float)maxSpeedDegS) * ((float)0x7FFF / maxSpeedDegS);
661 spdlimitreducerI = clip<float,int32_t>( spdlimitreducerI + ((speedreducer * 0.015) * torqueScaler),0,power);
662
663 // Accel limit. Not really useful. Maybe replace with torque slew rate limit?
664// float accreducer = (float)((metric.current.accel*torqueSign) - (float)maxAccelDegSS) * getAccelScalerNormalized();
665// acclimitreducerI = clip<float,int32_t>( acclimitreducerI + ((accreducer * 0.02) * torqueScaler),0,power);
666
667
668 // Only reduce torque. Don't invert it to prevent oscillation
669 float torqueReduction = spdlimitreducerI + speedreducer * 0.025;// accreducer * 0.025 + acclimitreducerI
670 if(torque > 0){
671 torqueReduction = clip<float,int32_t>(torqueReduction,0,torque);
672 }else{
673 torqueReduction = clip<float,int32_t>(-torqueReduction,torque,0);
674 }
675
676 torque -= torqueReduction;
677 }
678 // Torque slew rate limiter
679 if(maxTorqueRateMS > 0){
680 torque = clip<int32_t,int32_t>(torque, metric.previous.torque - maxTorqueRateMS,metric.previous.torque + maxTorqueRateMS);
681 }
682// if(torque - metric.previous.torque)
683 if(outOfBounds){
684 torque = 0;
685 }
686
687 // Torque calculated. Now sending to driver
688 torque = (invertAxis) ? -torque : torque;
689 metric.current.torque = torque;
690 torque = clip<int32_t, int32_t>(torque, -power, power);
691
692 bool torqueChanged = torque != metric.previous.torque;
693
694 if (abs(torque) == power){
695 pulseClipLed();
696 }
697
698 *totalTorque = torque;
699 return (torqueChanged);
700}

717CommandStatus Axis::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
718
719 switch(static_cast<Axis_commands>(cmd.cmdId)){
720
721 case Axis_commands::power:
722 if (cmd.type == CMDtype::get)
723 {
724 replies.emplace_back(this->power);
725 }
726 else if (cmd.type == CMDtype::set)
727 {
728 setPower(cmd.val);
729 }
730 break;
731
732 case Axis_commands::degrees:
733 handleGetSetFunc(cmd, replies, degreesOfRotation, &Axis::setDegrees,this);
734// if (cmd.type == CMDtype::get)
735// {
736// replies.emplace_back(degreesOfRotation);
737// }
738// else if (cmd.type == CMDtype::set)
739// {
740// setDegrees(cmd.val);
741// }
742 break;
743
744 case Axis_commands::esgain:
745 handleGetSet(cmd, replies, this->endstopStrength);
746 break;
747
748 case Axis_commands::zeroenc:
749 this->setPos(0);
750 break;
751
752 case Axis_commands::invert:
753 if (cmd.type == CMDtype::get)
754 {
755 replies.emplace_back(invertAxis ? 1 : 0);
756 }
757 else if (cmd.type == CMDtype::set)
758 {
759 invertAxis = cmd.val >= 1 ? true : false;
760 resetMetrics(-metric.current.posDegrees);
761 }
762 break;
763
764 case Axis_commands::idlespring:
765 if (cmd.type == CMDtype::get)
766 {
767 replies.emplace_back(idlespringstrength);
768 }
769 else if (cmd.type == CMDtype::set)
770 {
771 setIdleSpringStrength(cmd.val);
772 }
773 break;
774
775 case Axis_commands::axisdamper:
776 if (cmd.type == CMDtype::get)
777 {
778 replies.emplace_back(damperIntensity);
779 }
